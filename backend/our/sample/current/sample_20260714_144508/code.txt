def fjsp_solver(n_jobs, n_machines, durations):
    # Initialize schedule and tracking structures
    schedule = [[None] * len(durations[j]) for j in range(n_jobs)]
    machine_ready_time = [0] * n_machines
    machine_load = [0] * n_machines  # total processing time assigned to each machine
    job_ready_time = [0] * n_jobs
    next_op = [0] * n_jobs
    remaining = sum(len(job) for job in durations)

    while remaining > 0:
        # Find jobs with remaining operations
        ready_jobs = [j for j in range(n_jobs) if next_op[j] < len(durations[j])]
        # Choose job with earliest ready time (tie broken by job index for stability)
        job_idx = min(ready_jobs, key=lambda j: (job_ready_time[j], j))
        op_idx = next_op[job_idx]
        candidates = durations[job_idx][op_idx]

        # Machine selection: fastest (shortest processing time) then least loaded
        # If tie, lowest machine_id
        machine_id, processing_time = min(
            candidates,
            key=lambda pair: (pair[1], machine_load[pair[0]], pair[0])
        )

        # Compute start time
        start = max(machine_ready_time[machine_id], job_ready_time[job_idx])
        end = start + processing_time

        # Record assignment
        schedule[job_idx][op_idx] = (machine_id, start, processing_time)

        # Update state
        machine_ready_time[machine_id] = end
        machine_load[machine_id] += processing_time
        job_ready_time[job_idx] = end
        next_op[job_idx] += 1
        remaining -= 1

    return schedule
