def fjsp_solver(n_jobs, n_machines, durations):
    # ---------- helper data ----------
    n_ops = [len(job) for job in durations]
    total_ops = sum(n_ops)
    # Precompute remaining minimum work for each job from each operation
    remaining_min_work = [[0] * (n_ops[j] + 1) for j in range(n_jobs)]
    for j in range(n_jobs):
        for op in range(n_ops[j] - 1, -1, -1):
            best = min(durations[j][op], key=lambda x: x[1])[1]
            remaining_min_work[j][op] = best + remaining_min_work[j][op + 1]

    # ---------- state ----------
    schedule = [[None] * n_ops[j] for j in range(n_jobs)]
    machine_ready_time = [0] * n_machines
    job_ready_time = [0] * n_jobs
    next_op = [0] * n_jobs
    remaining = total_ops

    ALPHA = 0.85  # keep same hybrid factor

    while remaining > 0:
        ready_jobs = [j for j in range(n_jobs) if next_op[j] < n_ops[j]]

        # ---------- job selection (mutated weights) ----------
        def job_priority(j_idx):
            op_idx = next_op[j_idx]
            candidates = durations[j_idx][op_idx]
            min_start = min(
                max(machine_ready_time[m], job_ready_time[j_idx]) for m, _ in candidates
            )
            # base priority
            base_prio = min_start
            # elite priority with mutated coefficients to target worst gaps
            optimistic_completion = min_start + remaining_min_work[j_idx][op_idx]
            avg_machine_load = (
                sum(machine_ready_time[m] for m, _ in candidates) / len(candidates)
                if candidates else 0.0
            )
            # changed: -0.45 instead of -0.3, +0.15 instead of 0.2, +0.3 instead of 0.25
            elite_prio = (
                min_start
                - 0.45 * remaining_min_work[j_idx][op_idx]
                + 0.15 * avg_machine_load
                + 0.3 * optimistic_completion
            )
            hybrid = ALPHA * elite_prio + (1 - ALPHA) * base_prio
            return (hybrid, j_idx)

        job_idx = min(ready_jobs, key=job_priority)
        op_idx = next_op[job_idx]
        candidates = durations[job_idx][op_idx]

        # ---------- machine selection (stable) ----------
        def machine_priority(candidate):
            m_id, p_time = candidate
            start_time = max(machine_ready_time[m_id], job_ready_time[job_idx])
            completion = start_time + p_time
            # new machine priority (unchanged from elite)
            machine_load = machine_ready_time[m_id] + p_time
            job_rem = remaining_min_work[job_idx][op_idx]
            new_prio = completion + 0.4 * machine_load + 0.3 * job_rem - 0.2 * p_time
            # base machine priority
            base_prio = start_time + p_time
            hybrid = ALPHA * new_prio + (1 - ALPHA) * base_prio
            return (hybrid, machine_ready_time[m_id], p_time, m_id)

        machine_id, processing_time = min(candidates, key=machine_priority)

        start_time = max(machine_ready_time[machine_id], job_ready_time[job_idx])
        schedule[job_idx][op_idx] = (machine_id, start_time, processing_time)
        end_time = start_time + processing_time
        machine_ready_time[machine_id] = end_time
        job_ready_time[job_idx] = end_time
        next_op[job_idx] += 1
        remaining -= 1

    return schedule
