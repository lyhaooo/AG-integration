def fjsp_solver(n_jobs, n_machines, durations):
    # Precompute remaining minimal work for each job from each operation
    n_ops = [len(job) for job in durations]
    remaining_min_work = [[0] * (n_ops[j] + 1) for j in range(n_jobs)]
    for j in range(n_jobs):
        for op in range(n_ops[j] - 1, -1, -1):
            best = min(durations[j][op], key=lambda x: x[1])[1]
            remaining_min_work[j][op] = best + remaining_min_work[j][op + 1]

    # State variables
    machine_ready_time = [0] * n_machines
    job_ready_time = [0] * n_jobs
    next_op = [0] * n_jobs
    total_ops = sum(n_ops)
    schedule = [[None] * n_ops[j] for j in range(n_jobs)]

    # Hybridization parameter (mix elite and historical lesson)
    ALPHA = 0.70   # weight on elite strategy

    # ----- Priority functions (elite) -----
    def elite_job_priority(job_idx):
        op_idx = next_op[job_idx]
        candidates = durations[job_idx][op_idx]
        min_start = min(
            max(machine_ready_time[m], job_ready_time[job_idx])
            for m, _ in candidates
        )
        optimistic_completion = min_start + remaining_min_work[job_idx][op_idx]
        avg_machine_load = sum(machine_ready_time[m] for m, _ in candidates) / len(candidates)
        return min_start - 0.3 * remaining_min_work[job_idx][op_idx] + 0.2 * avg_machine_load + 0.25 * optimistic_completion

    def elite_machine_priority(job_idx, machine_id, processing_time):
        op_idx = next_op[job_idx]
        start_time = max(machine_ready_time[machine_id], job_ready_time[job_idx])
        completion = start_time + processing_time
        # total load on this machine if we take this operation
        machine_load = machine_ready_time[machine_id] + processing_time
        # remaining work of the job from this operation onward
        job_rem = remaining_min_work[job_idx][op_idx]
        # penalty for high machine load, reward for processing jobs with large remaining work
        return completion + 0.4 * machine_load + 0.3 * job_rem - 0.2 * processing_time

    # ----- Priority functions (historical lesson – #3, improved from #1) -----
    # This lesson emphasized completion time and remaining work more heavily.
    def historical_job_priority(job_idx):
        op_idx = next_op[job_idx]
        candidates = durations[job_idx][op_idx]
        min_start = min(
            max(machine_ready_time[m], job_ready_time[job_idx])
            for m, _ in candidates
        )
        optimistic_completion = min_start + remaining_min_work[job_idx][op_idx]
        avg_machine_load = sum(machine_ready_time[m] for m, _ in candidates) / len(candidates) if candidates else 0
        # Focus on urgency (remaining work) and completion time
        return min_start + 0.5 * remaining_min_work[job_idx][op_idx] - 0.1 * avg_machine_load + 0.2 * optimistic_completion

    def historical_machine_priority(job_idx, machine_id, processing_time):
        op_idx = next_op[job_idx]
        start_time = max(machine_ready_time[machine_id], job_ready_time[job_idx])
        completion = start_time + processing_time
        machine_load = machine_ready_time[machine_id] + processing_time
        job_rem = remaining_min_work[job_idx][op_idx]
        # Different balance: penalize machine load more, reward remaining work
        return completion + 0.5 * machine_load + 0.25 * job_rem - 0.1 * processing_time

    # ----- Hybrid priority functions -----
    def job_priority(job_idx):
        hybrid = ALPHA * elite_job_priority(job_idx) + (1 - ALPHA) * historical_job_priority(job_idx)
        return (hybrid, job_idx)

    def machine_priority(job_idx, machine_id, processing_time):
        hybrid = ALPHA * elite_machine_priority(job_idx, machine_id, processing_time) + \
                 (1 - ALPHA) * historical_machine_priority(job_idx, machine_id, processing_time)
        # Tie‑breaker: earlier machine ready time, shorter processing, lower machine id
        return (hybrid, machine_ready_time[machine_id], processing_time, machine_id)

    # ----- Constructive scheduler -----
    remaining = total_ops
    while remaining > 0:
        ready_jobs = [j for j in range(n_jobs) if next_op[j] < n_ops[j]]
        # Select job
        job_idx = min(ready_jobs, key=job_priority)
        op_idx = next_op[job_idx]
        candidates = durations[job_idx][op_idx]

        # Select machine
        machine_id, processing_time = min(candidates, key=lambda pair: machine_priority(job_idx, pair[0], pair[1]))

        start_time = max(machine_ready_time[machine_id], job_ready_time[job_idx])
        schedule[job_idx][op_idx] = (machine_id, start_time, processing_time)

        end_time = start_time + processing_time
        machine_ready_time[machine_id] = end_time
        job_ready_time[job_idx] = end_time
        next_op[job_idx] += 1
        remaining -= 1

    return schedule
