def fjsp_solver(n_jobs, n_machines, durations):
    # Precompute total remaining work per job (using minimum processing times)
    total_remaining = [0] * n_jobs
    for j in range(n_jobs):
        total = 0
        for op_candidates in durations[j]:
            total += min(p for _, p in op_candidates)
        total_remaining[j] = total

    schedule = [[None] * len(durations[j]) for j in range(n_jobs)]
    machine_ready_time = [0] * n_machines
    job_ready_time = [0] * n_jobs
    next_op = [0] * n_jobs
    unfinished = [len(durations[j]) for j in range(n_jobs)]

    while any(u > 0 for u in unfinished):
        # Find ready jobs (jobs with remaining operations)
        ready_jobs = [j for j in range(n_jobs) if next_op[j] < len(durations[j])]

        # Job selection: prefer fewer eligible machines, then larger remaining work
        def job_priority(j):
            op_idx = next_op[j]
            num_machines = len(durations[j][op_idx])
            # Remaining work from current operation onward
            # We can compute it quickly by subtracting already processed part from total_remaining[j]
            # But careful: total_remaining[j] is based on min times; we need the same metric.
            # Simpler: sum min processing times from current op to end.
            # Because total_remaining[j] was precomputed for all ops, we can subtract sum of mins for already completed ops.
            # Here we compute it on the fly for clarity.
            rem = 0
            for k in range(op_idx, len(durations[j])):
                rem += min(p for _, p in durations[j][k])
            return (num_machines, -rem, j)  # j for tie-breaking

        job_idx = min(ready_jobs, key=job_priority)

        op_idx = next_op[job_idx]
        candidates = durations[job_idx][op_idx]

        # Machine selection: choose candidate minimizing completion time
        machine_id, processing_time = min(
            candidates,
            key=lambda pair: (max(machine_ready_time[pair[0]], job_ready_time[job_idx]) + pair[1], pair[1], pair[0])
        )

        start_time = max(machine_ready_time[machine_id], job_ready_time[job_idx])
        schedule[job_idx][op_idx] = (machine_id, start_time, processing_time)
        end_time = start_time + processing_time
        machine_ready_time[machine_id] = end_time
        job_ready_time[job_idx] = end_time
        next_op[job_idx] += 1

    return schedule
