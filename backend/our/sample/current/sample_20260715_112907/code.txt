def fjsp_solver(n_jobs, n_machines, durations):
    # Precompute bottleneck scores: number of operations each machine appears in
    bottleneck_score = [0] * n_machines
    for job_ops in durations:
        for op_candidates in job_ops:
            for machine, _ in op_candidates:
                bottleneck_score[machine] += 1

    # Penalty factor - small to avoid completely ignoring good bottleneck options
    penalty = 0.5

    # Initialize data structures
    schedule = [[None] * len(durations[j]) for j in range(n_jobs)]
    machine_ready_time = [0] * n_machines
    job_ready_time = [0] * n_jobs
    next_op = [0] * n_jobs
    remaining = sum(len(job) for job in durations)

    while remaining > 0:
        # Select job with earliest ready time among those with remaining operations
        ready_jobs = [j for j in range(n_jobs) if next_op[j] < len(durations[j])]
        job_idx = min(ready_jobs, key=lambda j: job_ready_time[j])

        op_idx = next_op[job_idx]
        candidates = durations[job_idx][op_idx]

        # Choose candidate machine using a key that penalizes bottleneck machines
        def candidate_key(pair):
            m, dur = pair
            earliest_start = max(machine_ready_time[m], job_ready_time[job_idx])
            finish = earliest_start + dur
            penalized_finish = finish + penalty * bottleneck_score[m]
            return (penalized_finish, dur, m)

        machine_id, processing_time = min(candidates, key=candidate_key)

        start_time = max(machine_ready_time[machine_id], job_ready_time[job_idx])
        schedule[job_idx][op_idx] = (machine_id, start_time, processing_time)
        end_time = start_time + processing_time

        machine_ready_time[machine_id] = end_time
        job_ready_time[job_idx] = end_time
        next_op[job_idx] += 1
        remaining -= 1

    return schedule
