def fjsp_solver(n_jobs, n_machines, durations):
    # Precompute remaining minimum work for each job from each operation
    n_ops = [len(job) for job in durations]
    remaining_min_work = [[0] * (n_ops[j] + 1) for j in range(n_jobs)]
    for j in range(n_jobs):
        for op in range(n_ops[j] - 1, -1, -1):
            best = min(durations[j][op], key=lambda x: x[1])[1]
            remaining_min_work[j][op] = best + remaining_min_work[j][op + 1]

    # Initialize state
    schedule = [[None] * n_ops[j] for j in range(n_jobs)]
    machine_ready_time = [0] * n_machines
    job_ready_time = [0] * n_jobs
    next_op = [0] * n_jobs
    remaining = sum(n_ops)

    # Weights for job priority – mutated to target worst instance gaps
    ALPHA = 0.85

    # Job priority function (modified weights)
    def job_priority(job_idx):
        op_idx = next_op[job_idx]
        candidates = durations[job_idx][op_idx]
        min_start = min(
            max(machine_ready_time[m], job_ready_time[job_idx])
            for m, _ in candidates
        )
        # Base priority: earliest start
        base = min_start
        # Elite-inspired priority with adjusted coefficients
        optimistic = min_start + remaining_min_work[job_idx][op_idx]
        avg_load = sum(machine_ready_time[m] for m, _ in candidates) / len(candidates)
        elite = (min_start
                 - 0.4 * remaining_min_work[job_idx][op_idx]   # increased weight on remaining work
                 + 0.1 * avg_load
                 + 0.3 * optimistic)
        hybrid = ALPHA * elite + (1 - ALPHA) * base
        return (hybrid, job_idx)

    # Machine priority function (kept stable as in elite)
    def base_machine_priority(job_idx, machine_id, processing_time):
        start = max(machine_ready_time[machine_id], job_ready_time[job_idx])
        return start + processing_time

    def new_machine_priority(job_idx, machine_id, processing_time):
        op_idx = next_op[job_idx]
        start = max(machine_ready_time[machine_id], job_ready_time[job_idx])
        completion = start + processing_time
        load = machine_ready_time[machine_id] + processing_time
        job_rem = remaining_min_work[job_idx][op_idx]
        return completion + 0.4 * load + 0.3 * job_rem - 0.2 * processing_time

    def machine_priority(job_idx, machine_id, processing_time):
        hybrid = ALPHA * new_machine_priority(job_idx, machine_id, processing_time) + \
                 (1 - ALPHA) * base_machine_priority(job_idx, machine_id, processing_time)
        return (hybrid, machine_ready_time[machine_id], processing_time, machine_id)

    # Main scheduling loop
    while remaining > 0:
        ready_jobs = [j for j in range(n_jobs) if next_op[j] < n_ops[j]]
        # Prioritize jobs using the mutated weights
        job_idx = min(ready_jobs, key=job_priority)
        op_idx = next_op[job_idx]
        candidates = durations[job_idx][op_idx]
        # Select machine using the stable machine priority
        machine_id, processing_time = min(
            candidates,
            key=lambda pair: machine_priority(job_idx, pair[0], pair[1])
        )
        start_time = max(machine_ready_time[machine_id], job_ready_time[job_idx])
        schedule[job_idx][op_idx] = (machine_id, start_time, processing_time)
        end_time = start_time + processing_time
        machine_ready_time[machine_id] = end_time
        job_ready_time[job_idx] = end_time
        next_op[job_idx] += 1
        remaining -= 1

    return schedule
