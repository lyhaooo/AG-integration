def fjsp_solver(n_jobs, n_machines, durations):
    # Initialize schedule matrix
    schedule = [[None] * len(durations[j]) for j in range(n_jobs)]
    machine_ready_time = [0] * n_machines
    job_ready_time = [0] * n_jobs
    next_op = [0] * n_jobs
    remaining = sum(len(job) for job in durations)

    # Precompute total minimum processing time for each job (all operations)
    job_total_min = [0] * n_jobs
    for j in range(n_jobs):
        total = 0
        for candidates in durations[j]:
            min_pt = min(candidates, key=lambda pair: pair[1])[1]
            total += min_pt
        job_total_min[j] = total

    while remaining > 0:
        # Determine ready jobs (jobs that still have unscheduled operations)
        ready_jobs = [j for j in range(n_jobs) if next_op[j] < len(durations[j])]

        # Variant 1: Select job with largest total remaining minimum processing time.
        # Ties broken by job index (earliest first) for determinism.
        job_idx = max(ready_jobs, key=lambda j: (job_total_min[j], j))

        op_idx = next_op[job_idx]
        candidates = durations[job_idx][op_idx]

        # Choose machine: earliest possible finish time, break ties by processing time then machine id
        machine_id, processing_time = min(
            candidates,
            key=lambda pair: (
                max(machine_ready_time[pair[0]], job_ready_time[job_idx]) + pair[1],
                pair[1],
                pair[0]
            ),
        )
        start_time = max(machine_ready_time[machine_id], job_ready_time[job_idx])
        schedule[job_idx][op_idx] = (machine_id, start_time, processing_time)
        end_time = start_time + processing_time
        machine_ready_time[machine_id] = end_time
        job_ready_time[job_idx] = end_time

        # Update remaining work for this job: subtract the min processing time of the operation just scheduled
        job_total_min[job_idx] -= min(candidates, key=lambda pair: pair[1])[1]

        next_op[job_idx] += 1
        remaining -= 1

    return schedule
