def fjsp_solver(n_jobs, n_machines, durations):
    # Precompute remaining minimum work for each job from each operation
    n_ops = [len(job) for job in durations]
    remaining_min_work = [[0] * (n_ops[j] + 1) for j in range(n_jobs)]
    for j in range(n_jobs):
        for op in range(n_ops[j] - 1, -1, -1):
            best = min(durations[j][op], key=lambda x: x[1])[1]
            remaining_min_work[j][op] = best + remaining_min_work[j][op + 1]

    ALPHA = 0.85   # same as elite

    # Base job priority (earliest possible start)
    def base_job_priority(job_idx, mrt, jrt, nxt):
        op_idx = nxt[job_idx]
        candidates = durations[job_idx][op_idx]
        min_start = min(max(mrt[m], jrt[job_idx]) for m, _ in candidates)
        return min_start

    # Elite job priority (unchanged)
    def elite_job_priority(job_idx, mrt, jrt, nxt):
        op_idx = nxt[job_idx]
        candidates = durations[job_idx][op_idx]
        min_start = min(max(mrt[m], jrt[job_idx]) for m, _ in candidates)
        optimistic_completion = min_start + remaining_min_work[job_idx][op_idx]
        avg_machine_load = sum(mrt[m] for m, _ in candidates) / len(candidates) if candidates else 0
        return min_start - 0.3 * remaining_min_work[job_idx][op_idx] + 0.2 * avg_machine_load + 0.25 * optimistic_completion

    # Job selection: hybrid of base and elite
    def job_priority(job_idx, mrt, jrt, nxt):
        hybrid = ALPHA * elite_job_priority(job_idx, mrt, jrt, nxt) + (1 - ALPHA) * base_job_priority(job_idx, mrt, jrt, nxt)
        return (hybrid, job_idx)

    # --- Mutated machine priority to reduce bottleneck congestion ---
    # New machine scoring: heavier penalty on machine load, reduced influence of job remaining and processing time
    def mutated_new_machine_priority(job_idx, machine_id, processing_time, mrt, jrt, nxt):
        op_idx = nxt[job_idx]
        start = max(mrt[machine_id], jrt[job_idx])
        completion = start + processing_time
        machine_load = mrt[machine_id] + processing_time
        job_rem = remaining_min_work[job_idx][op_idx]
        # Increased weight on machine_load (0.7 vs earlier 0.4), reduced job_rem (0.2) and processing_time weight (-0.1)
        score = completion + 0.7 * machine_load + 0.2 * job_rem - 0.1 * processing_time
        return score

    # Base machine priority (earliest completion)
    def base_machine_priority(job_idx, machine_id, processing_time, mrt, jrt, nxt):
        start = max(mrt[machine_id], jrt[job_idx])
        return start + processing_time

    # Machine selection: hybrid + tie‑breaker on machine_load to avoid busy machines
    def machine_priority(job_idx, machine_id, processing_time, mrt, jrt, nxt):
        hybrid = ALPHA * mutated_new_machine_priority(job_idx, machine_id, processing_time, mrt, jrt, nxt) + \
                 (1 - ALPHA) * base_machine_priority(job_idx, machine_id, processing_time, mrt, jrt, nxt)
        # Tie‑breaker: machine_load (lower is better) then machine_id
        machine_load = mrt[machine_id] + processing_time
        return (hybrid, machine_load, machine_id)

    # Constructive scheduler
    schedule = [[None] * len(durations[j]) for j in range(n_jobs)]
    mrt = [0] * n_machines
    jrt = [0] * n_jobs
    nxt = [0] * n_jobs
    remaining = sum(len(job) for job in durations)

    while remaining > 0:
        ready_jobs = [j for j in range(n_jobs) if nxt[j] < len(durations[j])]
        # Choose job using the hybrid priority
        job_idx = min(ready_jobs, key=lambda j: job_priority(j, mrt, jrt, nxt))
        op_idx = nxt[job_idx]
        candidates = durations[job_idx][op_idx]

        # Choose machine using the mutated priority
        machine_id, processing_time = min(
            candidates,
            key=lambda pair: machine_priority(job_idx, pair[0], pair[1], mrt, jrt, nxt)
        )

        start_time = max(mrt[machine_id], jrt[job_idx])
        schedule[job_idx][op_idx] = (machine_id, start_time, processing_time)
        end_time = start_time + processing_time
        mrt[machine_id] = end_time
        jrt[job_idx] = end_time
        nxt[job_idx] += 1
        remaining -= 1

    return schedule
