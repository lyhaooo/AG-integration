def fjsp_solver(n_jobs, n_machines, durations):
    """
    Flexible Job Shop Scheduling heuristic.
    Job selection: choose the ready job whose current operation has the shortest
    best processing time (minimum among its candidate machines).
    Machine selection: choose the candidate that minimises the completion time
    (max of machine ready time and job ready time + processing time), with ties
    broken by processing time and machine id.
    """
    schedule = [[None] * len(durations[j]) for j in range(n_jobs)]
    machine_ready_time = [0] * n_machines
    job_ready_time = [0] * n_jobs
    next_op = [0] * n_jobs
    remaining = sum(len(job) for job in durations)

    while remaining > 0:
        # Jobs that still have unscheduled operations
        ready_jobs = [j for j in range(n_jobs) if next_op[j] < len(durations[j])]

        # Variant 2: choose ready job whose current operation has the shortest best processing time
        job_idx = min(
            ready_jobs,
            key=lambda j: min(p[1] for p in durations[j][next_op[j]])
        )

        op_idx = next_op[job_idx]
        candidates = durations[job_idx][op_idx]

        # Machine selection: earliest completion time, tie-break by duration then machine id
        machine_id, processing_time = min(
            candidates,
            key=lambda pair: (
                max(machine_ready_time[pair[0]], job_ready_time[job_idx]) + pair[1],
                pair[1],
                pair[0]
            )
        )

        start_time = max(machine_ready_time[machine_id], job_ready_time[job_idx])
        schedule[job_idx][op_idx] = (machine_id, start_time, processing_time)
        end_time = start_time + processing_time

        machine_ready_time[machine_id] = end_time
        job_ready_time[job_idx] = end_time
        next_op[job_idx] += 1
        remaining -= 1

    return schedule
