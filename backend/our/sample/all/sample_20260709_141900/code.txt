def fjsp_solver(n_jobs, n_machines, durations):
    # Precompute minimum processing time for each operation (for remaining work estimation)
    min_duration = []
    for job in durations:
        job_min = []
        for op_candidates in job:
            job_min.append(min(p for _, p in op_candidates))
        min_duration.append(job_min)

    schedule = [[None] * len(durations[j]) for j in range(n_jobs)]
    machine_ready_time = [0] * n_machines
    job_ready_time = [0] * n_jobs
    next_op = [0] * n_jobs
    remaining = sum(len(job) for job in durations)

    while remaining > 0:
        # Identify ready jobs (jobs with unscheduled operations)
        ready_jobs = [j for j in range(n_jobs) if next_op[j] < len(durations[j])]

        # Job selection: largest remaining work
        # Compute remaining work for each ready job
        def remaining_work(j):
            ops = durations[j]
            start = next_op[j]
            # sum of min durations from start to end
            return sum(min_duration[j][op] for op in range(start, len(ops)))

        # Select job with maximum remaining work, tie-breaking by job index
        job_idx = max(ready_jobs, key=lambda j: (remaining_work(j), -j))  # larger remaining work, smaller j (negative to invert)

        op_idx = next_op[job_idx]
        candidates = durations[job_idx][op_idx]

        # Machine selection: load-aware earliest completion
        # For each candidate, compute completion time and machine load tie-breaker
        def machine_score(pair):
            m, p = pair
            comp = max(machine_ready_time[m], job_ready_time[job_idx]) + p
            # primary key: completion time
            # secondary key: current machine load (machine_ready_time) to favor less loaded machines
            return (comp, machine_ready_time[m])

        machine_id, processing_time = min(candidates, key=machine_score)

        start_time = max(machine_ready_time[machine_id], job_ready_time[job_idx])
        schedule[job_idx][op_idx] = (machine_id, start_time, processing_time)

        end_time = start_time + processing_time
        machine_ready_time[machine_id] = end_time
        job_ready_time[job_idx] = end_time
        next_op[job_idx] += 1
        remaining -= 1

    return schedule
