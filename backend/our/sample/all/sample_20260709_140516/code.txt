def fjsp_solver(n_jobs, n_machines, durations):
    # compute total minimum processing time for each job once (static)
    total_min_time = []
    for j in range(n_jobs):
        total = 0
        for ops in durations[j]:
            total += min(p[1] for p in ops)
        total_min_time.append(total)

    schedule = [[None] * len(durations[j]) for j in range(n_jobs)]
    machine_ready_time = [0] * n_machines
    job_ready_time = [0] * n_jobs
    next_op = [0] * n_jobs
    remaining = sum(len(job) for job in durations)

    while remaining > 0:
        # ready jobs: jobs that still have unscheduled operations
        ready_jobs = [j for j in range(n_jobs) if next_op[j] < len(durations[j])]

        # compute total remaining minimum processing time for each ready job
        # (we can recompute or update dynamically; here recompute for clarity)
        job_priority = []
        for j in ready_jobs:
            rem_min = 0
            for op_idx in range(next_op[j], len(durations[j])):
                rem_min += min(p[1] for p in durations[j][op_idx])
            job_priority.append((j, rem_min))

        # select job with largest remaining minimum processing time
        # tie-break: job_ready_time, then job index
        job_idx = max(job_priority, key=lambda x: (x[1], job_ready_time[x[0]], -x[0]))[0]

        op_idx = next_op[job_idx]
        candidates = durations[job_idx][op_idx]
        # choose machine: earliest start time, then smallest processing time, then machine id
        machine_id, processing_time = min(
            candidates,
            key=lambda pair: (
                max(machine_ready_time[pair[0]], job_ready_time[job_idx]) + pair[1],
                pair[1],
                pair[0],
            ),
        )
        start_time = max(machine_ready_time[machine_id], job_ready_time[job_idx])
        schedule[job_idx][op_idx] = (machine_id, start_time, processing_time)
        end_time = start_time + processing_time
        machine_ready_time[machine_id] = end_time
        job_ready_time[job_idx] = end_time
        next_op[job_idx] += 1
        remaining -= 1

    return schedule
