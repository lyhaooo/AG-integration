def fjsp_solver(n_jobs, n_machines, durations):
    schedule = [[None] * len(durations[j]) for j in range(n_jobs)]
    machine_ready = [0] * n_machines
    job_ready = [0] * n_jobs
    next_op = [0] * n_jobs
    remaining = sum(len(job) for job in durations)

    while remaining > 0:
        ready = [j for j in range(n_jobs) if next_op[j] < len(durations[j])]
        # job selection: earliest job ready, tie-break by longer remaining job
        job_idx = min(ready, key=lambda j: (job_ready[j], -len(durations[j])))
        op_idx = next_op[job_idx]
        candidates = durations[job_idx][op_idx]

        best_metric = float('inf')
        best_candidate = None
        for machine, proc_time in candidates:
            start = max(machine_ready[machine], job_ready[job_idx])
            current_end = start + proc_time

            if op_idx + 1 < len(durations[job_idx]):
                next_candidates = durations[job_idx][op_idx + 1]
                best_next_end = min(
                    max(machine_ready[m2], current_end) + p2
                    for m2, p2 in next_candidates
                )
                metric = max(current_end, best_next_end)
            else:
                metric = current_end

            # tie-breaking: smaller processing time, then lower machine id
            if (metric < best_metric or
                (metric == best_metric and proc_time < best_candidate[1]) or
                (metric == best_metric and proc_time == best_candidate[1] and machine < best_candidate[0])):
                best_metric = metric
                best_candidate = (machine, proc_time)

        machine_id, processing_time = best_candidate
        start_time = max(machine_ready[machine_id], job_ready[job_idx])
        schedule[job_idx][op_idx] = (machine_id, start_time, processing_time)
        end_time = start_time + processing_time
        machine_ready[machine_id] = end_time
        job_ready[job_idx] = end_time
        next_op[job_idx] += 1
        remaining -= 1

    return schedule
