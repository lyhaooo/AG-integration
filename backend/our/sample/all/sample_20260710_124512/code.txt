import random

def fjsp_solver(n_jobs, n_machines, durations):
    # Precompute remaining minimum work for each operation (optimistic)
    n_ops = [len(job) for job in durations]
    total_ops = sum(n_ops)
    remaining_min_work = [[0] * (n_ops[j] + 1) for j in range(n_jobs)]
    for j in range(n_jobs):
        for op in range(n_ops[j] - 1, -1, -1):
            best = min(durations[j][op], key=lambda x: x[1])[1]
            remaining_min_work[j][op] = best + remaining_min_work[j][op + 1]

    # ---------- priority functions (modified version) ----------
    def job_priority(job_idx, mrt, jrt, nxt):
        op_idx = nxt[job_idx]
        candidates = durations[job_idx][op_idx]
        min_start = min(max(mrt[m], jrt[job_idx]) for m, _ in candidates)
        # weight: start time (0.4), remaining work (0.4), job ready time (0.2)
        return (0.4 * min_start + 0.4 * remaining_min_work[job_idx][op_idx] + 0.2 * jrt[job_idx], job_idx)

    def machine_priority(job_idx, machine_id, processing_time, mrt, jrt, nxt):
        start_time = max(mrt[machine_id], jrt[job_idx])
        completion = start_time + processing_time
        machine_load = mrt[machine_id] + processing_time
        job_rem = remaining_min_work[job_idx][nxt[job_idx]]
        # score: completion (0.5), machine load (0.3), job remaining (0.2)
        # we want to minimize, so lower is better
        return (0.5 * completion + 0.3 * machine_load + 0.2 * job_rem, machine_ready_time[machine_id], processing_time, machine_id)

    # ---------- constructive scheduler ----------
    def construct_schedule(forced_machine=None):
        """
        Build a schedule using the greedy rule with modified priorities.
        forced_machine: dict {(job, op): machine_id} to fix a machine for that operation.
        Returns (schedule, makespan) or None if forced machine unavailable.
        """
        sched = [[None] * n_ops[j] for j in range(n_jobs)]
        mrt = [0] * n_machines
        jrt = [0] * n_jobs
        nxt = [0] * n_jobs
        rem = total_ops

        while rem > 0:
            # Find ready jobs
            ready = [j for j in range(n_jobs) if nxt[j] < n_ops[j]]
            if not ready:
                break
            # Use job priority to select job
            best_job = min(ready, key=lambda j: job_priority(j, mrt, jrt, nxt))
            job_idx = best_job
            op_idx = nxt[job_idx]
            candidates = durations[job_idx][op_idx]

            # If forced machine, filter candidates
            if forced_machine and (job_idx, op_idx) in forced_machine:
                forced_m = forced_machine[(job_idx, op_idx)]
                # Check if forced machine is a valid candidate
                valid = [c for c in candidates if c[0] == forced_m]
                if not valid:
                    return None
                machine_id, processing_time = valid[0]
            else:
                # Use machine priority to select machine
                machine_id, processing_time = min(candidates, key=lambda c: machine_priority(job_idx, c[0], c[1], mrt, jrt, nxt))
                # Actually we need to return the full tuple but min works with key function
                # Better: get the best machine by score
                best_candidate = min(candidates, key=lambda c: (0.5 * max(mrt[c[0]], jrt[job_idx]) + c[1] + 0.3 * (mrt[c[0]] + c[1]) + 0.2 * remaining_min_work[job_idx][op_idx], mrt[c[0]], c[1], c[0]))
                machine_id, processing_time = best_candidate

            start_time = max(mrt[machine_id], jrt[job_idx])
            sched[job_idx][op_idx] = (machine_id, start_time, processing_time)
            end_time = start_time + processing_time
            mrt[machine_id] = end_time
            jrt[job_idx] = end_time
            nxt[job_idx] += 1
            rem -= 1

        # Check completeness
        for j in range(n_jobs):
            for op in range(n_ops[j]):
                if sched[j][op] is None:
                    return None
        makespan = max(jrt)
        return sched, makespan

    # ---------- build initial schedule ----------
    initial = construct_schedule()
    if initial is None:
        # fallback: use skeleton simple heuristic (should not happen)
        schedule = [[None] * n_ops[j] for j in range(n_jobs)]
        machine_ready_time = [0] * n_machines
        job_ready_time = [0] * n_jobs
        next_op = [0] * n_jobs
        remaining = total_ops
        while remaining > 0:
            ready = [j for j in range(n_jobs) if next_op[j] < n_ops[j]]
            job_idx = min(ready, key=lambda j: job_ready_time[j])
            op_idx = next_op[job_idx]
            candidates = durations[job_idx][op_idx]
            best = min(candidates, key=lambda c: (max(machine_ready_time[c[0]], job_ready_time[job_idx]) + c[1], c[1], c[0]))
            machine_id, processing_time = best
            start = max(machine_ready_time[machine_id], job_ready_time[job_idx])
            schedule[job_idx][op_idx] = (machine_id, start, processing_time)
            end = start + processing_time
            machine_ready_time[machine_id] = end
            job_ready_time[job_idx] = end
            next_op[job_idx] += 1
            remaining -= 1
        return schedule

    best_schedule, best_makespan = initial

    # ---------- local search: critical path re-routing ----------
    # Build operation graph from schedule to find critical path
    # For each operation, store its predecessor on job and on machine
    # We'll compute earliest and latest start times.

    # Preprocess schedule to get machine order lists
    machine_ops = [[] for _ in range(n_machines)]
    for j in range(n_jobs):
        for op in range(n_ops[j]):
            machine_id, start, proc = best_schedule[j][op]
            machine_ops[machine_id].append((start, j, op))
    for m in range(n_machines):
        machine_ops[m].sort(key=lambda x: x[0])

    # Forward pass to compute ES, EF
    ES = [[0] * n_ops[j] for j in range(n_jobs)]
    EF = [[0] * n_ops[j] for j in range(n_jobs)]
    for j in range(n_jobs):
        for op in range(n_ops[j]):
            # Precedence from job:
            if op == 0:
                job_pred_end = 0
            else:
                job_pred_end = EF[j][op-1]
            # Precedence from same machine previous operation:
            machine_id, start, proc = best_schedule[j][op]
            # find previous operation on that machine
            prev_end = 0
            ops_on_machine = machine_ops[machine_id]
            idx = next(i for i, (s, jj, oo) in enumerate(ops_on_machine) if jj == j and oo == op)
            if idx > 0:
                prev_end = ops_on_machine[idx-1][0] + durations[ops_on_machine[idx-1][1]][ops_on_machine[idx-1][2]][0][1]  # processing time of previous
            ES[j][op] = max(job_pred_end, prev_end)
            EF[j][op] = ES[j][op] + proc

    makespan = max(EF[j][-1] for j in range(n_jobs))

    # Backward pass to compute LS, LF
    LS = [[0] * n_ops[j] for j in range(n_jobs)]
    LF = [[0] * n_ops[j] for j in range(n_jobs)]
    for j in range(n_jobs):
        for op in reversed(range(n_ops[j])):
            # Successor on job:
            if op == n_ops[j]-1:
                job_succ_start = makespan
            else:
                job_succ_start = LS[j][op+1]
            # Successor on machine:
            machine_id, start, proc = best_schedule[j][op]
            ops_on_machine = machine_ops[machine_id]
            idx = next(i for i, (s, jj, oo) in enumerate(ops_on_machine) if jj == j and oo == op)
            if idx == len(ops_on_machine)-1:
                machine_succ_start = makespan
            else:
                machine_succ_start = ops_on_machine[idx+1][0]  # start of next operation on same machine
            LF[j][op] = min(job_succ_start, machine_succ_start)
            LS[j][op] = LF[j][op] - proc

    # Identify critical operations: those where ES == LS
    critical_ops = []
    for j in range(n_jobs):
        for op in range(n_ops[j]):
            if abs(ES[j][op] - LS[j][op]) < 1e-6:
                critical_ops.append((j, op))

    # Heuristic: try up to 3 random critical operations, try alternative machines
    random.seed(42)  # deterministic for reproducibility
    sample_size = min(3, len(critical_ops))
    if sample_size > 0:
        sampled_ops = random.sample(critical_ops, sample_size)
    else:
        sampled_ops = []

    for (j, op) in sampled_ops:
        current_machine = best_schedule[j][op][0]
        candidates = durations[j][op]
        # gather alternative machines
        alt_machines = [m for m, p in candidates if m != current_machine]
        for alt_m in alt_machines:
            forced = {(j, op): alt_m}
            result = construct_schedule(forced_machine=forced)
            if result is not None:
                new_sched, new_makespan = result
                if new_makespan < best_makespan:
                    best_schedule = new_sched
                    best_makespan = new_makespan
                    # Early break: we accept first improvement
                    # (or we could continue to try more but limited)
                    break
        else:
            continue
        break  # only one improvement step

    return best_schedule
