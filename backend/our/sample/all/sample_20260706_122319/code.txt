def fjsp_solver(n_jobs, n_machines, durations):
    schedule = [[None] * len(durations[j]) for j in range(n_jobs)]
    machine_ready_time = [0] * n_machines
    job_ready_time = [0] * n_jobs
    next_op = [0] * n_jobs
    remaining = sum(len(job) for job in durations)

    def current_best_time(job_idx):
        op_idx = next_op[job_idx]
        return min(processing_time for _, processing_time in durations[job_idx][op_idx])

    def job_priority(job_idx):
        op_idx = next_op[job_idx]
        return (
            job_ready_time[job_idx]
            + 0.5 * current_best_time(job_idx)  # Increased weight for current best time
            - 0.2 * (remaining_work(job_idx) / (len(durations[job_idx]) - op_idx + 1) if len(durations[job_idx]) - op_idx > 0 else 1),  # Adjusted remaining work factor
            job_idx,
        )
    
    def remaining_work(job_idx):
        return sum(durations[job_idx][i][1] for i in range(next_op[job_idx], len(durations[job_idx])))

    def machine_priority(job_idx, machine_id, processing_time):
        op_idx = next_op[job_idx]
        start_time = max(machine_ready_time[machine_id], job_ready_time[job_idx])
        completion_time = start_time + processing_time
        
        return (
            completion_time + 0.3 * (current_best_time(job_idx) - processing_time),  # Adjusted tie-breaker weight
            processing_time,
            machine_id,
        )

    while remaining > 0:
        ready_jobs = [job_idx for job_idx in range(n_jobs) if next_op[job_idx] < len(durations[job_idx])]
        job_idx = min(ready_jobs, key=job_priority)
        op_idx = next_op[job_idx]
        machine_id, processing_time = min(
            durations[job_idx][op_idx],
            key=lambda pair: machine_priority(job_idx, pair[0], pair[1]),
        )
        
        start_time = max(machine_ready_time[machine_id], job_ready_time[job_idx])
        end_time = start_time + processing_time

        schedule[job_idx][op_idx] = (machine_id, start_time, processing_time)
        machine_ready_time[machine_id] = end_time
        job_ready_time[job_idx] = end_time
        next_op[job_idx] += 1
        remaining -= 1

    return schedule
