def fjsp_solver(n_jobs, n_machines, durations):
    # Precompute remaining minimum work for each job
    n_ops = [len(job) for job in durations]
    total_ops = sum(n_ops)
    remaining_min_work = [[0] * (n_ops[j] + 1) for j in range(n_jobs)]
    for j in range(n_jobs):
        for op in range(n_ops[j] - 1, -1, -1):
            best = min(durations[j][op], key=lambda x: x[1])[1]
            remaining_min_work[j][op] = best + remaining_min_work[j][op + 1]

    ALPHA = 0.85

    def local_base_job_priority(mrt, jrt, nxt):
        def inner(job_idx):
            op_idx = nxt[job_idx]
            candidates = durations[job_idx][op_idx]
            min_start = min(max(mrt[m], jrt[job_idx]) for m, _ in candidates)
            return min_start
        return inner

    def local_elite_job_priority(mrt, jrt, nxt):
        def inner(job_idx):
            op_idx = nxt[job_idx]
            candidates = durations[job_idx][op_idx]
            min_start = min(max(mrt[m], jrt[job_idx]) for m, _ in candidates)
            optimistic_completion = min_start + remaining_min_work[job_idx][op_idx]
            avg_machine_load = sum(mrt[m] for m, _ in candidates) / len(candidates) if candidates else 0
            return min_start - 0.3 * remaining_min_work[job_idx][op_idx] + 0.2 * avg_machine_load + 0.3 * optimistic_completion
        return inner

    def local_job_priority(mrt, jrt, nxt):
        base = local_base_job_priority(mrt, jrt, nxt)
        elite = local_elite_job_priority(mrt, jrt, nxt)
        def inner(job_idx):
            hybrid = ALPHA * elite(job_idx) + (1 - ALPHA) * base(job_idx)
            return (hybrid, job_idx)
        return inner

    # machine priority with one-step lookahead
    def local_new_machine_priority(mrt, jrt, nxt):
        def inner(job_idx, machine_id, processing_time):
            op_idx = nxt[job_idx]
            start_time = max(mrt[machine_id], jrt[job_idx])
            completion = start_time + processing_time
            machine_load = mrt[machine_id] + processing_time
            job_rem = remaining_min_work[job_idx][op_idx]
            base_score = completion + 0.4 * machine_load + 0.3 * job_rem - 0.2 * processing_time

            # One-step lookahead: estimate extra delay for the next operation
            lookahead_penalty = 0.0
            if op_idx + 1 < n_ops[job_idx]:
                next_candidates = durations[job_idx][op_idx + 1]
                # Updated machine ready times if we choose this machine
                updated_mrt = mrt[:]
                updated_mrt[machine_id] = completion
                # Earliest start of next operation after current job completion
                next_earliest_start = min(
                    max(updated_mrt[m], completion)
                    for m, _ in next_candidates
                )
                lookahead_penalty = next_earliest_start - completion
            score = base_score + 0.3 * lookahead_penalty
            return score
        return inner

    def local_base_machine_priority(mrt, jrt, nxt):
        def inner(job_idx, machine_id, processing_time):
            start_time = max(mrt[machine_id], jrt[job_idx])
            return start_time + processing_time
        return inner

    def local_machine_priority(mrt, jrt, nxt):
        new_prior = local_new_machine_priority(mrt, jrt, nxt)
        base_prior = local_base_machine_priority(mrt, jrt, nxt)
        def inner(job_idx, machine_id, processing_time):
            hybrid = ALPHA * new_prior(job_idx, machine_id, processing_time) + \
                     (1 - ALPHA) * base_prior(job_idx, machine_id, processing_time)
            return (hybrid, mrt[machine_id], processing_time, machine_id)
        return inner

    # Construct schedule
    schedule = [[None] * n_ops[j] for j in range(n_jobs)]
    machine_ready_time = [0] * n_machines
    job_ready_time = [0] * n_jobs
    next_op = [0] * n_jobs
    remaining = total_ops

    # Create local priority functions with current state
    job_priority_fn = local_job_priority(machine_ready_time, job_ready_time, next_op)
    machine_priority_fn = local_machine_priority(machine_ready_time, job_ready_time, next_op)

    while remaining > 0:
        ready_jobs = [j for j in range(n_jobs) if next_op[j] < n_ops[j]]
        # Choose job
        job_idx = min(ready_jobs, key=job_priority_fn)
        op_idx = next_op[job_idx]
        candidates = durations[job_idx][op_idx]

        # Choose machine
        machine_id, processing_time = min(
            candidates,
            key=lambda pair: machine_priority_fn(job_idx, pair[0], pair[1])
        )

        start_time = max(machine_ready_time[machine_id], job_ready_time[job_idx])
        schedule[job_idx][op_idx] = (machine_id, start_time, processing_time)

        end_time = start_time + processing_time
        machine_ready_time[machine_id] = end_time
        job_ready_time[job_idx] = end_time
        next_op[job_idx] += 1
        remaining -= 1

    return schedule
