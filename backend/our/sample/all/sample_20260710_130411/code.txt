def fjsp_solver(n_jobs, n_machines, durations):
    # ---------- helper data ----------
    n_ops = [len(job) for job in durations]
    total_ops = sum(n_ops)
    # Precompute remaining minimum work for each job from each operation
    remaining_min_work = [[0] * (n_ops[j] + 1) for j in range(n_jobs)]
    for j in range(n_jobs):
        for op in range(n_ops[j] - 1, -1, -1):
            best = min(durations[j][op], key=lambda x: x[1])[1]
            remaining_min_work[j][op] = best + remaining_min_work[j][op + 1]

    ALPHA = 0.85

    # ---------- constructive scheduler ----------
    schedule = [[None] * n_ops[j] for j in range(n_jobs)]
    machine_ready_time = [0] * n_machines
    job_ready_time = [0] * n_jobs
    next_op = [0] * n_jobs
    remaining = total_ops

    def local_base_job_priority(job_idx):
        op_idx = next_op[job_idx]
        candidates = durations[job_idx][op_idx]
        min_start = min(max(machine_ready_time[m], job_ready_time[job_idx]) for m, _ in candidates)
        return min_start

    def local_elite_job_priority(job_idx):
        op_idx = next_op[job_idx]
        candidates = durations[job_idx][op_idx]
        min_start = min(max(machine_ready_time[m], job_ready_time[job_idx]) for m, _ in candidates)
        optimistic_completion = min_start + remaining_min_work[job_idx][op_idx]
        avg_machine_load = sum(machine_ready_time[m] for m, _ in candidates) / len(candidates) if candidates else 0
        return min_start - 0.3 * remaining_min_work[job_idx][op_idx] + 0.2 * avg_machine_load + 0.25 * optimistic_completion

    def local_job_priority(job_idx):
        hybrid = ALPHA * local_elite_job_priority(job_idx) + (1 - ALPHA) * local_base_job_priority(job_idx)
        return (hybrid, job_idx)

    # one-step lookahead for machine selection
    def lookahead_next_op(job_idx, op_idx, mrt_after, jrt_after):
        """returns earliest completion of the next operation of job (if exists) given state after current op"""
        if op_idx + 1 >= n_ops[job_idx]:
            return 0
        next_candidates = durations[job_idx][op_idx + 1]
        best = float('inf')
        for m, p in next_candidates:
            start = max(mrt_after[m], jrt_after)
            best = min(best, start + p)
        return best

    def local_new_machine_priority(job_idx, machine_id, processing_time):
        op_idx = next_op[job_idx]
        start_time = max(machine_ready_time[machine_id], job_ready_time[job_idx])
        completion = start_time + processing_time
        machine_load = machine_ready_time[machine_id] + processing_time
        job_rem = remaining_min_work[job_idx][op_idx]

        # Simulate state after placing this operation on machine machine_id
        mrt_after = machine_ready_time[:]  # copy
        mrt_after[machine_id] = completion
        jrt_after = completion
        future = lookahead_next_op(job_idx, op_idx, mrt_after, jrt_after)

        score = (completion
                 + 0.4 * machine_load
                 + 0.3 * job_rem
                 - 0.2 * processing_time
                 + 0.15 * future)  # one-step lookahead term
        return score

    def local_base_machine_priority(job_idx, machine_id, processing_time):
        start_time = max(machine_ready_time[machine_id], job_ready_time[job_idx])
        return start_time + processing_time

    def local_machine_priority(job_idx, machine_id, processing_time):
        hybrid = ALPHA * local_new_machine_priority(job_idx, machine_id, processing_time) + \
                 (1 - ALPHA) * local_base_machine_priority(job_idx, machine_id, processing_time)
        return (hybrid, machine_ready_time[machine_id], processing_time, machine_id)

    while remaining > 0:
        ready_jobs = [j for j in range(n_jobs) if next_op[j] < n_ops[j]]
        job_idx = min(ready_jobs, key=local_job_priority)
        op_idx = next_op[job_idx]
        candidates = durations[job_idx][op_idx]
        machine_id, processing_time = min(
            candidates,
            key=lambda pair: local_machine_priority(job_idx, pair[0], pair[1])
        )
        start_time = max(machine_ready_time[machine_id], job_ready_time[job_idx])
        schedule[job_idx][op_idx] = (machine_id, start_time, processing_time)
        end_time = start_time + processing_time
        machine_ready_time[machine_id] = end_time
        job_ready_time[job_idx] = end_time
        next_op[job_idx] += 1
        remaining -= 1

    return schedule
