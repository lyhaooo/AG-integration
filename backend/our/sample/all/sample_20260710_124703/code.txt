def fjsp_solver(n_jobs, n_machines, durations):
    # ---------- helper data ----------
    n_ops = [len(job) for job in durations]
    total_ops = sum(n_ops)
    # Precompute remaining minimum work for each job from each operation
    remaining_min_work = [[0] * (n_ops[j] + 1) for j in range(n_jobs)]
    for j in range(n_jobs):
        for op in range(n_ops[j] - 1, -1, -1):
            best = min(durations[j][op], key=lambda x: x[1])[1]
            remaining_min_work[j][op] = best + remaining_min_work[j][op + 1]

    # ---------- priority functions (new: job-priority and machine-scoring) ----------
    def job_priority(job_idx, next_op, job_ready_time, machine_ready_time):
        op_idx = next_op[job_idx]
        candidates = durations[job_idx][op_idx]
        # earliest possible start time
        min_start = min(
            max(machine_ready_time[m], job_ready_time[job_idx])
            for m, _ in candidates
        )
        # remaining work from this operation onward (minimum)
        rem = remaining_min_work[job_idx][op_idx]
        # critical-ratio style: prefer jobs with small (ready+remaining) but also penalize large ready time
        # new: combine start + remaining with a bias for early release
        return min_start + 0.7 * rem + 0.3 * job_ready_time[job_idx]

    def machine_priority(job_idx, machine_id, processing_time, next_op, job_ready_time, machine_ready_time):
        start_time = max(machine_ready_time[machine_id], job_ready_time[job_idx])
        completion = start_time + processing_time
        # machine load: current ready time + this operation's duration (a measure of future load)
        machine_load = machine_ready_time[machine_id] + processing_time
        # remaining work of the job after this operation (if any)
        op_idx = next_op[job_idx]
        job_rem = remaining_min_work[job_idx][op_idx + 1] if op_idx + 1 < n_ops[job_idx] else 0
        # score: penalize high completion time and machine load, reward processing jobs with high remaining work
        # new coefficients: more weight on load, less on completion
        return completion + 0.6 * machine_load + 0.2 * job_rem - 0.1 * processing_time

    # ---------- constructive scheduler ----------
    schedule = [[None] * n_ops[j] for j in range(n_jobs)]
    machine_ready_time = [0] * n_machines
    job_ready_time = [0] * n_jobs
    next_op = [0] * n_jobs
    remaining = total_ops

    while remaining > 0:
        # select job using new job_priority
        ready_jobs = [j for j in range(n_jobs) if next_op[j] < n_ops[j]]
        job_idx = min(ready_jobs, key=lambda j: job_priority(j, next_op, job_ready_time, machine_ready_time))

        op_idx = next_op[job_idx]
        candidates = durations[job_idx][op_idx]

        # select machine using new machine_priority
        machine_id, processing_time = min(
            candidates,
            key=lambda pair: machine_priority(job_idx, pair[0], pair[1], next_op, job_ready_time, machine_ready_time)
        )

        start_time = max(machine_ready_time[machine_id], job_ready_time[job_idx])
        schedule[job_idx][op_idx] = (machine_id, start_time, processing_time)
        end_time = start_time + processing_time

        machine_ready_time[machine_id] = end_time
        job_ready_time[job_idx] = end_time
        next_op[job_idx] += 1
        remaining -= 1

    return schedule
