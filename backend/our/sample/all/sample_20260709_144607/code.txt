def fjsp_solver(n_jobs, n_machines, durations):
    # Initialize schedule
    schedule = [[None] * len(durations[j]) for j in range(n_jobs)]
    machine_ready_time = [0] * n_machines
    machine_load = [0] * n_machines  # total processing time assigned
    job_ready_time = [0] * n_jobs
    next_op = [0] * n_jobs

    # Compute initial remaining work using minimum processing time per operation
    remaining_work = [0] * n_jobs
    for j in range(n_jobs):
        total = 0
        for ops in durations[j]:
            min_pt = min(pt for _, pt in ops)
            total += min_pt
        remaining_work[j] = total

    total_ops = sum(len(job) for job in durations)
    completed_ops = 0

    while completed_ops < total_ops:
        # Select job with largest remaining work; break ties by earliest ready time
        ready_jobs = [j for j in range(n_jobs) if next_op[j] < len(durations[j])]
        job_idx = max(ready_jobs, key=lambda j: (remaining_work[j], -job_ready_time[j]))

        op_idx = next_op[job_idx]
        candidates = durations[job_idx][op_idx]

        # Choose machine with load-aware earliest completion
        best_pair = None
        best_score = None
        for m, pt in candidates:
            completion = max(machine_ready_time[m], job_ready_time[job_idx]) + pt
            load = machine_load[m]
            # Primary key: completion, secondary: load, tertiary: machine id
            score = (completion, load, m)
            if best_score is None or score < best_score:
                best_score = score
                best_pair = (m, pt)

        machine_id, processing_time = best_pair
        start_time = max(machine_ready_time[machine_id], job_ready_time[job_idx])
        schedule[job_idx][op_idx] = (machine_id, start_time, processing_time)
        end_time = start_time + processing_time

        # Update machine state
        machine_ready_time[machine_id] = end_time
        machine_load[machine_id] += processing_time

        # Update job state
        job_ready_time[job_idx] = end_time
        remaining_work[job_idx] -= processing_time
        next_op[job_idx] += 1
        completed_ops += 1

    return schedule
