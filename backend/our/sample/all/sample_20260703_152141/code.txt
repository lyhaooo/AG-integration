def fjsp_solver(n_jobs, n_machines, durations):
    schedule = [[None] * len(durations[j]) for j in range(n_jobs)]
    current_time = [0] * n_machines

    for job_idx in range(n_jobs):
        for op_idx in range(len(durations[job_idx])):
            machine_id, processing_time = durations[job_idx][op_idx][0]  # Choose the first candidate
            start_time = max(current_time[machine_id], sum(schedule[job_idx][i][1] if schedule[job_idx][i] else 0 for i in range(op_idx)))
            schedule[job_idx][op_idx] = (machine_id, start_time, processing_time)
            current_time[machine_id] = start_time + processing_time

    return schedule
