def fjsp_solver(n_jobs, n_machines, durations):
    schedule = [[None] * len(durations[j]) for j in range(n_jobs)]
    machine_ready_time = [0] * n_machines
    job_ready_time = [0] * n_jobs
    next_op = [0] * n_jobs
    machine_load = [0] * n_machines          # track total processing time assigned to each machine
    remaining = sum(len(job) for job in durations)

    while remaining > 0:
        # --- Job selection: prioritize by operation flexibility (fewer machines -> more constrained) ---
        ready_jobs = [j for j in range(n_jobs) if next_op[j] < len(durations[j])]
        # Primary key: number of candidate machines (lower = more constrained)
        # Secondary key: job ready time (earlier first)
        job_idx = min(ready_jobs, key=lambda j: (len(durations[j][next_op[j]]), job_ready_time[j]))

        op_idx = next_op[job_idx]
        candidates = durations[job_idx][op_idx]

        # --- Machine selection: avoid overloading high-demand machines ---
        # Choose machine that minimises (start_time, future_load, processing_time, machine_id)
        machine_id, processing_time = min(
            candidates,
            key=lambda pair: (
                max(machine_ready_time[pair[0]], job_ready_time[job_idx]),  # earliest start
                machine_load[pair[0]] + pair[1],                           # cumulative load after assignment
                pair[1],                                                    # processing time tie-breaker
                pair[0]                                                     # machine id tie-breaker
            )
        )

        start_time = max(machine_ready_time[machine_id], job_ready_time[job_idx])
        schedule[job_idx][op_idx] = (machine_id, start_time, processing_time)
        end_time = start_time + processing_time

        # Update state
        machine_ready_time[machine_id] = end_time
        machine_load[machine_id] += processing_time
        job_ready_time[job_idx] = end_time
        next_op[job_idx] += 1
        remaining -= 1

    return schedule
