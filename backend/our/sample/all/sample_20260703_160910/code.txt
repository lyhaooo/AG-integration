def fjsp_solver(n_jobs, n_machines, durations):
    schedule = [[None] * len(durations[j]) for j in range(n_jobs)]
    machine_ready_time = [0] * n_machines
    job_ready_time = [0] * n_jobs
    next_op = [0] * n_jobs
    remaining = sum(len(job) for job in durations)

    while remaining > 0:
        ready_jobs = [j for j in range(n_jobs) if next_op[j] < len(durations[j])]
        job_idx = min(ready_jobs, key=lambda j: job_ready_time[j])
        op_idx = next_op[job_idx]
        candidates = durations[job_idx][op_idx]

        # Add one-step lookahead to consider the next operation's completion time
        lookahead_candidates = []
        for candidate in candidates:
            machine_id, processing_time = candidate
            next_op_time = float('inf')
            if next_op[job_idx] + 1 < len(durations[job_idx]):
                next_candidates = durations[job_idx][next_op[job_idx] + 1]
                for next_candidate in next_candidates:
                    next_machine_id, next_processing_time = next_candidate
                    next_op_time = min(next_op_time, machine_ready_time[next_machine_id] + next_processing_time)

            start_time = max(machine_ready_time[machine_id], job_ready_time[job_idx])
            completion_time = start_time + processing_time
            lookahead_candidates.append((machine_id, processing_time, completion_time, next_op_time))

        # Choose the best candidate considering the lookahead
        machine_id, processing_time, completion_time, next_op_time = min(
            lookahead_candidates,
            key=lambda pair: (completion_time + next_op_time, pair[1], pair[0]),
        )

        start_time = max(machine_ready_time[machine_id], job_ready_time[job_idx])
        schedule[job_idx][op_idx] = (machine_id, start_time, processing_time)
        end_time = start_time + processing_time
        machine_ready_time[machine_id] = end_time
        job_ready_time[job_idx] = end_time
        next_op[job_idx] += 1
        remaining -= 1

    return schedule
