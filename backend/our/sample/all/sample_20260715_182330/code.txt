def fjsp_solver(n_jobs, n_machines, durations):
    # Initialize schedule structure
    schedule = [[None] * len(durations[j]) for j in range(n_jobs)]
    machine_ready_time = [0] * n_machines
    job_ready_time = [0] * n_jobs
    next_op = [0] * n_jobs
    total_ops = sum(len(job) for job in durations)
    remaining = total_ops

    while remaining > 0:
        # Identify jobs that still have unscheduled operations
        ready_jobs = [j for j in range(n_jobs) if next_op[j] < len(durations[j])]

        # Compute total remaining minimum processing time for each ready job
        # and select the job with the largest value (tie-break by earliest ready time)
        def total_rem_min_dur(job_idx):
            ops = durations[job_idx][next_op[job_idx]:]
            return sum(min(candidates, key=lambda x: x[1])[1] for candidates in ops)

        best_job = max(ready_jobs,
                       key=lambda j: (total_rem_min_dur(j),
                                      -job_ready_time[j]))

        job_idx = best_job
        op_idx = next_op[job_idx]
        candidates = durations[job_idx][op_idx]

        # Choose machine: minimize completion time, then duration, then machine id
        machine_id, processing_time = min(
            candidates,
            key=lambda pair: (max(machine_ready_time[pair[0]],
                                  job_ready_time[job_idx]) + pair[1],
                              pair[1],
                              pair[0])
        )

        start_time = max(machine_ready_time[machine_id],
                         job_ready_time[job_idx])
        end_time = start_time + processing_time

        schedule[job_idx][op_idx] = (machine_id, start_time, processing_time)

        # Update state
        machine_ready_time[machine_id] = end_time
        job_ready_time[job_idx] = end_time
        next_op[job_idx] += 1
        remaining -= 1

    return schedule
