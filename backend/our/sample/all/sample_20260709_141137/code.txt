def fjsp_solver(n_jobs, n_machines, durations):
    schedule = [[None] * len(durations[j]) for j in range(n_jobs)]
    machine_ready_time = [0] * n_machines
    job_ready_time = [0] * n_jobs
    next_op = [0] * n_jobs
    remaining = sum(len(job) for job in durations)

    while remaining > 0:
        ready_jobs = [j for j in range(n_jobs) if next_op[j] < len(durations[j])]
        job_idx = min(ready_jobs, key=lambda j: job_ready_time[j])
        op_idx = next_op[job_idx]
        candidates = durations[job_idx][op_idx]
        has_next = (op_idx + 1 < len(durations[job_idx]))

        if has_next:
            next_candidates = durations[job_idx][op_idx + 1]
            best_key = None
            best_machine = None
            best_proc = None
            for m, p in candidates:
                start = max(machine_ready_time[m], job_ready_time[job_idx])
                end = start + p
                # best possible completion of next operation
                best_next = float('inf')
                for m2, p2 in next_candidates:
                    mach_avail = machine_ready_time[m2] if m2 != m else end
                    start2 = max(mach_avail, end)
                    end2 = start2 + p2
                    if end2 < best_next:
                        best_next = end2
                score = max(end, best_next)
                key = (score, p, m)
                if best_key is None or key < best_key:
                    best_key = key
                    best_machine = m
                    best_proc = p
            machine_id, processing_time = best_machine, best_proc
        else:
            # No next operation: pick machine with earliest completion
            best_key = None
            best_machine = None
            best_proc = None
            for m, p in candidates:
                start = max(machine_ready_time[m], job_ready_time[job_idx])
                end = start + p
                key = (end, p, m)
                if best_key is None or key < best_key:
                    best_key = key
                    best_machine = m
                    best_proc = p
            machine_id, processing_time = best_machine, best_proc

        start_time = max(machine_ready_time[machine_id], job_ready_time[job_idx])
        schedule[job_idx][op_idx] = (machine_id, start_time, processing_time)
        end_time = start_time + processing_time
        machine_ready_time[machine_id] = end_time
        job_ready_time[job_idx] = end_time
        next_op[job_idx] += 1
        remaining -= 1

    return schedule
