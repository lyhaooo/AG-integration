def fjsp_solver(n_jobs, n_machines, durations):
    # Precompute remaining minimum work for each job from each operation
    n_ops = [len(job) for job in durations]
    total_ops = sum(n_ops)
    remaining_min_work = [[0] * (n_ops[j] + 1) for j in range(n_jobs)]
    for j in range(n_jobs):
        for op in range(n_ops[j] - 1, -1, -1):
            best = min(durations[j][op], key=lambda x: x[1])[1]
            remaining_min_work[j][op] = best + remaining_min_work[j][op + 1]

    # Schedule construction
    schedule = [[None] * n_ops[j] for j in range(n_jobs)]
    machine_ready_time = [0] * n_machines
    job_ready_time = [0] * n_jobs
    next_op = [0] * n_jobs
    remaining = total_ops

    # Hybridisation parameters
    ALPHA = 0.85       # weight for elite vs base in job selection
    BETA = 0.7         # weight for elite vs historical in elite part
    GAMMA = 0.3        # weight for historical vs elite in machine part (1-GAMMA for elite)

    while remaining > 0:
        ready_jobs = [j for j in range(n_jobs) if next_op[j] < n_ops[j]]

        # ===== Job priority =====
        def job_priority(job_idx):
            op_idx = next_op[job_idx]
            candidates = durations[job_idx][op_idx]
            min_start = min(
                max(machine_ready_time[m], job_ready_time[job_idx])
                for m, _ in candidates
            )

            # Elite job priority (from best solver)
            optimistic_completion = min_start + remaining_min_work[job_idx][op_idx]
            avg_machine_load = sum(machine_ready_time[m] for m, _ in candidates) / len(candidates) if candidates else 0
            elite_val = min_start - 0.3 * remaining_min_work[job_idx][op_idx] + 0.2 * avg_machine_load + 0.25 * optimistic_completion

            # Historical job priority (derived from Gen(E)#2 lesson)
            historical_val = min_start - 0.2 * remaining_min_work[job_idx][op_idx] + 0.3 * avg_machine_load + 0.2 * optimistic_completion

            # Combine elite and historical
            combined_elite = BETA * elite_val + (1 - BETA) * historical_val

            # Base priority: earliest possible completion
            base_val = min_start

            # Final hybrid
            return ALPHA * combined_elite + (1 - ALPHA) * base_val

        job_idx = min(ready_jobs, key=job_priority)
        op_idx = next_op[job_idx]
        candidates = durations[job_idx][op_idx]

        # ===== Machine priority =====
        def machine_priority(pair):
            m_id, p_time = pair
            start_time = max(machine_ready_time[m_id], job_ready_time[job_idx])
            completion = start_time + p_time

            # Elite machine priority
            machine_load = machine_ready_time[m_id] + p_time
            job_rem = remaining_min_work[job_idx][op_idx]
            elite_val = completion + 0.4 * machine_load + 0.3 * job_rem - 0.2 * p_time

            # Historical machine priority
            historical_val = completion + 0.3 * machine_load + 0.4 * job_rem - 0.1 * p_time

            # Combine elite and historical
            combined_elite = GAMMA * historical_val + (1 - GAMMA) * elite_val

            # Base: earliest completion
            base_val = start_time + p_time

            # Final hybrid
            return ALPHA * combined_elite + (1 - ALPHA) * base_val

        machine_id, processing_time = min(candidates, key=machine_priority)

        start_time = max(machine_ready_time[machine_id], job_ready_time[job_idx])
        schedule[job_idx][op_idx] = (machine_id, start_time, processing_time)
        end_time = start_time + processing_time
        machine_ready_time[machine_id] = end_time
        job_ready_time[job_idx] = end_time
        next_op[job_idx] += 1
        remaining -= 1

    return schedule
