def fjsp_solver(n_jobs, n_machines, durations):
    schedule = [[None] * len(durations[j]) for j in range(n_jobs)]
    machine_ready_time = [0] * n_machines
    job_ready_time = [0] * n_jobs
    next_op = [0] * n_jobs
    remaining = sum(len(job) for job in durations)

    while remaining > 0:
        # Select the job with the highest total remaining processing time (using minimum durations)
        ready_jobs = [j for j in range(n_jobs) if next_op[j] < len(durations[j])]
        def remaining_work(j):
            total = 0
            for op_idx in range(next_op[j], len(durations[j])):
                min_proc = min(p for _, p in durations[j][op_idx])
                total += min_proc
            return total
        job_idx = max(ready_jobs, key=remaining_work)
        op_idx = next_op[job_idx]
        candidates = durations[job_idx][op_idx]

        # Evaluate each candidate with one-step lookahead to the next operation
        best_score = float('inf')
        best_pair = None
        for machine_id, proc_time in candidates:
            start_current = max(machine_ready_time[machine_id], job_ready_time[job_idx])
            finish_current = start_current + proc_time

            if op_idx + 1 < len(durations[job_idx]):
                # Simulate next operation's earliest finish given the current choice
                temp_machine_ready = machine_ready_time[:]
                temp_machine_ready[machine_id] = finish_current
                next_candidates = durations[job_idx][op_idx + 1]
                best_next_finish = min(
                    max(temp_machine_ready[m2], finish_current) + proc2
                    for (m2, proc2) in next_candidates
                )
                score = max(finish_current, best_next_finish)  # minimize the later of the two finishes
            else:
                score = finish_current

            # Tie-breaker: lower score, then lower processing time, then lower machine id
            if (score < best_score or
                (score == best_score and (proc_time < best_pair[1] or 
                 (proc_time == best_pair[1] and machine_id < best_pair[0])))):
                best_score = score
                best_pair = (machine_id, proc_time)

        machine_id, proc_time = best_pair
        start_time = max(machine_ready_time[machine_id], job_ready_time[job_idx])
        schedule[job_idx][op_idx] = (machine_id, start_time, proc_time)
        end_time = start_time + proc_time
        machine_ready_time[machine_id] = end_time
        job_ready_time[job_idx] = end_time
        next_op[job_idx] += 1
        remaining -= 1

    return schedule
