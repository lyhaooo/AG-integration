def fjsp_solver(n_jobs, n_machines, durations):
    # Precompute minimum processing time for each operation
    min_times = []
    for job_ops in durations:
        job_min = []
        for candidates in job_ops:
            job_min.append(min(p for _, p in candidates))
        min_times.append(job_min)
    
    # Remaining work per job (sum of min times of unscheduled operations)
    remaining_work = [sum(job_min) for job_min in min_times]
    
    schedule = [[None] * len(durations[j]) for j in range(n_jobs)]
    machine_ready_time = [0] * n_machines
    job_ready_time = [0] * n_jobs
    next_op = [0] * n_jobs
    remaining = sum(len(job) for job in durations)
    
    while remaining > 0:
        # Select job with largest remaining work among those with unscheduled operations
        ready_jobs = [j for j in range(n_jobs) if next_op[j] < len(durations[j])]
        # In case of tie, break by job index (consistent)
        job_idx = max(ready_jobs, key=lambda j: (remaining_work[j], j))
        
        op_idx = next_op[job_idx]
        candidates = durations[job_idx][op_idx]
        
        # Load-aware earliest completion: choose machine with smallest completion time,
        # then with smallest machine ready time (load balancing), then machine id.
        machine_id, processing_time = min(
            candidates,
            key=lambda pair: (
                max(machine_ready_time[pair[0]], job_ready_time[job_idx]) + pair[1],
                machine_ready_time[pair[0]],
                pair[0]
            )
        )
        
        start_time = max(machine_ready_time[machine_id], job_ready_time[job_idx])
        schedule[job_idx][op_idx] = (machine_id, start_time, processing_time)
        end_time = start_time + processing_time
        
        machine_ready_time[machine_id] = end_time
        job_ready_time[job_idx] = end_time
        next_op[job_idx] += 1
        
        # Update remaining work: subtract minimum time of this operation
        remaining_work[job_idx] -= min_times[job_idx][op_idx]
        
        remaining -= 1
    
    return schedule
