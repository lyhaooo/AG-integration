def fjsp_solver(n_jobs, n_machines, durations):
    # Initialize schedule as list of lists with None placeholders
    schedule = [[None] * len(durations[j]) for j in range(n_jobs)]
    machine_ready_time = [0] * n_machines
    job_ready_time = [0] * n_jobs
    next_op = [0] * n_jobs
    remaining = sum(len(job) for job in durations)

    while remaining > 0:
        # Ready jobs: those with unscheduled operations
        ready_jobs = [j for j in range(n_jobs) if next_op[j] < len(durations[j])]

        # Compute total remaining minimum processing time for each ready job
        remaining_min = {}
        for j in ready_jobs:
            total = 0
            for op_idx in range(next_op[j], len(durations[j])):
                # Minimum processing time among all candidate machines
                min_time = min(durations[j][op_idx], key=lambda pair: pair[1])[1]
                total += min_time
            remaining_min[j] = total

        # Select job with largest total remaining minimum processing time
        # (If tie, Python's max returns the first encountered – acceptable)
        job_idx = max(ready_jobs, key=lambda j: remaining_min[j])
        op_idx = next_op[job_idx]
        candidates = durations[job_idx][op_idx]

        # Machine selection: minimize completion time (start + processing)
        machine_id, processing_time = min(
            candidates,
            key=lambda pair: (
                max(machine_ready_time[pair[0]], job_ready_time[job_idx]) + pair[1],
                pair[1],
                pair[0]
            )
        )

        start_time = max(machine_ready_time[machine_id], job_ready_time[job_idx])
        schedule[job_idx][op_idx] = (machine_id, start_time, processing_time)
        end_time = start_time + processing_time

        # Update resource states
        machine_ready_time[machine_id] = end_time
        job_ready_time[job_idx] = end_time
        next_op[job_idx] += 1
        remaining -= 1

    return schedule
