def fjsp_solver(n_jobs, n_machines, durations):
    # Precompute the minimum processing time for each operation
    min_proc = [[min(p[1] for p in candidates) for candidates in job] for job in durations]

    schedule = [[None] * len(durations[j]) for j in range(n_jobs)]
    machine_ready_time = [0] * n_machines
    job_ready_time = [0] * n_jobs
    next_op = [0] * n_jobs
    remaining = sum(len(job) for job in durations)

    while remaining > 0:
        # Determine ready jobs (jobs that still have unscheduled operations)
        ready_jobs = [j for j in range(n_jobs) if next_op[j] < len(durations[j])]

        # Select job with largest total remaining minimum processing time
        # Tie‑break: smaller job_ready_time (earliest available)
        best_job = None
        best_remaining = -1.0
        best_ready_time = float('inf')
        for j in ready_jobs:
            # Sum of min processing times for all remaining operations of job j
            rem = sum(min_proc[j][i] for i in range(next_op[j], len(durations[j])))
            if rem > best_remaining or (rem == best_remaining and job_ready_time[j] < best_ready_time):
                best_remaining = rem
                best_ready_time = job_ready_time[j]
                best_job = j
        job_idx = best_job
        op_idx = next_op[job_idx]

        candidates = durations[job_idx][op_idx]
        # Choose machine with earliest finish time, then shortest processing time, then smallest machine id
        machine_id, processing_time = min(
            candidates,
            key=lambda pair: (
                max(machine_ready_time[pair[0]], job_ready_time[job_idx]) + pair[1],
                pair[1],
                pair[0]
            )
        )
        start_time = max(machine_ready_time[machine_id], job_ready_time[job_idx])
        end_time = start_time + processing_time

        schedule[job_idx][op_idx] = (machine_id, start_time, processing_time)
        machine_ready_time[machine_id] = end_time
        job_ready_time[job_idx] = end_time
        next_op[job_idx] += 1
        remaining -= 1

    return schedule
