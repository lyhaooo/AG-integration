def fjsp_solver(n_jobs, n_machines, durations):
    # ---------- precompute remaining minimum work ----------
    n_ops = [len(job) for job in durations]
    total_ops = sum(n_ops)
    remaining_min_work = [[0] * (n_ops[j] + 1) for j in range(n_jobs)]
    for j in range(n_jobs):
        for op in range(n_ops[j] - 1, -1, -1):
            best = min(durations[j][op], key=lambda x: x[1])[1]
            remaining_min_work[j][op] = best + remaining_min_work[j][op + 1]

    ALPHA = 0.85  # hybrid weight (can be tuned)

    # ---------- constructive scheduler with one-step lookahead ----------
    def build_schedule(forced_machine=None):
        # local state
        sched = [[None] * n_ops[j] for j in range(n_jobs)]
        mrt = [0] * n_machines
        jrt = [0] * n_jobs
        nxt = [0] * n_jobs
        rem = total_ops

        # ---------- local priority functions (capture mrt, jrt, nxt) ----------
        def local_base_job_priority(job_idx):
            op_idx = nxt[job_idx]
            candidates = durations[job_idx][op_idx]
            min_start = min(max(mrt[m], jrt[job_idx]) for m, _ in candidates)
            return min_start

        def local_elite_job_priority(job_idx):
            op_idx = nxt[job_idx]
            candidates = durations[job_idx][op_idx]
            min_start = min(max(mrt[m], jrt[job_idx]) for m, _ in candidates)
            optimistic_completion = min_start + remaining_min_work[job_idx][op_idx]
            avg_machine_load = sum(mrt[m] for m, _ in candidates) / len(candidates) if candidates else 0
            return min_start - 0.3 * remaining_min_work[job_idx][op_idx] + 0.2 * avg_machine_load + 0.25 * optimistic_completion

        def local_job_priority(job_idx):
            hybrid = ALPHA * local_elite_job_priority(job_idx) + (1 - ALPHA) * local_base_job_priority(job_idx)
            return (hybrid, job_idx)

        # ---------- machine priority with one-step lookahead ----------
        def local_machine_priority(job_idx, machine_id, processing_time):
            op_idx = nxt[job_idx]
            start_time = max(mrt[machine_id], jrt[job_idx])
            completion = start_time + processing_time

            # --- one-step lookahead: impact on next operation of the same job ---
            lookahead_penalty = 0.0
            if op_idx + 1 < n_ops[job_idx]:
                # next operation candidates
                next_candidates = durations[job_idx][op_idx + 1]
                # simulate placing current operation: update machine ready time for the used machine
                updated_mrt = mrt[:]
                updated_mrt[machine_id] = completion
                # for each candidate machine for the next operation, compute earliest start
                best_next_completion = float('inf')
                for next_m, next_p in next_candidates:
                    next_start = max(updated_mrt[next_m], completion)
                    best_next_completion = min(best_next_completion, next_start + next_p)
                # penalize if the next operation is delayed (higher = worse)
                lookahead_penalty = 0.15 * best_next_completion
            # original terms (from elite parent)
            machine_load = mrt[machine_id] + processing_time
            job_rem = remaining_min_work[job_idx][op_idx]
            score = completion + 0.4 * machine_load + 0.3 * job_rem - 0.2 * processing_time + lookahead_penalty
            return score

        def local_base_machine_priority(job_idx, machine_id, processing_time):
            start_time = max(mrt[machine_id], jrt[job_idx])
            return start_time + processing_time

        def local_hybrid_machine_priority(job_idx, machine_id, processing_time):
            hybrid = ALPHA * local_machine_priority(job_idx, machine_id, processing_time) + \
                     (1 - ALPHA) * local_base_machine_priority(job_idx, machine_id, processing_time)
            return (hybrid, mrt[machine_id], processing_time, machine_id)

        # ---------- main scheduling loop ----------
        while rem > 0:
            ready_jobs = [j for j in range(n_jobs) if nxt[j] < n_ops[j]]
            job_idx = min(ready_jobs, key=local_job_priority)
            op_idx = nxt[job_idx]
            candidates = durations[job_idx][op_idx]

            # forced machine (if any)
            if forced_machine is not None and (job_idx, op_idx) in forced_machine:
                forced_m_id = forced_machine[(job_idx, op_idx)]
                selected = None
                for m_id, p_time in candidates:
                    if m_id == forced_m_id:
                        selected = (m_id, p_time)
                        break
                if selected is None:
                    return None  # infeasible
                machine_id, processing_time = selected
            else:
                machine_id, processing_time = min(
                    candidates,
                    key=lambda pair: local_hybrid_machine_priority(job_idx, pair[0], pair[1])
                )

            start_time = max(mrt[machine_id], jrt[job_idx])
            sched[job_idx][op_idx] = (machine_id, start_time, processing_time)
            end_time = start_time + processing_time
            mrt[machine_id] = end_time
            jrt[job_idx] = end_time
            nxt[job_idx] += 1
            rem -= 1

        makespan = max(jrt) if n_jobs > 0 else 0
        return sched, makespan

    # run scheduler without forced machines
    schedule, makespan = build_schedule()
    return schedule
