def fjsp_solver(n_jobs, n_machines, durations):
    # Precompute minimum processing time for each operation
    min_pt = [[min(candidates, key=lambda p: p[1])[1] for candidates in job_ops] for job_ops in durations]

    # Total remaining minimum processing time per job
    total_remaining = [sum(job_min) for job_min in min_pt]

    schedule = [[None] * len(durations[j]) for j in range(n_jobs)]
    machine_ready = [0] * n_machines
    job_ready = [0] * n_jobs
    next_op = [0] * n_jobs
    remaining = sum(len(job) for job in durations)

    while remaining > 0:
        # Select ready job with largest total remaining minimum processing time
        # Tie-break by larger job_ready_time (longer waiting job wins)
        ready_jobs = [j for j in range(n_jobs) if next_op[j] < len(durations[j])]
        job_idx = max(ready_jobs, key=lambda j: (total_remaining[j], job_ready[j]))
        op_idx = next_op[job_idx]
        candidates = durations[job_idx][op_idx]

        # Choose machine that minimizes completion time
        machine_id, processing_time = min(
            candidates,
            key=lambda pair: (max(machine_ready[pair[0]], job_ready[job_idx]) + pair[1], pair[1], pair[0])
        )

        start_time = max(machine_ready[machine_id], job_ready[job_idx])
        end_time = start_time + processing_time

        schedule[job_idx][op_idx] = (machine_id, start_time, processing_time)

        machine_ready[machine_id] = end_time
        job_ready[job_idx] = end_time

        # Update remaining heuristic
        total_remaining[job_idx] -= min_pt[job_idx][op_idx]

        next_op[job_idx] += 1
        remaining -= 1

    return schedule
