def fjsp_solver(n_jobs, n_machines, durations):
    schedule = [[None] * len(durations[j]) for j in range(n_jobs)]
    machine_ready_time = [0] * n_machines
    job_ready_time = [0] * n_jobs
    next_op = [0] * n_jobs
    remaining = sum(len(job) for job in durations)

    while remaining > 0:
        # select job with earliest available time
        ready_jobs = [j for j in range(n_jobs) if next_op[j] < len(durations[j])]
        job_idx = min(ready_jobs, key=lambda j: job_ready_time[j])
        op_idx = next_op[job_idx]
        candidates = durations[job_idx][op_idx]

        # Evaluate each candidate with one-step lookahead
        best_key = None
        best_pair = None
        for machine, proc_time in candidates:
            # current operation completion
            start = max(machine_ready_time[machine], job_ready_time[job_idx])
            end = start + proc_time

            # lookahead: best completion of next operation (if exists)
            if op_idx + 1 < len(durations[job_idx]):
                next_candidates = durations[job_idx][op_idx + 1]
                # temporary job ready time after current operation
                job_ready_after = end
                best_next_end = min(
                    max(
                        (end if m2 == machine else machine_ready_time[m2]),
                        job_ready_after
                    ) + t2
                    for m2, t2 in next_candidates
                )
            else:
                # no next operation – use current end as neutral influence
                best_next_end = end

            # combine: minimize sum of current and next best completion, tie-break by processing time then machine id
            key = (end + best_next_end, proc_time, machine)

            if best_key is None or key < best_key:
                best_key = key
                best_pair = (machine, proc_time)

        machine_id, processing_time = best_pair
        start_time = max(machine_ready_time[machine_id], job_ready_time[job_idx])
        schedule[job_idx][op_idx] = (machine_id, start_time, processing_time)
        end_time = start_time + processing_time

        # update ready times
        machine_ready_time[machine_id] = end_time
        job_ready_time[job_idx] = end_time
        next_op[job_idx] += 1
        remaining -= 1

    return schedule
