def fjsp_solver(n_jobs, n_machines, durations):
    # Initialize data structures
    schedule = [[None] * len(durations[j]) for j in range(n_jobs)]
    machine_ready_time = [0] * n_machines
    job_ready_time = [0] * n_jobs
    next_op = [0] * n_jobs
    remaining = sum(len(job) for job in durations)

    # Precompute total minimum processing time for each job (used for remaining work)
    # Not strictly needed but helps readability
    total_min_processing = []
    for job in durations:
        total = 0
        for candidates in job:
            total += min(p for _, p in candidates)
        total_min_processing.append(total)

    while remaining > 0:
        # Jobs that still have unscheduled operations
        ready_jobs = [j for j in range(n_jobs) if next_op[j] < len(durations[j])]

        # Job selection rule: fewer eligible machines, then larger remaining work
        def job_key(j):
            op = next_op[j]
            num_eligible = len(durations[j][op])
            # Remaining work (using minimum processing times for remaining operations)
            rem = sum(
                min(p for _, p in durations[j][op])
                for op in range(next_op[j], len(durations[j]))
            )
            return (num_eligible, -rem)

        job_idx = min(ready_jobs, key=job_key)
        op_idx = next_op[job_idx]
        candidates = durations[job_idx][op_idx]

        # Machine selection: earliest completion time (with tie-breaking by processing time, then machine id)
        machine_id, processing_time = min(
            candidates,
            key=lambda pair: (
                max(machine_ready_time[pair[0]], job_ready_time[job_idx]) + pair[1],
                pair[1],
                pair[0],
            ),
        )

        start_time = max(machine_ready_time[machine_id], job_ready_time[job_idx])
        schedule[job_idx][op_idx] = (machine_id, start_time, processing_time)
        end_time = start_time + processing_time

        machine_ready_time[machine_id] = end_time
        job_ready_time[job_idx] = end_time
        next_op[job_idx] += 1
        remaining -= 1

    return schedule
