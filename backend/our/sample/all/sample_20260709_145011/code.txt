def fjsp_solver(n_jobs, n_machines, durations):
    # Initialize schedule with None for each operation
    schedule = [[None] * len(durations[j]) for j in range(n_jobs)]
    # Tracks next available time per machine
    machine_ready_time = [0] * n_machines
    # Tracks earliest next start time per job (finish of last scheduled op)
    job_ready_time = [0] * n_jobs
    # Index of next operation to schedule for each job
    next_op = [0] * n_jobs
    # Total number of operations remaining
    remaining = sum(len(job) for job in durations)

    while remaining > 0:
        # Determine which jobs have a ready operation
        ready_jobs = [j for j in range(n_jobs) if next_op[j] < len(durations[j])]

        # Variant 20: choose the job whose current operation has the shortest best processing time
        # Tie-break by earliest job ready time (then job index)
        def job_key(j):
            op_idx = next_op[j]
            best_min = min(p[1] for p in durations[j][op_idx])
            return (best_min, job_ready_time[j], j)

        job_idx = min(ready_jobs, key=job_key)

        op_idx = next_op[job_idx]
        candidates = durations[job_idx][op_idx]

        # Machine selection: minimise max(machine_ready, job_ready) + processing_time,
        # tie-break by processing_time then machine_id
        machine_id, processing_time = min(
            candidates,
            key=lambda pair: (
                max(machine_ready_time[pair[0]], job_ready_time[job_idx]) + pair[1],
                pair[1],
                pair[0]
            )
        )

        start_time = max(machine_ready_time[machine_id], job_ready_time[job_idx])
        schedule[job_idx][op_idx] = (machine_id, start_time, processing_time)
        end_time = start_time + processing_time

        machine_ready_time[machine_id] = end_time
        job_ready_time[job_idx] = end_time
        next_op[job_idx] += 1
        remaining -= 1

    return schedule
