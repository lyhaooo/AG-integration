def fjsp_solver(n_jobs, n_machines, durations):
    from collections import defaultdict
    import heapq

    # Initialize the schedule and machine availability
    schedule = [[None] * len(durations[job_idx]) for job_idx in range(n_jobs)]
    machine_availability = [0] * n_machines
    job_operations = [(job_idx, op_idx) for job_idx in range(n_jobs) for op_idx in range(len(durations[job_idx]))]
    
    # Function to sort operations by duration
    def sort_operations(operations):
        return sorted(operations, key=lambda x: durations[x[0]][x[1]][1])

    # Function to select the next operation based on the sorting strategy
    def select_next_operation(operations):
        return sort_operations(operations)

    # Main scheduling loop
    operations = job_operations
    while operations:
        # Select the next operation to schedule
        operations = select_next_operation(operations)
        for i, (job_idx, op_idx) in enumerate(operations):
            machine_id, processing_time = durations[job_idx][op_idx]
            start_time = max(machine_availability[machine_id], sum(schedule[job_idx][prev_op_idx][1] for prev_op_idx in range(op_idx) if schedule[job_idx][prev_op_idx] is not None))
            finish_time = start_time + processing_time
            
            # Schedule the operation if it can be completed
            if all(schedule[job_idx][prev_op_idx] is not None for prev_op_idx in range(op_idx)):
                schedule[job_idx][op_idx] = (machine_id, start_time, processing_time)
                machine_availability[machine_id] = finish_time
                operations.pop(i)  # Remove the scheduled operation
                break
    return schedule
