def fjsp_solver(n_jobs, n_machines, durations):
    # Initialize schedule and tracking arrays
    schedule = [[None] * len(durations[j]) for j in range(n_jobs)]
    machine_ready_time = [0] * n_machines
    job_ready_time = [0] * n_jobs
    next_op = [0] * n_jobs
    total_ops = sum(len(job) for job in durations)
    
    # Precompute total minimal work for each job efficiently
    # total_min_work[j] = sum of min processing times over all operations of job j
    total_min_work = [0] * n_jobs
    for j in range(n_jobs):
        for op_candidates in durations[j]:
            total_min_work[j] += min(p for _, p in op_candidates)
    
    # remaining_min_work[j] = sum of min processing times over unfinished operations
    remaining_min_work = total_min_work[:]  # copy
    
    scheduled_ops = 0
    while scheduled_ops < total_ops:
        # Identify jobs with pending operations
        ready_jobs = [j for j in range(n_jobs) if next_op[j] < len(durations[j])]
        
        # Job selection: minimize number of eligible machines for current op,
        # then maximize remaining total minimal work
        def job_key(j):
            op_idx = next_op[j]
            num_machines = len(durations[j][op_idx])
            rem = remaining_min_work[j]
            return (num_machines, -rem)
        
        job_idx = min(ready_jobs, key=job_key)
        op_idx = next_op[job_idx]
        candidates = durations[job_idx][op_idx]
        
        # Machine selection: earliest completion time, then min processing time, then machine id
        machine_id, processing_time = min(
            candidates,
            key=lambda pair: (
                max(machine_ready_time[pair[0]], job_ready_time[job_idx]) + pair[1],
                pair[1],
                pair[0],
            ),
        )
        
        start_time = max(machine_ready_time[machine_id], job_ready_time[job_idx])
        schedule[job_idx][op_idx] = (machine_id, start_time, processing_time)
        end_time = start_time + processing_time
        
        # Update machine and job ready times
        machine_ready_time[machine_id] = end_time
        job_ready_time[job_idx] = end_time
        
        # Update remaining work for the job: subtract min time of current operation
        current_min = min(p for _, p in candidates)
        remaining_min_work[job_idx] -= current_min
        
        next_op[job_idx] += 1
        scheduled_ops += 1
    
    return schedule
