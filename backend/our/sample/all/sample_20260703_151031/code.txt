def fjsp_solver(n_jobs, n_machines, durations):
    schedule = [[None] * len(durations[j]) for j in range(n_jobs)]
    machine_ready_time = [0] * n_machines
    job_ready_time = [0] * n_jobs
    next_op = [0] * n_jobs
    remaining = sum(len(job) for job in durations)
    
    while remaining > 0:
        ready_jobs = [j for j in range(n_jobs) if next_op[j] < len(durations[j])]
        job_idx = min(ready_jobs, key=lambda j: job_ready_time[j])
        op_idx = next_op[job_idx]
        candidates = durations[job_idx][op_idx]
        
        def lookahead_score(pair):
            machine_id, processing_time = pair
            next_candidates = durations[job_idx][op_idx + 1] if op_idx + 1 < len(durations[job_idx]) else []
            next_completion_times = []
            for next_pair in next_candidates:
                next_machine_id, next_processing_time = next_pair
                next_start_time = max(machine_ready_time[next_machine_id], job_ready_time[job_idx])
                next_completion_times.append(next_start_time + next_processing_time)
            min_next_completion = min(next_completion_times) if next_completion_times else float('inf')
            current_start_time = max(machine_ready_time[machine_id], job_ready_time[job_idx])
            return current_start_time + processing_time + min_next_completion
        
        machine_id, processing_time = min(
            candidates,
            key=lookahead_score
        )
        
        start_time = max(machine_ready_time[machine_id], job_ready_time[job_idx])
        schedule[job_idx][op_idx] = (machine_id, start_time, processing_time)
        end_time = start_time + processing_time
        machine_ready_time[machine_id] = end_time
        job_ready_time[job_idx] = end_time
        next_op[job_idx] += 1
        remaining -= 1
    
    return schedule
