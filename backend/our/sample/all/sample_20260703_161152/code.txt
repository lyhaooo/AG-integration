def fjsp_solver(n_jobs, n_machines, durations):
    schedule = [[None] * len(durations[j]) for j in range(n_jobs)]
    machine_ready_time = [0] * n_machines
    job_ready_time = [0] * n_jobs
    next_op = [0] * n_jobs
    remaining = sum(len(job) for job in durations)

    while remaining > 0:
        ready_jobs = [j for j in range(n_jobs) if next_op[j] < len(durations[j])]
        job_idx = min(ready_jobs, key=lambda j: job_priority(j, job_ready_time, next_op, durations))

        op_idx = next_op[job_idx]
        candidates = durations[job_idx][op_idx]
        machine_id, processing_time = min(
            candidates,
            key=lambda pair: machine_priority(job_idx, pair[0], pair[1], machine_ready_time, job_ready_time, next_op, durations),
        )

        start_time = max(machine_ready_time[machine_id], job_ready_time[job_idx])
        schedule[job_idx][op_idx] = (machine_id, start_time, processing_time)
        end_time = start_time + processing_time
        machine_ready_time[machine_id] = end_time
        job_ready_time[job_idx] = end_time
        next_op[job_idx] += 1
        remaining -= 1

    return schedule

def job_priority(job_idx, job_ready_time, next_op, durations):
    op_idx = next_op[job_idx]
    return (
        job_ready_time[job_idx]
        + 0.2 * current_best_time(job_idx, durations)  # Adjusted weight to increase focus on best time
        - 0.1 * remaining_min_work(job_idx, op_idx, durations),  # Slightly reduced weight for remaining work
        job_idx,
    )

def current_best_time(job_idx, durations):
    op_idx = next_op[job_idx]
    return min(processing_time for _, processing_time in durations[job_idx][op_idx])

def remaining_min_work(job_idx, op_idx, durations):
    return sum(min(processing_time for _, processing_time in durations[job_idx][i]) for i in range(op_idx, len(durations[job_idx])))

def machine_priority(job_idx, machine_id, processing_time, machine_ready_time, job_ready_time, next_op, durations):
    op_idx = next_op[job_idx]
    start_time = max(machine_ready_time[machine_id], job_ready_time[job_idx])
    completion_time = start_time + processing_time
    
    lookahead_completion = 0
    if op_idx + 1 < len(durations[job_idx]):
        lookahead_completion = min(
            max(machine_ready_time[next_machine], completion_time) + next_time
            for next_machine, next_time in durations[job_idx][op_idx + 1]
        )
        
    return (
        completion_time + 0.2 * lookahead_completion,  # Increased weight for lookahead completion
        processing_time,
        machine_id,
    )
