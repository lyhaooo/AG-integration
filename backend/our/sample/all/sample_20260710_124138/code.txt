def fjsp_solver(n_jobs, n_machines, durations):
    # Precompute the minimum possible processing time for each operation
    min_pt = []
    for job_ops in durations:
        job_min = []
        for candidates in job_ops:
            job_min.append(min(p for _, p in candidates))
        min_pt.append(job_min)

    # Total remaining minimum processing time per job (initially sum over all ops)
    remaining_min_sum = [sum(job_min) for job_min in min_pt]

    # Scheduling data structures
    schedule = [[None] * len(durations[j]) for j in range(n_jobs)]
    machine_ready_time = [0] * n_machines
    job_ready_time = [0] * n_jobs
    next_op = [0] * n_jobs
    remaining = sum(len(job) for job in durations)

    while remaining > 0:
        # Determine ready jobs (those with unscheduled operations)
        ready_jobs = [j for j in range(n_jobs) if next_op[j] < len(durations[j])]

        # Select job with largest total remaining minimum processing time.
        # Tie-break by job index (higher index chosen, deterministic)
        job_idx = max(ready_jobs, key=lambda j: (remaining_min_sum[j], j))

        op_idx = next_op[job_idx]
        candidates = durations[job_idx][op_idx]

        # Choose machine and start time to minimize completion time
        # (max(machine_ready, job_ready) + processing_time, then by duration, then by machine id)
        machine_id, processing_time = min(
            candidates,
            key=lambda pair: (
                max(machine_ready_time[pair[0]], job_ready_time[job_idx]) + pair[1],
                pair[1],
                pair[0]
            )
        )

        start_time = max(machine_ready_time[machine_id], job_ready_time[job_idx])
        schedule[job_idx][op_idx] = (machine_id, start_time, processing_time)
        end_time = start_time + processing_time

        # Update ready times
        machine_ready_time[machine_id] = end_time
        job_ready_time[job_idx] = end_time
        next_op[job_idx] += 1
        remaining -= 1

        # Update remaining minimum sum for this job (subtract min of just scheduled op)
        remaining_min_sum[job_idx] -= min_pt[job_idx][op_idx]

    return schedule
