def fjsp_solver(n_jobs, n_machines, durations):
    schedule = [[None] * len(durations[j]) for j in range(n_jobs)]
    machine_ready_time = [0] * n_machines
    job_ready_time = [0] * n_jobs
    next_op = [0] * n_jobs
    remaining = sum(len(job) for job in durations)

    # Precompute minimum possible processing time for each operation (used in priority)
    min_op_time = [[0] * len(durations[j]) for j in range(n_jobs)]
    for j in range(n_jobs):
        for op_idx in range(len(durations[j])):
            min_op_time[j][op_idx] = min(p for _, p in durations[j][op_idx])

    # Precompute remaining minimum work from each operation onward
    remaining_min_work = [[0] * (len(durations[j]) + 1) for j in range(n_jobs)]
    for j in range(n_jobs):
        for op_idx in range(len(durations[j]) - 1, -1, -1):
            remaining_min_work[j][op_idx] = remaining_min_work[j][op_idx + 1] + min_op_time[j][op_idx]

    def current_best_time(job_idx):
        return min_op_time[job_idx][next_op[job_idx]]

    def job_priority(job_idx):
        op_idx = next_op[job_idx]
        # Hybrid: increase weight on remaining work and include a small machine load factor
        return (
            job_ready_time[job_idx]
            + 0.3 * current_best_time(job_idx)
            - 0.3 * remaining_min_work[job_idx][op_idx],
            job_idx,
        )

    def machine_priority(job_idx, machine_id, processing_time):
        op_idx = next_op[job_idx]
        start_time = max(machine_ready_time[machine_id], job_ready_time[job_idx])
        completion_time = start_time + processing_time

        # Lookahead: estimate completion of next operation if scheduled immediately after this one
        lookahead_completion = 0
        if op_idx + 1 < len(durations[job_idx]):
            next_candidates = durations[job_idx][op_idx + 1]
            lookahead_completion = min(
                max(machine_ready_time[m], completion_time) + t
                for m, t in next_candidates
            )

        # Added term: prefer machines that are less loaded (lower ready time)
        # Also adjust weights to give more importance to completion time and lookahead
        return (
            completion_time
            + 0.25 * lookahead_completion
            + 0.15 * (current_best_time(job_idx) - processing_time)
            + 0.1 * machine_ready_time[machine_id],  # favor idle machines
            processing_time,
            machine_id,
        )

    while remaining > 0:
        ready_jobs = [j for j in range(n_jobs) if next_op[j] < len(durations[j])]
        job_idx = min(ready_jobs, key=job_priority)
        op_idx = next_op[job_idx]
        candidates = durations[job_idx][op_idx]
        machine_id, processing_time = min(
            candidates,
            key=lambda pair: machine_priority(job_idx, pair[0], pair[1]),
        )

        start_time = max(machine_ready_time[machine_id], job_ready_time[job_idx])
        end_time = start_time + processing_time

        schedule[job_idx][op_idx] = (machine_id, start_time, processing_time)
        machine_ready_time[machine_id] = end_time
        job_ready_time[job_idx] = end_time
        next_op[job_idx] += 1
        remaining -= 1

    return schedule
