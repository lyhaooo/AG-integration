def fjsp_solver(n_jobs, n_machines, durations):
    # schedule[job][op] = (machine_id, start_time, duration)
    schedule = [[None] * len(durations[j]) for j in range(n_jobs)]

    # State trackers
    machine_ready = [0] * n_machines
    job_ready = [0] * n_jobs
    next_op = [0] * n_jobs
    remaining_ops = sum(len(job) for job in durations)

    # Pre‑compute the shortest possible processing time for each operation
    min_dur = [[min(p for _, p in op) for op in job] for job in durations]
    # Remaining work estimate: sum of min durations of unscheduled operations
    rem_work = [sum(min_dur[j]) for j in range(n_jobs)]

    while remaining_ops > 0:
        # Jobs that still have an unscheduled operation
        ready = [j for j in range(n_jobs) if next_op[j] < len(durations[j])]

        # Variant 18: prefer job with
        #   1) fewest eligible machines for its current operation,
        #   2) largest remaining work (so -rem_work for ascending order),
        #   3) earliest job ready time as tie‑breaker.
        job_idx = min(
            ready,
            key=lambda j: (
                len(durations[j][next_op[j]]),   # fewer machines first
                -rem_work[j],                    # larger remaining work first
                job_ready[j]                     # earliest ready time (stable tie‑break)
            )
        )

        op_idx = next_op[job_idx]
        candidates = durations[job_idx][op_idx]

        # Select machine: earliest completion time, then shorter proc time, then machine id
        machine_id, proc_time = min(
            candidates,
            key=lambda pair: (
                max(machine_ready[pair[0]], job_ready[job_idx]) + pair[1],
                pair[1],
                pair[0]
            )
        )

        start = max(machine_ready[machine_id], job_ready[job_idx])
        end = start + proc_time
        schedule[job_idx][op_idx] = (machine_id, start, proc_time)

        # Update state
        machine_ready[machine_id] = end
        job_ready[job_idx] = end
        next_op[job_idx] += 1
        rem_work[job_idx] -= min_dur[job_idx][op_idx]
        remaining_ops -= 1

    return schedule
