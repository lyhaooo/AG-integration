def fjsp_solver(n_jobs, n_machines, durations):
    # ---------- helper data ----------
    n_ops = [len(job) for job in durations]
    total_ops = sum(n_ops)
    # Precompute remaining minimum work for each job from each operation
    remaining_min_work = [[0] * (n_ops[j] + 1) for j in range(n_jobs)]
    # Precompute minimum processing time of the next operation (for lookahead)
    next_min_pt = [[0] * (n_ops[j]) for j in range(n_jobs)]
    for j in range(n_jobs):
        for op in range(n_ops[j] - 1, -1, -1):
            best = min(durations[j][op], key=lambda x: x[1])[1]
            remaining_min_work[j][op] = best + remaining_min_work[j][op + 1]
            if op < n_ops[j] - 1:
                next_min_pt[j][op] = min(durations[j][op + 1], key=lambda x: x[1])[1]
            else:
                next_min_pt[j][op] = 0

    ALPHA = 0.7  # slightly different mixing

    # ---------- constructive scheduler ----------
    def build_schedule():
        sched = [[None] * n_ops[j] for j in range(n_jobs)]
        mrt = [0] * n_machines
        jrt = [0] * n_jobs
        nxt = [0] * n_jobs
        rem = total_ops

        # local priority functions with lookahead
        def job_priority(job_idx):
            op_idx = nxt[job_idx]
            candidates = durations[job_idx][op_idx]
            # earliest possible start for this operation
            min_start = min(max(mrt[m], jrt[job_idx]) for m, _ in candidates)
            # optimistic completion of job from this operation onward
            optimistic_complete = min_start + remaining_min_work[job_idx][op_idx]
            # lookahead: earliest start of next operation if we pick the best machine
            next_earliest = float('inf')
            for m, pt in candidates:
                start_i = max(mrt[m], jrt[job_idx])
                # if there is a next operation, consider its earliest start
                if op_idx + 1 < n_ops[job_idx]:
                    # estimate earliest start of next op: after current, on any machine
                    next_start = start_i + pt + next_min_pt[job_idx][op_idx]
                else:
                    next_start = start_i + pt
                if next_start < next_earliest:
                    next_earliest = next_start
            # job priority: mix base (min_start) with lookahead and remaining work
            hybrid = (0.4 * min_start
                      + 0.3 * optimistic_complete
                      + 0.3 * next_earliest)
            return (hybrid, job_idx)

        def machine_priority(job_idx, machine_id, processing_time):
            op_idx = nxt[job_idx]
            start_time = max(mrt[machine_id], jrt[job_idx])
            completion = start_time + processing_time
            # future load on this machine if we take this operation
            future_load = mrt[machine_id] + processing_time
            # remaining work of the job
            job_rem = remaining_min_work[job_idx][op_idx]
            # one-step lookahead: estimated start of the job's next operation (if any)
            if op_idx + 1 < n_ops[job_idx]:
                # after this operation, job ready time becomes completion
                # next operation could start at max(machine_ready_time[m], completion)
                # We take the minimum possible start over all machines for the next op
                next_candidates = durations[job_idx][op_idx + 1]
                earliest_next_start = float('inf')
                for m2, pt2 in next_candidates:
                    # the next operation cannot start before completion
                    start2 = max(mrt[m2] if m2 != machine_id else completion, completion)
                    earliest_next_start = min(earliest_next_start, start2)
                lookahead = earliest_next_start
            else:
                lookahead = completion  # no next operation, just completion
            # score: minimize completion, future load, and lookahead; reward remaining work
            score = (completion
                     + 0.3 * future_load
                     + 0.4 * lookahead
                     - 0.2 * job_rem)
            # tie-breaking: prefer smaller machine load and then machine id
            return (score, mrt[machine_id], machine_id)

        while rem > 0:
            ready_jobs = [j for j in range(n_jobs) if nxt[j] < n_ops[j]]
            # choose job
            job_idx = min(ready_jobs, key=job_priority)
            op_idx = nxt[job_idx]
            candidates = durations[job_idx][op_idx]
            # choose machine
            machine_id, processing_time = min(
                candidates,
                key=lambda pair: machine_priority(job_idx, pair[0], pair[1])
            )
            start_time = max(mrt[machine_id], jrt[job_idx])
            sched[job_idx][op_idx] = (machine_id, start_time, processing_time)
            end_time = start_time + processing_time
            mrt[machine_id] = end_time
            jrt[job_idx] = end_time
            nxt[job_idx] += 1
            rem -= 1
        return sched

    return build_schedule()
