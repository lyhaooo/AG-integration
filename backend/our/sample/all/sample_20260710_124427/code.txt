def fjsp_solver(n_jobs, n_machines, durations):
    # ---------- helper data ----------
    n_ops = [len(job) for job in durations]
    total_ops = sum(n_ops)
    # Precompute remaining minimum work for each job from each operation
    remaining_min_work = [[0] * (n_ops[j] + 1) for j in range(n_jobs)]
    for j in range(n_jobs):
        for op in range(n_ops[j] - 1, -1, -1):
            best = min(durations[j][op], key=lambda x: x[1])[1]
            remaining_min_work[j][op] = best + remaining_min_work[j][op + 1]

    ALPHA = 0.75  # changed from 0.85 to give more influence to lesson-based rules

    # ---------- priority functions (hybrid of elite and a historical lesson) ----------
    # The historical lesson is based on a simpler heuristic that places more emphasis
    # on the immediate start time and the load of the machines, similar to the skeleton
    # but with a different balancing. We implement it as the "lesson" part.

    def lesson_job_priority(job_idx, mrt, jrt, nxt):
        op_idx = nxt[job_idx]
        candidates = durations[job_idx][op_idx]
        min_start = min(
            max(mrt[m], jrt[job_idx])
            for m, _ in candidates
        )
        # Lesson adds a small penalty based on average machine load and a bias towards
        # jobs with many remaining operations (inverse of remaining work)
        avg_load = sum(mrt[m] for m, _ in candidates) / len(candidates) if candidates else 0
        rem_ops = n_ops[job_idx] - op_idx
        return min_start + 0.1 * avg_load - 0.05 * rem_ops

    def elite_job_priority(job_idx, mrt, jrt, nxt):
        op_idx = nxt[job_idx]
        candidates = durations[job_idx][op_idx]
        min_start = min(
            max(mrt[m], jrt[job_idx])
            for m, _ in candidates
        )
        optimistic_completion = min_start + remaining_min_work[job_idx][op_idx]
        avg_machine_load = sum(mrt[m] for m, _ in candidates) / len(candidates) if candidates else 0
        # Modified coefficients: more weight on machine load, less on remaining work
        return min_start - 0.2 * remaining_min_work[job_idx][op_idx] + 0.3 * avg_machine_load + 0.25 * optimistic_completion

    def job_priority(job_idx, mrt, jrt, nxt):
        hybrid = ALPHA * elite_job_priority(job_idx, mrt, jrt, nxt) + \
                 (1 - ALPHA) * lesson_job_priority(job_idx, mrt, jrt, nxt)
        return (hybrid, jrt[job_idx], job_idx)  # tie-breaker: earlier job ready

    def lesson_machine_priority(job_idx, machine_id, processing_time, mrt, jrt, nxt):
        start_time = max(mrt[machine_id], jrt[job_idx])
        completion = start_time + processing_time
        # Lesson prefers machines that are already lightly loaded
        return completion + 0.1 * mrt[machine_id]

    def elite_machine_priority(job_idx, machine_id, processing_time, mrt, jrt, nxt):
        op_idx = nxt[job_idx]
        start_time = max(mrt[machine_id], jrt[job_idx])
        completion = start_time + processing_time
        machine_load = mrt[machine_id] + processing_time
        job_rem = remaining_min_work[job_idx][op_idx]
        # Modified coefficients: less penalty on machine load, more on job remaining
        return completion + 0.3 * machine_load + 0.4 * job_rem - 0.1 * processing_time

    def machine_priority(job_idx, machine_id, processing_time, mrt, jrt, nxt):
        hybrid = ALPHA * elite_machine_priority(job_idx, machine_id, processing_time, mrt, jrt, nxt) + \
                 (1 - ALPHA) * lesson_machine_priority(job_idx, machine_id, processing_time, mrt, jrt, nxt)
        return (hybrid, mrt[machine_id], processing_time, machine_id)  # tie-breaker: earlier machine ready, shorter processing

    # ---------- constructive scheduler ----------
    def build_schedule():
        schedule = [[None] * n_ops[j] for j in range(n_jobs)]
        machine_ready_time = [0] * n_machines
        job_ready_time = [0] * n_jobs
        next_op = [0] * n_jobs
        remaining = total_ops

        while remaining > 0:
            ready_jobs = [j for j in range(n_jobs) if next_op[j] < n_ops[j]]
            # Choose job using hybrid priority
            job_idx = min(ready_jobs, key=lambda j: job_priority(j, machine_ready_time, job_ready_time, next_op))
            op_idx = next_op[job_idx]
            candidates = durations[job_idx][op_idx]
            # Choose machine using hybrid priority
            machine_id, processing_time = min(
                candidates,
                key=lambda pair: machine_priority(job_idx, pair[0], pair[1],
                                                   machine_ready_time, job_ready_time, next_op)
            )
            start_time = max(machine_ready_time[machine_id], job_ready_time[job_idx])
            schedule[job_idx][op_idx] = (machine_id, start_time, processing_time)
            end_time = start_time + processing_time
            machine_ready_time[machine_id] = end_time
            job_ready_time[job_idx] = end_time
            next_op[job_idx] += 1
            remaining -= 1

        return schedule

    return build_schedule()
