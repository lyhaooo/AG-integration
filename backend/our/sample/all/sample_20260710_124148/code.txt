def fjsp_solver(n_jobs, n_machines, durations):
    # Initialize schedule, ready times, next operations, and remaining count
    schedule = [[None] * len(durations[j]) for j in range(n_jobs)]
    machine_ready_time = [0] * n_machines
    job_ready_time = [0] * n_jobs
    next_op = [0] * n_jobs
    remaining = sum(len(job) for job in durations)

    while remaining > 0:
        # Identify jobs that still have unscheduled operations
        ready_jobs = [j for j in range(n_jobs) if next_op[j] < len(durations[j])]

        # Variant 14: Choose the ready job whose current operation
        # has the shortest best (minimum) processing time.
        # Precompute min processing time for each ready job's next operation.
        min_pt = {}
        for j in ready_jobs:
            op_idx = next_op[j]
            candidates = durations[j][op_idx]
            # Best processing time is the minimum among all machine options
            best_pt = min(pt for _, pt in candidates)
            min_pt[j] = best_pt

        # Select job with smallest min processing time; tie-break by job ready time
        job_idx = min(ready_jobs, key=lambda j: (min_pt[j], job_ready_time[j]))

        op_idx = next_op[job_idx]
        candidates = durations[job_idx][op_idx]

        # Choose machine and start time: best combination that minimizes
        # max(machine_ready, job_ready) + processing_time,
        # with secondary keys processing_time and machine id.
        machine_id, processing_time = min(
            candidates,
            key=lambda pair: (
                max(machine_ready_time[pair[0]], job_ready_time[job_idx]) + pair[1],
                pair[1],
                pair[0]
            )
        )

        start_time = max(machine_ready_time[machine_id], job_ready_time[job_idx])
        schedule[job_idx][op_idx] = (machine_id, start_time, processing_time)
        end_time = start_time + processing_time

        machine_ready_time[machine_id] = end_time
        job_ready_time[job_idx] = end_time
        next_op[job_idx] += 1
        remaining -= 1

    return schedule
