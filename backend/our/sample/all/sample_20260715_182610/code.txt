def fjsp_solver(n_jobs, n_machines, durations):
    # Precompute static bottleneck count: number of operations that can use each machine
    machine_bottleneck = [0] * n_machines
    for job in durations:
        for op in job:
            for machine_id, _ in op:
                machine_bottleneck[machine_id] += 1

    # Initialize structures
    schedule = [[None] * len(durations[j]) for j in range(n_jobs)]
    machine_ready_time = [0] * n_machines
    job_ready_time = [0] * n_jobs
    next_op = [0] * n_jobs
    remaining = sum(len(job) for job in durations)

    while remaining > 0:
        # Select the job with the earliest ready time from those with unscheduled operations
        ready_jobs = [j for j in range(n_jobs) if next_op[j] < len(durations[j])]
        if not ready_jobs:
            break
        job_idx = min(ready_jobs, key=lambda j: job_ready_time[j])
        op_idx = next_op[job_idx]
        candidates = durations[job_idx][op_idx]

        # Choose machine: earliest finish time + penalty for bottleneck machines
        # Penalty is 0.5 * machine_bottleneck to discourage using bottleneck machines
        def candidate_key(pair):
            m, p = pair
            start = max(machine_ready_time[m], job_ready_time[job_idx])
            finish = start + p
            return finish + 0.5 * machine_bottleneck[m]

        machine_id, processing_time = min(candidates, key=candidate_key)

        # Schedule the operation
        start_time = max(machine_ready_time[machine_id], job_ready_time[job_idx])
        schedule[job_idx][op_idx] = (machine_id, start_time, processing_time)
        end_time = start_time + processing_time
        machine_ready_time[machine_id] = end_time
        job_ready_time[job_idx] = end_time
        next_op[job_idx] += 1
        remaining -= 1

    return schedule
