def fjsp_solver(n_jobs, n_machines, durations):
    # ---------- precomputation ----------
    n_ops = [len(job) for job in durations]
    total_ops = sum(n_ops)
    remaining_min_work = [[0] * (n_ops[j] + 1) for j in range(n_jobs)]
    for j in range(n_jobs):
        for op in range(n_ops[j] - 1, -1, -1):
            best = min(durations[j][op], key=lambda x: x[1])[1]
            remaining_min_work[j][op] = best + remaining_min_work[j][op + 1]

    # ---------- constructive scheduler ----------
    def build_schedule(forced_machine=None):
        """
        forced_machine: dict {(job, op): machine_id}
        Returns (schedule, makespan, job_ready_time) or None if forced machine invalid.
        """
        sched = [[None] * n_ops[j] for j in range(n_jobs)]
        mrt = [0] * n_machines
        jrt = [0] * n_jobs
        nxt = [0] * n_jobs
        rem = total_ops

        while rem > 0:
            # -- job selection with modified priority --
            ready = [j for j in range(n_jobs) if nxt[j] < n_ops[j]]
            def job_priority(j):
                op = nxt[j]
                cand = durations[j][op]
                min_start = min(max(mrt[m], jrt[j]) for m, _ in cand)
                rem_work = remaining_min_work[j][op]
                avg_load = sum(mrt[m] for m, _ in cand) / len(cand) if cand else 0.0
                min_pt = min(p for _, p in cand)
                return (min_start + 0.6 * rem_work + 0.3 * avg_load - 0.2 * min_pt, j)
            job_idx = min(ready, key=job_priority)[1]

            op_idx = nxt[job_idx]
            candidates = durations[job_idx][op_idx]

            # -- machine selection (with optional forced assignment) --
            if forced_machine and (job_idx, op_idx) in forced_machine:
                forced_m = forced_machine[(job_idx, op_idx)]
                valid = False
                for m, p in candidates:
                    if m == forced_m:
                        machine_id = m
                        processing_time = p
                        valid = True
                        break
                if not valid:
                    return None
            else:
                def machine_priority(m, p):
                    start = max(mrt[m], jrt[job_idx])
                    return (start + p, mrt[m] + p, p, m)
                machine_id, processing_time = min(candidates, key=lambda x: machine_priority(x[0], x[1]))

            start_time = max(mrt[machine_id], jrt[job_idx])
            sched[job_idx][op_idx] = (machine_id, start_time, processing_time)
            end_time = start_time + processing_time
            mrt[machine_id] = end_time
            jrt[job_idx] = end_time
            nxt[job_idx] += 1
            rem -= 1

        makespan = max(jrt)
        return sched, makespan, jrt

    # ---------- initial schedule ----------
    best_sched, best_makespan, jrt = build_schedule()

    # ---------- lightweight local search on critical path ----------
    for _ in range(3):
        # Identify a critical operation: last operation of job finishing at makespan
        crit_job = max(range(n_jobs), key=lambda j: jrt[j])
        crit_op = n_ops[crit_job] - 1
        current_machine = best_sched[crit_job][crit_op][0]
        candidates = durations[crit_job][crit_op]
        alternatives = [m for m, _ in candidates if m != current_machine]
        if not alternatives:
            continue

        for alt_m in alternatives:
            forced = {(crit_job, crit_op): alt_m}
            new_sched, new_makespan, new_jrt = build_schedule(forced_machine=forced)
            if new_sched is None:
                continue
            if new_makespan < best_makespan:
                best_sched = new_sched
                best_makespan = new_makespan
                jrt = new_jrt
                break  # accept first improvement
        # else no improvement found for this critical operation

    return best_sched
