def fjsp_solver(n_jobs, n_machines, durations):
    # Initialize schedule: list of lists, each slot will hold a tuple (machine, start, duration)
    schedule = [[None] * len(durations[j]) for j in range(n_jobs)]
    # Track next available time for each machine
    machine_ready_time = [0.0] * n_machines
    # Track end time of last scheduled operation for each job
    job_ready_time = [0.0] * n_jobs
    # Index of next operation to schedule for each job
    next_op = [0] * n_jobs
    # Total number of operations remaining
    remaining = sum(len(job) for job in durations)

    while remaining > 0:
        # Identify jobs that still have unscheduled operations
        ready_jobs = [j for j in range(n_jobs) if next_op[j] < len(durations[j])]

        # --- VARIANT 8: Job selection based on shortest best processing time ---
        # For each ready job, compute the minimum processing time among its current operation's candidates
        # Choose the job with the smallest such minimum processing time.
        # Tie-break with job_ready_time (so that jobs ready earlier are preferred in a tie).
        def best_pt(j):
            op_idx = next_op[j]
            candidates = durations[j][op_idx]
            # min processing time among all candidate machines
            return min(pair[1] for pair in candidates)

        job_idx = min(ready_jobs, key=lambda j: (best_pt(j), job_ready_time[j]))
        # ----------------------------------------------------------------------

        op_idx = next_op[job_idx]
        candidates = durations[job_idx][op_idx]

        # Machine selection: choose the (machine, processing_time) that minimizes
        # (max(machine_ready, job_ready) + processing_time, processing_time, machine_id)
        # This is a common heuristic to reduce completion time while favouring shorter ops and lower machine index.
        machine_id, processing_time = min(
            candidates,
            key=lambda pair: (
                max(machine_ready_time[pair[0]], job_ready_time[job_idx]) + pair[1],
                pair[1],
                pair[0],
            ),
        )

        start_time = max(machine_ready_time[machine_id], job_ready_time[job_idx])
        # Record the assignment
        schedule[job_idx][op_idx] = (machine_id, start_time, processing_time)

        end_time = start_time + processing_time
        # Update machine and job ready times
        machine_ready_time[machine_id] = end_time
        job_ready_time[job_idx] = end_time

        next_op[job_idx] += 1
        remaining -= 1

    return schedule
