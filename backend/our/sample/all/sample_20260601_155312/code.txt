def apply_assignment_rule(schedule, jobs, machines):

    machine_availability = {machine: 0 for machine in machines}
    
    for operation in schedule:
        best_machine = None
        earliest_completion_time = float('inf')

        # Evaluate each machine for the best assignment
        for machine in machines:
            completion_time = machine_availability[machine] + operation.processing_time
