def fjsp_solver(n_jobs, n_machines, durations):
    # schedule: list of lists, each inner list length = number of ops for that job
    schedule = [[None] * len(durations[j]) for j in range(n_jobs)]
    
    # machine availability
    machine_ready_time = [0] * n_machines
    # cumulative processing time assigned to each machine (machine load)
    machine_load = [0] * n_machines
    
    # job availability
    job_ready_time = [0] * n_jobs
    next_op = [0] * n_jobs
    
    remaining = sum(len(job) for job in durations)
    
    while remaining > 0:
        # select the job with earliest ready time (any ready job)
        ready_jobs = [j for j in range(n_jobs) if next_op[j] < len(durations[j])]
        job_idx = min(ready_jobs, key=lambda j: job_ready_time[j])
        
        op_idx = next_op[job_idx]
        candidates = durations[job_idx][op_idx]
        
        # choose machine: minimize (earliest completion time + current machine load)
        def score(candidate):
            m, p = candidate
            est = max(machine_ready_time[m], job_ready_time[job_idx])
            completion = est + p
            return completion + machine_load[m]
        
        machine_id, processing_time = min(candidates, key=score)
        
        start_time = max(machine_ready_time[machine_id], job_ready_time[job_idx])
        end_time = start_time + processing_time
        
        # record result
        schedule[job_idx][op_idx] = (machine_id, start_time, processing_time)
        
        # update machine and job states
        machine_ready_time[machine_id] = end_time
        machine_load[machine_id] += processing_time
        job_ready_time[job_idx] = end_time
        next_op[job_idx] += 1
        remaining -= 1
    
    return schedule
