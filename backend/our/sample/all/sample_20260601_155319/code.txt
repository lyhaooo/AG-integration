def apply_assignment_rule(schedule, jobs, machines):

    for operation in schedule:
        best_machine = None
        best_finish_time = float('inf')

        # Evaluate each machine to find the best one for the operation
        for machine in machines:
            finish_time = calculate_finish_time(schedule, operation, machine)
            if finish_time < best_finish_time:
                best_finish_time = finish_time
                best_machine = machine

        # Assign the operation to the best machine found
        if best_machine is not None:
            assign_operation_to_machine(operation, best_machine)

def calculate_finish_time(schedule, operation, machine):
    # Calculate the finish time of an operation on a given machine
    current_time = 0
    # Calculate the current time based on the existing schedule for the machine
    for scheduled_operation in schedule:
        if scheduled_operation.machine == machine:
            current_time += scheduled_operation.processing_time
    return current_time + operation.processing_time

def assign_operation_to_machine(operation, machine):
    # Assign the operation to the machine in the schedule
    operation.machine = machine
