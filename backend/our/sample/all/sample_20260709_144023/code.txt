def fjsp_solver(n_jobs, n_machines, durations):
    # Initialize schedule structure
    schedule = [[None] * len(durations[j]) for j in range(n_jobs)]
    machine_ready_time = [0] * n_machines
    job_ready_time = [0] * n_jobs
    next_op = [0] * n_jobs
    remaining = sum(len(job) for job in durations)

    # Precompute minimum processing time for every operation
    min_time_per_op = [[0] * len(durations[j]) for j in range(n_jobs)]
    for j in range(n_jobs):
        for o in range(len(durations[j])):
            candidates = durations[j][o]
            # pick the smallest processing time among candidate machines
            min_time_per_op[j][o] = min(pt for _, pt in candidates)

    # Total remaining minimum processing time for each job
    total_remaining_min = [sum(min_time_per_op[j]) for j in range(n_jobs)]

    while remaining > 0:
        # Determine which jobs still have unscheduled operations
        ready_jobs = [j for j in range(n_jobs) if next_op[j] < len(durations[j])]

        # Variant 13: choose job with largest total remaining minimum processing time
        # Tie-break: smaller job_ready_time (earliest ready) 
        job_idx = max(ready_jobs, key=lambda j: (total_remaining_min[j], -job_ready_time[j]))

        op_idx = next_op[job_idx]
        candidates = durations[job_idx][op_idx]

        # Machine selection: minimize (max(machine_ready, job_ready) + processing_time,
        # then processing_time, then machine_id)
        machine_id, processing_time = min(
            candidates,
            key=lambda pair: (
                max(machine_ready_time[pair[0]], job_ready_time[job_idx]) + pair[1],
                pair[1],
                pair[0]
            )
        )

        start_time = max(machine_ready_time[machine_id], job_ready_time[job_idx])
        schedule[job_idx][op_idx] = (machine_id, start_time, processing_time)
        end_time = start_time + processing_time

        # Update ready time of machine and job
        machine_ready_time[machine_id] = end_time
        job_ready_time[job_idx] = end_time

        # Subtract the minimum processing time of this operation from the job's remaining total
        total_remaining_min[job_idx] -= min_time_per_op[job_idx][op_idx]

        # Advance operation pointer
        next_op[job_idx] += 1
        remaining -= 1

    return schedule
