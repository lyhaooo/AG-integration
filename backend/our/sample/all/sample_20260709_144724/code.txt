def fjsp_solver(n_jobs, n_machines, durations):
    # Precompute minimum processing time for each operation (for work remaining estimation)
    min_pt = [[min(p for _, p in candidates) for candidates in job] for job in durations]

    # Initialize schedule and tracking structures
    schedule = [[None] * len(durations[j]) for j in range(n_jobs)]
    machine_ready_time = [0] * n_machines
    machine_load = [0] * n_machines
    job_ready_time = [0] * n_jobs
    next_op = [0] * n_jobs
    remaining_ops = sum(len(job) for job in durations)

    while remaining_ops > 0:
        # Compute remaining work for each job (sum of min processing times of unscheduled ops)
        remaining_work = []
        for j in range(n_jobs):
            if next_op[j] < len(durations[j]):
                work = sum(min_pt[j][op] for op in range(next_op[j], len(durations[j])))
                remaining_work.append(work)
            else:
                remaining_work.append(0)

        # Select job with largest remaining work among ready jobs
        ready_jobs = [j for j in range(n_jobs) if next_op[j] < len(durations[j])]
        job_idx = max(ready_jobs, key=lambda j: remaining_work[j])

        op_idx = next_op[job_idx]
        candidates = durations[job_idx][op_idx]

        # Choose machine: load-aware earliest completion
        best_machine = None
        best_score = None  # (completion_time, load, machine_id)
        for m, pt in candidates:
            comp_time = max(machine_ready_time[m], job_ready_time[job_idx]) + pt
            load = machine_load[m]
            score = (comp_time, load, m)
            if best_score is None or score < best_score:
                best_score = score
                best_machine = m
                processing_time = pt

        machine_id = best_machine
        start_time = max(machine_ready_time[machine_id], job_ready_time[job_idx])
        schedule[job_idx][op_idx] = (machine_id, start_time, processing_time)
        end_time = start_time + processing_time

        # Update machine and job state
        machine_ready_time[machine_id] = end_time
        machine_load[machine_id] += processing_time
        job_ready_time[job_idx] = end_time
        next_op[job_idx] += 1
        remaining_ops -= 1

    return schedule
