def fjsp_solver(n_jobs, n_machines, durations):
    # Initialize schedule with None for each operation
    schedule = [[None] * len(durations[j]) for j in range(n_jobs)]
    
    # Machine and job ready times
    machine_ready = [0] * n_machines
    job_ready = [0] * n_jobs
    
    # Next operation index for each job
    next_op = [0] * n_jobs
    
    # Total number of operations remaining
    remaining = sum(len(job) for job in durations)
    
    while remaining > 0:
        # Collect jobs that still have unscheduled operations
        ready_jobs = [j for j in range(n_jobs) if next_op[j] < len(durations[j])]
        
        # Helper to compute total remaining work for a job based on minimum processing times
        def remaining_work(j):
            ops = durations[j][next_op[j]:]
            return sum(min(p[1] for p in op) for op in ops)
        
        # Job selection: fewer eligible machines for current operation, then larger remaining work
        job_idx = min(
            ready_jobs,
            key=lambda j: (
                len(durations[j][next_op[j]]),   # number of eligible machines (smaller is better)
                -remaining_work(j)               # larger remaining work preferred (negative for min)
            )
        )
        
        op_idx = next_op[job_idx]
        candidates = durations[job_idx][op_idx]
        
        # Machine selection: earliest completion time, tie-breaking by processing time then machine id
        machine_id, proc_time = min(
            candidates,
            key=lambda pair: (
                max(machine_ready[pair[0]], job_ready[job_idx]) + pair[1],  # earliest completion time
                pair[1],   # shorter processing time
                pair[0]    # lower machine index
            )
        )
        
        start_time = max(machine_ready[machine_id], job_ready[job_idx])
        schedule[job_idx][op_idx] = (machine_id, start_time, proc_time)
        
        end_time = start_time + proc_time
        machine_ready[machine_id] = end_time
        job_ready[job_idx] = end_time
        
        next_op[job_idx] += 1
        remaining -= 1
    
    return schedule
