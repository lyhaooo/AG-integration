def fjsp_solver(n_jobs, n_machines, durations):
    # Precompute min processing time per operation for quick access
    op_min_time = [[min(pair[1] for pair in candidates) for candidates in job_ops] for job_ops in durations]
    
    # Initialize scheduling structures
    schedule = [[None] * len(durations[j]) for j in range(n_jobs)]
    machine_ready_time = [0] * n_machines
    job_ready_time = [0] * n_jobs
    next_op = [0] * n_jobs
    remaining = sum(len(job) for job in durations)
    
    while remaining > 0:
        # Identify jobs that still have unscheduled operations
        ready_jobs = [j for j in range(n_jobs) if next_op[j] < len(durations[j])]
        
        # Compute total remaining minimum processing time for each ready job
        def remaining_min_sum(j):
            return sum(op_min_time[j][op] for op in range(next_op[j], len(durations[j])))
        
        # Choose job with largest total remaining min processing time;
        # tie-break by smaller job_ready_time (i.e., earliest availability)
        job_idx = max(ready_jobs, key=lambda j: (remaining_min_sum(j), -job_ready_time[j]))
        
        op_idx = next_op[job_idx]
        candidates = durations[job_idx][op_idx]
        
        # Dispatch operation to the machine that minimizes completion time
        machine_id, processing_time = min(
            candidates,
            key=lambda pair: (
                max(machine_ready_time[pair[0]], job_ready_time[job_idx]) + pair[1],
                pair[1],
                pair[0]
            )
        )
        
        start_time = max(machine_ready_time[machine_id], job_ready_time[job_idx])
        schedule[job_idx][op_idx] = (machine_id, start_time, processing_time)
        
        end_time = start_time + processing_time
        machine_ready_time[machine_id] = end_time
        job_ready_time[job_idx] = end_time
        next_op[job_idx] += 1
        remaining -= 1
    
    return schedule
