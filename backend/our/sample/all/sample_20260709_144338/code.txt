def fjsp_solver(n_jobs, n_machines, durations):
    schedule = [[None] * len(durations[j]) for j in range(n_jobs)]
    machine_ready_time = [0] * n_machines
    job_ready_time = [0] * n_jobs
    next_op = [0] * n_jobs
    remaining = sum(len(job) for job in durations)

    # initial bottleneck counts: for each machine, number of future operations that can use it
    bottleneck_count = [0] * n_machines
    for job in durations:
        for op in job:
            for m, _ in op:
                bottleneck_count[m] += 1

    # compute global average processing time for penalty scaling
    total_pt = 0
    total_ops = 0
    for job in durations:
        for op in job:
            for _, pt in op:
                total_pt += pt
                total_ops += 1
    avg_pt = total_pt / total_ops if total_ops > 0 else 1
    penalty_weight = 0.05 * avg_pt  # small relative to processing times

    while remaining > 0:
        # select job with smallest job_ready_time among those with unfinished operations
        ready_jobs = [j for j in range(n_jobs) if next_op[j] < len(durations[j])]
        job_idx = min(ready_jobs, key=lambda j: job_ready_time[j])
        op_idx = next_op[job_idx]
        candidates = durations[job_idx][op_idx]

        # choose machine with composite score: start_time + duration + penalty * bottleneck_count
        def key_fn(pair):
            m, p = pair
            start = max(machine_ready_time[m], job_ready_time[job_idx])
            end = start + p
            penalty = penalty_weight * bottleneck_count[m]
            return (end + penalty, p, m)   # tie-breakers: second processing time, third machine id

        machine_id, processing_time = min(candidates, key=key_fn)

        start_time = max(machine_ready_time[machine_id], job_ready_time[job_idx])
        schedule[job_idx][op_idx] = (machine_id, start_time, processing_time)
        end_time = start_time + processing_time
        machine_ready_time[machine_id] = end_time
        job_ready_time[job_idx] = end_time
        next_op[job_idx] += 1
        remaining -= 1

        # update bottleneck counts: remove this operation's candidate machines
        for m, _ in candidates:
            bottleneck_count[m] -= 1

    return schedule
