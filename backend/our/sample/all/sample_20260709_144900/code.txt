def fjsp_solver(n_jobs, n_machines, durations):
    schedule = [[None] * len(durations[j]) for j in range(n_jobs)]
    machine_ready_time = [0] * n_machines
    job_ready_time = [0] * n_jobs
    next_op = [0] * n_jobs

    # Precompute remaining min work per job per operation
    remaining_min_work = [[0] * (len(durations[j]) + 1) for j in range(n_jobs)]
    for job_idx in range(n_jobs):
        for op_idx in range(len(durations[job_idx]) - 1, -1, -1):
            best_time = min(p for _, p in durations[job_idx][op_idx])
            remaining_min_work[job_idx][op_idx] = remaining_min_work[job_idx][op_idx + 1] + best_time

    ALPHA = 0.85

    def base_job_priority(job_idx):
        op_idx = next_op[job_idx]
        candidates = durations[job_idx][op_idx]
        min_start = min(
            max(machine_ready_time[m], job_ready_time[job_idx])
            for m, _ in candidates
        )
        return min_start

    def elite_job_priority(job_idx):
        op_idx = next_op[job_idx]
        candidates = durations[job_idx][op_idx]

        # Best completion time for current operation
        best_comp = min(
            max(machine_ready_time[m], job_ready_time[job_idx]) + p
            for m, p in candidates
        )

        # One-step lookahead: estimate additional time needed for next operation
        next_lookahead = 0
        if op_idx + 1 < len(durations[job_idx]):
            next_candidates = durations[job_idx][op_idx + 1]
            temp_job_ready = best_comp
            next_best_comp = min(
                max(machine_ready_time[m], temp_job_ready) + p
                for m, p in next_candidates
            )
            next_lookahead = next_best_comp - best_comp

        min_start = min(
            max(machine_ready_time[m], job_ready_time[job_idx])
            for m, _ in candidates
        )
        optimistic_completion = min_start + remaining_min_work[job_idx][op_idx]
        avg_machine_load = sum(machine_ready_time[m] for m, _ in candidates) / len(candidates)
        return (
            min_start
            - 0.2 * remaining_min_work[job_idx][op_idx]
            + 0.3 * avg_machine_load
            + 0.1 * optimistic_completion
            + 0.5 * next_lookahead
        )

    def job_priority(job_idx):
        hybrid = ALPHA * elite_job_priority(job_idx) + (1 - ALPHA) * base_job_priority(job_idx)
        return (hybrid, job_idx)

    def base_machine_priority(job_idx, machine_id, processing_time):
        start_time = max(machine_ready_time[machine_id], job_ready_time[job_idx])
        return start_time + processing_time

    def elite_machine_priority(job_idx, machine_id, processing_time):
        op_idx = next_op[job_idx]
        start_time = max(machine_ready_time[machine_id], job_ready_time[job_idx])
        completion_time = start_time + processing_time

        # One-step lookahead: pessimistic next operation completion
        lookahead_completion = 0
        if op_idx + 1 < len(durations[job_idx]):
            next_candidates = durations[job_idx][op_idx + 1]
            next_completions = [
                max(machine_ready_time[next_machine], completion_time) + next_time
                for next_machine, next_time in next_candidates
            ]
            if next_completions:
                lookahead_completion = max(next_completions)

        avg_ready = sum(machine_ready_time) / n_machines if n_machines > 0 else 0
        load_bias = machine_ready_time[machine_id] - avg_ready
        min_p = min(p for _, p in durations[job_idx][op_idx])
        congestion_penalty = 1.0 * max(0, load_bias) ** 2 if load_bias > 0 else 0

        # Adjusted coefficients: stronger lookahead influence, milder load penalty
        return (
            completion_time
            + 1.2 * lookahead_completion
            + 0.5 * (min_p - processing_time)
            + 3.5 * load_bias
            + 0.2 * congestion_penalty
        )

    def machine_priority(job_idx, machine_id, processing_time):
        hybrid = ALPHA * elite_machine_priority(job_idx, machine_id, processing_time) + \
                 (1 - ALPHA) * base_machine_priority(job_idx, machine_id, processing_time)
        return (hybrid, machine_ready_time[machine_id], processing_time, machine_id)

    remaining = sum(len(job) for job in durations)
    while remaining > 0:
        ready_jobs = [j for j in range(n_jobs) if next_op[j] < len(durations[j])]
        job_idx = min(ready_jobs, key=job_priority)
        op_idx = next_op[job_idx]
        candidates = durations[job_idx][op_idx]
        machine_id, processing_time = min(
            candidates,
            key=lambda pair: machine_priority(job_idx, pair[0], pair[1]),
        )
        start_time = max(machine_ready_time[machine_id], job_ready_time[job_idx])
        schedule[job_idx][op_idx] = (machine_id, start_time, processing_time)
        end_time = start_time + processing_time
        machine_ready_time[machine_id] = end_time
        job_ready_time[job_idx] = end_time
        next_op[job_idx] += 1
        remaining -= 1

    return schedule
