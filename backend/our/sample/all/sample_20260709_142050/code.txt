def fjsp_solver(n_jobs, n_machines, durations):
    # Pre-compute minimal processing time per operation
    min_pt = []
    for job in durations:
        job_min = []
        for op in job:
            job_min.append(min(pair[1] for pair in op))
        min_pt.append(job_min)

    # Pre-compute cumulative remaining work from each position
    cum_rem = []
    for job_min in min_pt:
        cum = [0] * (len(job_min) + 1)
        for i in range(len(job_min)-1, -1, -1):
            cum[i] = cum[i+1] + job_min[i]
        cum_rem.append(cum)

    schedule = [[None] * len(durations[j]) for j in range(n_jobs)]
    machine_ready_time = [0] * n_machines
    job_ready_time = [0] * n_jobs
    next_op = [0] * n_jobs
    remaining_ops = sum(len(job) for job in durations)

    while remaining_ops > 0:
        # Identify jobs with unscheduled operations
        ready_jobs = [j for j in range(n_jobs) if next_op[j] < len(durations[j])]

        # Job selection: fewest eligible machines first, then larger remaining work
        def job_key(j):
            # Primary: number of machines eligible for current operation
            n_mach = len(durations[j][next_op[j]])
            # Secondary: larger remaining work (negate to sort descending)
            rem = cum_rem[j][next_op[j]]
            return (n_mach, -rem)

        job_idx = min(ready_jobs, key=job_key)
        op_idx = next_op[job_idx]
        candidates = durations[job_idx][op_idx]

        # Machine selection: minimize completion time, breaking ties by duration then machine id
        def machine_key(pair):
            m, t = pair
            start = max(machine_ready_time[m], job_ready_time[job_idx])
            return (start + t, t, m)

        machine_id, processing_time = min(candidates, key=machine_key)
        start_time = max(machine_ready_time[machine_id], job_ready_time[job_idx])
        schedule[job_idx][op_idx] = (machine_id, start_time, processing_time)
        end_time = start_time + processing_time

        machine_ready_time[machine_id] = end_time
        job_ready_time[job_idx] = end_time
        next_op[job_idx] += 1
        remaining_ops -= 1

    return schedule
