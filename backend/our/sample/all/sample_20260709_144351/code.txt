def fjsp_solver(n_jobs, n_machines, durations):
    # Schedule initialization
    schedule = [[None] * len(durations[j]) for j in range(n_jobs)]
    machine_ready_time = [0] * n_machines
    job_ready_time = [0] * n_jobs
    next_op = [0] * n_jobs
    remaining = sum(len(job) for job in durations)

    while remaining > 0:
        # Select the job with the earliest ready time among those with pending operations
        ready_jobs = [j for j in range(n_jobs) if next_op[j] < len(durations[j])]
        job_idx = min(ready_jobs, key=lambda j: job_ready_time[j])
        op_idx = next_op[job_idx]

        candidates = durations[job_idx][op_idx]

        # Machine selection: minimize earliest completion time + current machine load
        # Current machine load is machine_ready_time[m] (cumulative processing time)
        best_machine, best_time = min(
            candidates,
            key=lambda pair: (
                max(machine_ready_time[pair[0]], job_ready_time[job_idx])
                + pair[1]
                + machine_ready_time[pair[0]],  # add load to ECT
                pair[1],
                pair[0],
            ),
        )

        start_time = max(machine_ready_time[best_machine], job_ready_time[job_idx])
        schedule[job_idx][op_idx] = (best_machine, start_time, best_time)
        end_time = start_time + best_time

        # Update state
        machine_ready_time[best_machine] = end_time
        job_ready_time[job_idx] = end_time
        next_op[job_idx] += 1
        remaining -= 1

    return schedule
