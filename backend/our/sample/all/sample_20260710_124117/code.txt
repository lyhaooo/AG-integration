def fjsp_solver(n_jobs, n_machines, durations):
    # Precompute minimal processing time for each operation
    min_durations = [[min(p for _, p in op) for op in job] for job in durations]

    # Scheduling data structures
    schedule = [[None] * len(job) for job in durations]
    machine_ready = [0] * n_machines
    job_ready = [0] * n_jobs
    next_op = [0] * n_jobs
    remaining = sum(len(job) for job in durations)

    while remaining > 0:
        # Determine jobs that still have unscheduled operations
        ready_jobs = [j for j in range(n_jobs) if next_op[j] < len(durations[j])]

        # Compute selection keys for each ready job
        def job_key(j):
            op_idx = next_op[j]
            # Number of eligible machines for current operation (smaller is better)
            num_machines = len(durations[j][op_idx])
            # Remaining work: sum of minimal durations of yet-unscheduled operations (larger is better)
            rem_work = sum(min_durations[j][k] for k in range(op_idx, len(durations[j])))
            # Tie‑breaking by job index (deterministic)
            return (num_machines, -rem_work, j)

        # Select job according to the priority rule
        job_idx = min(ready_jobs, key=job_key)

        op_idx = next_op[job_idx]
        candidates = durations[job_idx][op_idx]

        # Choose machine: best candidate based on earliest completion time
        # (preserving the same machine selection logic as the skeleton)
        best_machine, best_processing = min(
            candidates,
            key=lambda pair: (
                max(machine_ready[pair[0]], job_ready[job_idx]) + pair[1],
                pair[1],
                pair[0]
            )
        )

        start_time = max(machine_ready[best_machine], job_ready[job_idx])
        end_time = start_time + best_processing

        # Store assignment
        schedule[job_idx][op_idx] = (best_machine, start_time, best_processing)

        # Update state
        machine_ready[best_machine] = end_time
        job_ready[job_idx] = end_time
        next_op[job_idx] += 1
        remaining -= 1

    return schedule
