def fjsp_solver(n_jobs, n_machines, durations):
    # ---------- helper data ----------
    n_ops = [len(job) for job in durations]
    total_ops = sum(n_ops)
    # Precompute remaining minimum work for each job from each operation
    remaining_min_work = [[0] * (n_ops[j] + 1) for j in range(n_jobs)]
    for j in range(n_jobs):
        for op in range(n_ops[j] - 1, -1, -1):
            best = min(durations[j][op], key=lambda x: x[1])[1]
            remaining_min_work[j][op] = best + remaining_min_work[j][op + 1]

    ALPHA = 0.85

    # ---------- top-level priority functions (not used directly, only for reference) ----------
    # (kept to preserve structure, but inner functions are used)

    # ---------- constructive scheduler ----------
    def build_schedule(forced_machine=None):
        # reinitialize state
        schedule = [[None] * n_ops[j] for j in range(n_jobs)]
        mrt = [0] * n_machines
        jrt = [0] * n_jobs
        nxt = [0] * n_jobs
        rem = total_ops

        # ---------- job priority ----------
        def base_job_priority(job_idx):
            op_idx = nxt[job_idx]
            candidates = durations[job_idx][op_idx]
            min_start = min(
                max(mrt[m], jrt[job_idx]) for m, _ in candidates
            )
            return min_start

        def elite_job_priority(job_idx):
            op_idx = nxt[job_idx]
            candidates = durations[job_idx][op_idx]
            min_start = min(
                max(mrt[m], jrt[job_idx]) for m, _ in candidates
            )
            optimistic_completion = min_start + remaining_min_work[job_idx][op_idx]
            avg_machine_load = sum(mrt[m] for m, _ in candidates) / len(candidates) if candidates else 0
            return min_start - 0.3 * remaining_min_work[job_idx][op_idx] + 0.2 * avg_machine_load + 0.25 * optimistic_completion

        def job_priority(job_idx):
            hybrid = ALPHA * elite_job_priority(job_idx) + (1 - ALPHA) * base_job_priority(job_idx)
            return (hybrid, job_idx)

        # ---------- mutated machine priority ----------
        def base_machine_priority(job_idx, machine_id, processing_time):
            start_time = max(mrt[machine_id], jrt[job_idx])
            return start_time + processing_time

        def mutated_machine_priority(job_idx, machine_id, processing_time):
            op_idx = nxt[job_idx]
            start_time = max(mrt[machine_id], jrt[job_idx])
            completion = start_time + processing_time
            # total load on this machine if we take this operation
            machine_load = mrt[machine_id] + processing_time
            # remaining work of the job from this operation onward
            job_rem = remaining_min_work[job_idx][op_idx]
            # ---- tuned weights to reduce bottleneck machine congestion ----
            # Increase penalty for machine load (0.6 instead of 0.4)
            # Change processing_time coefficient from -0.2 to +0.2 to penalize long ops
            score = completion + 0.6 * machine_load + 0.2 * job_rem + 0.2 * processing_time
            return score

        def machine_priority(job_idx, machine_id, processing_time):
            hybrid = ALPHA * mutated_machine_priority(job_idx, machine_id, processing_time) + \
                     (1 - ALPHA) * base_machine_priority(job_idx, machine_id, processing_time)
            # tie-breaker: prefer machines with lower ready time and shorter processing time
            return (hybrid, mrt[machine_id], processing_time, machine_id)

        # ---------- scheduling loop ----------
        while rem > 0:
            ready_jobs = [j for j in range(n_jobs) if nxt[j] < n_ops[j]]
            # choose job
            job_idx = min(ready_jobs, key=job_priority)
            op_idx = nxt[job_idx]
            candidates = durations[job_idx][op_idx]

            # check forced machine
            if forced_machine is not None and (job_idx, op_idx) in forced_machine:
                forced_m_id = forced_machine[(job_idx, op_idx)]
                selected = None
                for m_id, p_time in candidates:
                    if m_id == forced_m_id:
                        selected = (m_id, p_time)
                        break
                if selected is None:
                    # forced machine not available -> infeasible
                    return None
                machine_id, processing_time = selected
            else:
                # normal selection using mutated machine priority
                machine_id, processing_time = min(candidates, key=lambda pair: machine_priority(job_idx, pair[0], pair[1]))

            start_time = max(mrt[machine_id], jrt[job_idx])
            schedule[job_idx][op_idx] = (machine_id, start_time, processing_time)
            end_time = start_time + processing_time
            mrt[machine_id] = end_time
            jrt[job_idx] = end_time
            nxt[job_idx] += 1
            rem -= 1

        return schedule

    # build schedule without forced machines
    result = build_schedule(forced_machine=None)
    return result
