def fjsp_solver(n_jobs, n_machines, durations):
    import heapq

    # Initialize schedules and next available times for each machine
    schedule = [[None] * len(durations[job_idx]) for job_idx in range(n_jobs)]
    machine_available_times = [0] * n_machines

    # Create a list of operations with their job index, operation index, machine, and processing time
    operations = []
    for job_idx in range(n_jobs):
        for op_idx, (machine_id, processing_time) in enumerate(durations[job_idx]):
            operations.append((job_idx, op_idx, machine_id, processing_time))

    # Sort operations based on the shortest processing time first (for initial assignment)
    operations.sort(key=lambda x: x[3])

    # Schedule each operation
    for job_idx, op_idx, machine_id, processing_time in operations:
        # Check the available start time for this machine
        start_time = machine_available_times[machine_id]
        
        # Check for job precedence
        if op_idx > 0:  # Not the first operation
            prev_machine, prev_duration = schedule[job_idx][op_idx - 1] if schedule[job_idx][op_idx - 1] else (None, None)
            start_time = max(start_time, schedule[job_idx][op_idx - 1][1] if schedule[job_idx][op_idx - 1] else 0)
        
        # Schedule the operation
        end_time = start_time + processing_time
        schedule[job_idx][op_idx] = (machine_id, start_time, processing_time)
        
        # Update the machine's next available time
        machine_available_times[machine_id] = end_time

    return schedule
