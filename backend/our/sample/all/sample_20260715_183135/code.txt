def fjsp_solver(n_jobs, n_machines, durations):
    # Precompute minimum processing time for each operation
    min_time_per_op = []
    for job in durations:
        job_mins = []
        for candidates in job:
            job_mins.append(min(p for _, p in candidates))
        min_time_per_op.append(job_mins)

    # Total remaining minimum processing time for each job
    total_remaining_min = [sum(mins) for mins in min_time_per_op]

    # Scheduling structures
    schedule = [[None] * len(durations[j]) for j in range(n_jobs)]
    machine_ready_time = [0] * n_machines
    job_ready_time = [0] * n_jobs
    next_op = [0] * n_jobs
    remaining = sum(len(job) for job in durations)

    while remaining > 0:
        # Select ready job with largest total remaining minimum processing time
        ready_jobs = [j for j in range(n_jobs) if next_op[j] < len(durations[j])]
        # Tie-break by job_ready_time (earlier is better) then job index
        job_idx = max(
            ready_jobs,
            key=lambda j: (total_remaining_min[j], -job_ready_time[j], -j)
        )

        op_idx = next_op[job_idx]
        candidates = durations[job_idx][op_idx]

        # Choose machine: minimize completion time, then processing time, then machine id
        machine_id, processing_time = min(
            candidates,
            key=lambda pair: (
                max(machine_ready_time[pair[0]], job_ready_time[job_idx]) + pair[1],
                pair[1],
                pair[0]
            )
        )

        start_time = max(machine_ready_time[machine_id], job_ready_time[job_idx])
        schedule[job_idx][op_idx] = (machine_id, start_time, processing_time)
        end_time = start_time + processing_time

        machine_ready_time[machine_id] = end_time
        job_ready_time[job_idx] = end_time
        next_op[job_idx] += 1

        # Update total remaining minimum processing time for this job
        total_remaining_min[job_idx] -= min_time_per_op[job_idx][op_idx]

        remaining -= 1

    return schedule
