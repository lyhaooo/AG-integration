def fjsp_solver(n_jobs, n_machines, durations):
    # Initialize schedule structure
    schedule = [[None] * len(durations[j]) for j in range(n_jobs)]
    machine_ready_time = [0] * n_machines
    job_ready_time = [0] * n_jobs
    next_op = [0] * n_jobs
    remaining = sum(len(job) for job in durations)

    # Helper to compute remaining work (sum of min processing times) from a given operation index
    def remaining_work(job_idx, start_op):
        total = 0
        for op_idx in range(start_op, len(durations[job_idx])):
            min_time = min(candidates[1] for candidates in durations[job_idx][op_idx])
            total += min_time
        return total

    while remaining > 0:
        # Identify ready jobs (still have unscheduled operations)
        ready_jobs = [j for j in range(n_jobs) if next_op[j] < len(durations[j])]

        # Select job according to variant 3:
        # 1) Fewer eligible machines for the current operation (primary)
        # 2) Larger remaining work (secondary)
        # 3) Job index as tie-breaker for determinism
        def job_key(j):
            op = next_op[j]
            num_machines = len(durations[j][op])
            rem = remaining_work(j, op)
            return (num_machines, -rem, j)

        job_idx = min(ready_jobs, key=job_key)

        op_idx = next_op[job_idx]
        candidates = durations[job_idx][op_idx]

        # Machine selection (same as skeleton): minimize start time + processing time
        machine_id, processing_time = min(
            candidates,
            key=lambda pair: (
                max(machine_ready_time[pair[0]], job_ready_time[job_idx]) + pair[1],
                pair[1],
                pair[0]
            )
        )

        start_time = max(machine_ready_time[machine_id], job_ready_time[job_idx])
        end_time = start_time + processing_time

        # Record the operation
        schedule[job_idx][op_idx] = (machine_id, start_time, processing_time)

        # Update times
        machine_ready_time[machine_id] = end_time
        job_ready_time[job_idx] = end_time
        next_op[job_idx] += 1
        remaining -= 1

    return schedule
