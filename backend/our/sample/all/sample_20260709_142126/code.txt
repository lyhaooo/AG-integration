def fjsp_solver(n_jobs, n_machines, durations):
    # Precompute minimum processing time for each operation
    min_durations = [[0]*len(durations[j]) for j in range(n_jobs)]
    total_min_processing = [0]*n_jobs
    for j in range(n_jobs):
        for op_idx, candidates in enumerate(durations[j]):
            min_time = min(p[1] for p in candidates)
            min_durations[j][op_idx] = min_time
            total_min_processing[j] += min_time

    schedule = [[None]*len(durations[j]) for j in range(n_jobs)]
    machine_ready_time = [0]*n_machines
    job_ready_time = [0]*n_jobs
    next_op = [0]*n_jobs
    remaining_work = total_min_processing[:]   # based on min processing times
    remaining_ops = sum(len(job) for job in durations)

    while remaining_ops > 0:
        # Determine ready jobs
        ready_jobs = [j for j in range(n_jobs) if next_op[j] < len(durations[j])]

        # Select job: fewer eligible machines first, then larger remaining work, then earlier ready time
        def job_key(j):
            op_idx = next_op[j]
            n_machines = len(durations[j][op_idx])
            return (n_machines, -remaining_work[j], job_ready_time[j], j)

        job_idx = min(ready_jobs, key=job_key)

        op_idx = next_op[job_idx]
        candidates = durations[job_idx][op_idx]

        # Choose machine: earliest possible finish time, tie-break by processing time, then machine id
        machine_id, processing_time = min(
            candidates,
            key=lambda pair: (
                max(machine_ready_time[pair[0]], job_ready_time[job_idx]) + pair[1],
                pair[1],
                pair[0]
            )
        )
        start_time = max(machine_ready_time[machine_id], job_ready_time[job_idx])
        schedule[job_idx][op_idx] = (machine_id, start_time, processing_time)
        end_time = start_time + processing_time

        machine_ready_time[machine_id] = end_time
        job_ready_time[job_idx] = end_time
        # Update remaining work (based on min processing times)
        remaining_work[job_idx] -= min_durations[job_idx][op_idx]
        next_op[job_idx] += 1
        remaining_ops -= 1

    return schedule
