def fjsp_solver(n_jobs, n_machines, durations):
    schedule = [[None] * len(durations[j]) for j in range(n_jobs)]
    machine_ready_time = [0] * n_machines
    job_ready_time = [0] * n_jobs
    next_op = [0] * n_jobs
    remaining = sum(len(job) for job in durations)

    while remaining > 0:
        ready_jobs = [j for j in range(n_jobs) if next_op[j] < len(durations[j])]
        job_idx = min(ready_jobs, key=lambda j: job_ready_time[j])
        op_idx = next_op[job_idx]
        candidates = durations[job_idx][op_idx]

        best_machine = None
        best_start_time = float('inf')
        best_processing_time = float('inf')
        
        for machine_id, processing_time in candidates:
            # Predict next operation's best completion time
            next_op_idx = next_op[job_idx] + 1
            if next_op_idx < len(durations[job_idx]):
                next_candidates = durations[job_idx][next_op_idx]
                next_completion_times = [
                    max(machine_ready_time[next_machine], job_ready_time[job_idx]) + next_duration
                    for next_machine, next_duration in next_candidates
                ]
                best_next_completion = min(next_completion_times) if next_completion_times else float('inf')
            else:
                best_next_completion = 0

            start_time = max(machine_ready_time[machine_id], job_ready_time[job_idx])
            if start_time + processing_time + best_next_completion < best_start_time:
                best_start_time = start_time + processing_time + best_next_completion
                best_machine = machine_id
                best_processing_time = processing_time

        start_time = max(machine_ready_time[best_machine], job_ready_time[job_idx])
        schedule[job_idx][op_idx] = (best_machine, start_time, best_processing_time)
        end_time = start_time + best_processing_time
        machine_ready_time[best_machine] = end_time
        job_ready_time[job_idx] = end_time
        next_op[job_idx] += 1
        remaining -= 1

    return schedule
