def fjsp_solver(n_jobs, n_machines, durations):
    # ---------- helper data ----------
    n_ops = [len(job) for job in durations]
    total_ops = sum(n_ops)
    # Precompute remaining minimum work for each job from each operation
    remaining_min_work = [[0] * (n_ops[j] + 1) for j in range(n_jobs)]
    for j in range(n_jobs):
        for op in range(n_ops[j] - 1, -1, -1):
            best = min(durations[j][op], key=lambda x: x[1])[1]
            remaining_min_work[j][op] = best + remaining_min_work[j][op + 1]

    ALPHA = 0.85

    # ---------- priority functions (identical to elite parent except machine) ----------
    def base_job_priority(job_idx):
        op_idx = next_op[job_idx]
        candidates = durations[job_idx][op_idx]
        min_start = min(
            max(machine_ready_time[m], job_ready_time[job_idx])
            for m, _ in candidates
        )
        return min_start

    def elite_job_priority(job_idx):
        op_idx = next_op[job_idx]
        candidates = durations[job_idx][op_idx]
        min_start = min(
            max(machine_ready_time[m], job_ready_time[job_idx])
            for m, _ in candidates
        )
        optimistic_completion = min_start + remaining_min_work[job_idx][op_idx]
        avg_machine_load = sum(machine_ready_time[m] for m, _ in candidates) / len(candidates)
        return min_start - 0.3 * remaining_min_work[job_idx][op_idx] + 0.2 * avg_machine_load + 0.25 * optimistic_completion

    def job_priority(job_idx):
        hybrid = ALPHA * elite_job_priority(job_idx) + (1 - ALPHA) * base_job_priority(job_idx)
        return (hybrid, job_idx)

    # NEW machine scoring – concretely different from elite
    def new_machine_priority(job_idx, machine_id, processing_time):
        op_idx = next_op[job_idx]
        start_time = max(machine_ready_time[machine_id], job_ready_time[job_idx])
        completion = start_time + processing_time
        # total load on this machine if we take this operation
        machine_load = machine_ready_time[machine_id] + processing_time
        # remaining work of the job from this operation onward
        job_rem = remaining_min_work[job_idx][op_idx]
        # penalty for high machine load, reward for processing jobs with large remaining work
        score = completion + 0.4 * machine_load + 0.3 * job_rem - 0.2 * processing_time
        return score

    def base_machine_priority(job_idx, machine_id, processing_time):
        start_time = max(machine_ready_time[machine_id], job_ready_time[job_idx])
        return start_time + processing_time

    def machine_priority(job_idx, machine_id, processing_time):
        hybrid = ALPHA * new_machine_priority(job_idx, machine_id, processing_time) + \
                 (1 - ALPHA) * base_machine_priority(job_idx, machine_id, processing_time)
        return (hybrid, machine_ready_time[machine_id], processing_time, machine_id)

    # ---------- constructive scheduler ----------
    def build_schedule(forced_machine=None):
        """
        forced_machine: dict {(job, op): machine_id} to fix a specific machine for that operation.
        Returns (schedule, makespan) or None if forced machine not available.
        """
        # reinitialize state
        sched = [[None] * n_ops[j] for j in range(n_jobs)]
        mrt = [0] * n_machines
        jrt = [0] * n_jobs
        nxt = [0] * n_jobs
        rem = total_ops

        # auxiliary for job_priority and machine_priority (they need these variables)
        nonlocal_ref = {'machine_ready_time': mrt, 'job_ready_time': jrt, 'next_op': nxt,
                        'durations': durations, 'remaining_min_work': remaining_min_work,
                        'ALPHA': ALPHA}
        # We must wrap the helpers to use the local mrt, jrt, nxt
        # Define them again inside this scope to capture the current state
        def local_base_job_priority(job_idx):
            op_idx = nxt[job_idx]
            candidates = durations[job_idx][op_idx]
            min_start = min(max(mrt[m], jrt[job_idx]) for m, _ in candidates)
            return min_start

        def local_elite_job_priority(job_idx):
            op_idx = nxt[job_idx]
            candidates = durations[job_idx][op_idx]
            min_start = min(max(mrt[m], jrt[job_idx]) for m, _ in candidates)
            optimistic_completion = min_start + remaining_min_work[job_idx][op_idx]
            avg_machine_load = sum(mrt[m] for m, _ in candidates) / len(candidates) if candidates else 0
            return min_start - 0.3 * remaining_min_work[job_idx][op_idx] + 0.2 * avg_machine_load + 0.25 * optimistic_completion

        def local_job_priority(job_idx):
            hybrid = ALPHA * local_elite_job_priority(job_idx) + (1 - ALPHA) * local_base_job_priority(job_idx)
            return (hybrid, job_idx)

        def local_new_machine_priority(job_idx, machine_id, processing_time):
            op_idx = nxt[job_idx]
            start_time = max(mrt[machine_id], jrt[job_idx])
            completion = start_time + processing_time
            machine_load = mrt[machine_id] + processing_time
            job_rem = remaining_min_work[job_idx][op_idx]
            return completion + 0.4 * machine_load + 0.3 * job_rem - 0.2 * processing_time

        def local_base_machine_priority(job_idx, machine_id, processing_time):
            start_time = max(mrt[machine_id], jrt[job_idx])
            return start_time + processing_time

        def local_machine_priority(job_idx, machine_id, processing_time):
            hybrid = ALPHA * local_new_machine_priority(job_idx, machine_id, processing_time) + \
                     (1 - ALPHA) * local_base_machine_priority(job_idx, machine_id, processing_time)
            return (hybrid, mrt[machine_id], processing_time, machine_id)

        while rem > 0:
            ready_jobs = [j for j in range(n_jobs) if nxt[j] < n_ops[j]]
            # choose job
            job_idx = min(ready_jobs, key=local_job_priority)
            op_idx = nxt[job_idx]
            candidates = durations[job_idx][op_idx]

            # check forced machine
            if forced_machine is not None and (job_idx, op_idx) in forced_machine:
                forced_m_id = forced_machine[(job_idx, op_idx)]
                # find the processing time for that machine
                selected = None
                for m_id, p_time in candidates:
                    if m_id == forced_m_id:
                        selected = (m_id, p_time)
                        break
                if selected is None:
                    # forced machine not available -> infeasible
                    return None
                machine_id, processing_time = selected
            else:
                # normal selection
                machine_id, processing_time = min(candidates, key=lambda pair: local_machine_priority(job_idx, pair[0], pair[1]))

            start_time = max(mrt[machine_id], jrt[job_idx])
            sched[job_idx][op_idx] = (machine_id, start_time, processing_time)
            end_time = start_time + processing_time
            mrt[machine_id] = end_time
            jrt[job_idx] = end_time
            nxt[job_idx] += 1
            rem -= 1

        makespan = max(jrt)
        return sched, makespan

    # ---------- initial solution ----------
    initial = build_schedule()
    if initial is None:
        # fallback (should not happen)
        return [[(0,0,0)] * n_ops[j] for j in range(n_jobs)]
    best_schedule, best_makespan = initial

    # ---------- critical path identification ----------
    def compute_critical_path(schedule, makespan):
        # build a graph: each operation knows its predecessor on same job and same machine
        # We'll trace from the job(s) that finish at makespan
        # First, record for each operation: end_time, job_prev, machine_prev
        job_end = []  # list of (job_idx, end_time) for last operation of each job
        op_info = {}  # (j,op) -> {'start': start, 'end': end, 'machine': m, 'duration': d}
        for j in range(n_jobs):
            for op in range(n_ops[j]):
                m, st, dur = schedule[j][op]
                et = st + dur
                op_info[(j, op)] = {'start': st, 'end': et, 'machine': m, 'duration': dur}
            if n_ops[j] > 0:
                job_end.append((j, op_info[(j, n_ops[j]-1)]['end']))
        # find the job(s) with max end (makespan)
        max_end = max(end for _, end in job_end)
        critical_job_end = [j for j, end in job_end if end == max_end]

        # For each machine, build list of operations sorted by start time
        machine_ops = [[] for _ in range(n_machines)]
        for j in range(n_jobs):
            for op in range(n_ops[j]):
                m = op_info[(j, op)]['machine']
                machine_ops[m].append((j, op, op_info[(j, op)]['start'], op_info[(j, op)]['end']))
        for m in range(n_machines):
            machine_ops[m].sort(key=lambda x: x[2])  # sort by start

        # Build predecessor links
        # job predecessor: previous op in same job
        # machine predecessor: operation on same machine that finishes before this one starts
        predecessors = {}  # (j,op) -> list of (j_prev, op_prev)
        for j in range(n_jobs):
            for op in range(n_ops[j]):
                preds = []
                if op > 0:
                    preds.append((j, op-1))
                # find machine predecessor: the operation on same machine with max end time <= start
                m = op_info[(j, op)]['machine']
                start = op_info[(j, op)]['start']
                # scan machine list in reverse
                for mj, mop, _, end in reversed(machine_ops[m]):
                    if end <= start and not (mj == j and mop == op):  # avoid self
                        preds.append((mj, mop))
                        break
                predecessors[(j, op)] = preds

        # Reconstruct critical path by backtracking from each finishing job
        critical_path = []
        # We'll use DFS from finish to start, longest path
        # Since we only need one critical path, we can trace backwards using longest path
        # Simple: for each finish node, follow the predecessor with latest end time?
        # Actually we want the path that determines makespan. Typically, start from a finishing job and move to predecessor with largest end time.
        for j in critical_job_end[:1]:  # take first finishing job
            cur = (j, n_ops[j]-1)
            critical_path.append(cur)
            while True:
                preds = predecessors[cur]
                if not preds:
                    break
                # choose predecessor with maximum end time
                best = max(preds, key=lambda p: op_info[p]['end'])
                critical_path.append(best)
                cur = best
            break  # only one critical path
        critical_path.reverse()  # now from start to end
        return critical_path

    # ---------- local search on critical path ----------
    def local_search(schedule, makespan):
        max_iter = 5
        for _ in range(max_iter):
            cp = compute_critical_path(schedule, makespan)
            improved = False
            for (job_idx, op_idx) in cp:
                candidates = durations[job_idx][op_idx]
                current_machine = schedule[job_idx][op_idx][0]
                # try all alternative machines (including current to check if something else better)
                for m_id, p_time in candidates:
                    if m_id == current_machine:
                        continue
                    # force this operation to machine m_id
                    forced = {(job_idx, op_idx): m_id}
                    result = build_schedule(forced_machine=forced)
                    if result is None:
                        continue
                    new_schedule, new_makespan = result
                    if new_makespan < makespan:
                        # accept improvement
                        schedule, makespan = new_schedule, new_makespan
                        improved = True
                        break  # take first improvement
                if improved:
                    break  # restart critical path from new schedule
            if not improved:
                break
        return schedule, makespan

    # apply local search
    best_schedule, best_makespan = local_search(best_schedule, best_makespan)
    return best_schedule
