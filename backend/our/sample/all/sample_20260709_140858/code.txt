def fjsp_solver(n_jobs, n_machines, durations):
    # Initialize schedule as a list of lists with None placeholders.
    schedule = [[None] * len(durations[j]) for j in range(n_jobs)]

    # Tracking structures
    machine_ready_time = [0] * n_machines
    machine_load = [0] * n_machines          # total processing time assigned so far
    job_ready_time = [0] * n_jobs
    next_op = [0] * n_jobs
    remaining = sum(len(job) for job in durations)

    while remaining > 0:
        # Jobs that still have unscheduled operations
        ready_jobs = [j for j in range(n_jobs) if next_op[j] < len(durations[j])]
        # Choose job with earliest ready time (FIFO-like)
        job_idx = min(ready_jobs, key=lambda j: job_ready_time[j])

        op_idx = next_op[job_idx]
        candidates = durations[job_idx][op_idx]

        # Machine selection: minimize (earliest completion time + current machine load)
        # Tie-breaking: shorter processing time, then lower machine id.
        machine_id, processing_time = min(
            candidates,
            key=lambda pair: (
                max(machine_ready_time[pair[0]], job_ready_time[job_idx]) + pair[1] + machine_load[pair[0]],
                pair[1],
                pair[0]
            )
        )

        # Compute start time and schedule
        start_time = max(machine_ready_time[machine_id], job_ready_time[job_idx])
        schedule[job_idx][op_idx] = (machine_id, start_time, processing_time)

        # Update state
        end_time = start_time + processing_time
        machine_ready_time[machine_id] = end_time
        machine_load[machine_id] += processing_time
        job_ready_time[job_idx] = end_time
        next_op[job_idx] += 1
        remaining -= 1

    return schedule
