def fjsp_solver(n_jobs, n_machines, durations):
    # ---------- precompute remaining min work ----------
    n_ops = [len(job) for job in durations]
    total_ops = sum(n_ops)
    remaining_min_work = [[0] * (n_ops[j] + 1) for j in range(n_jobs)]
    for j in range(n_jobs):
        for op in range(n_ops[j] - 1, -1, -1):
            best = min(durations[j][op], key=lambda x: x[1])[1]
            remaining_min_work[j][op] = best + remaining_min_work[j][op + 1]

    ALPHA = 0.85

    # ---------- job priority (adjusted lookahead weights) ----------
    def base_job_priority(job_idx, mrt, jrt, nxt):
        op_idx = nxt[job_idx]
        candidates = durations[job_idx][op_idx]
        min_start = min(max(mrt[m], jrt[job_idx]) for m, _ in candidates)
        return min_start

    def elite_job_priority(job_idx, mrt, jrt, nxt):
        op_idx = nxt[job_idx]
        candidates = durations[job_idx][op_idx]
        min_start = min(max(mrt[m], jrt[job_idx]) for m, _ in candidates)
        optimistic_completion = min_start + remaining_min_work[job_idx][op_idx]
        avg_machine_load = sum(mrt[m] for m, _ in candidates) / len(candidates) if candidates else 0
        # increased weight on remaining work and optimistic completion
        return min_start - 0.5 * remaining_min_work[job_idx][op_idx] + 0.2 * avg_machine_load + 0.35 * optimistic_completion

    def job_priority(job_idx, mrt, jrt, nxt):
        hybrid = ALPHA * elite_job_priority(job_idx, mrt, jrt, nxt) + (1 - ALPHA) * base_job_priority(job_idx, mrt, jrt, nxt)
        return (hybrid, job_idx)

    # ---------- machine priority with one-step lookahead ----------
    def new_machine_priority(job_idx, machine_id, processing_time, mrt, jrt, nxt):
        op_idx = nxt[job_idx]
        start_time = max(mrt[machine_id], jrt[job_idx])
        completion = start_time + processing_time
        # one-step lookahead: estimate effect on next operation
        lookahead = 0.0
        if op_idx + 1 < n_ops[job_idx]:
            next_candidates = durations[job_idx][op_idx + 1]
            # temporarily update machine ready time for this machine
            new_mrt = mrt.copy()
            new_mrt[machine_id] = completion
            new_jrt = completion  # job is ready after this operation
            # find min earliest start for next operation given updated state
            best_next_start = float('inf')
            for m_next, p_next in next_candidates:
                start_next = max(new_mrt[m_next], new_jrt)
                best_next_start = min(best_next_start, start_next)
            # add a penalty proportional to the next start time
            lookahead = 0.3 * best_next_start

        machine_load = mrt[machine_id] + processing_time
        job_rem = remaining_min_work[job_idx][op_idx]
        score = completion + 0.4 * machine_load + 0.3 * job_rem - 0.2 * processing_time + lookahead
        return score

    def base_machine_priority(job_idx, machine_id, processing_time, mrt, jrt, nxt):
        start_time = max(mrt[machine_id], jrt[job_idx])
        return start_time + processing_time

    def machine_priority(job_idx, machine_id, processing_time, mrt, jrt, nxt):
        hybrid = ALPHA * new_machine_priority(job_idx, machine_id, processing_time, mrt, jrt, nxt) + \
                 (1 - ALPHA) * base_machine_priority(job_idx, machine_id, processing_time, mrt, jrt, nxt)
        return (hybrid, mrt[machine_id], processing_time, machine_id)

    # ---------- constructive scheduler ----------
    schedule = [[None] * n_ops[j] for j in range(n_jobs)]
    machine_ready_time = [0] * n_machines
    job_ready_time = [0] * n_jobs
    next_op = [0] * n_jobs
    remaining = total_ops

    while remaining > 0:
        ready_jobs = [j for j in range(n_jobs) if next_op[j] < n_ops[j]]
        # choose job
        job_idx = min(ready_jobs, key=lambda j: job_priority(j, machine_ready_time, job_ready_time, next_op))
        op_idx = next_op[job_idx]
        candidates = durations[job_idx][op_idx]
        # choose machine
        machine_id, processing_time = min(
            candidates,
            key=lambda pair: machine_priority(job_idx, pair[0], pair[1], machine_ready_time, job_ready_time, next_op)
        )
        start_time = max(machine_ready_time[machine_id], job_ready_time[job_idx])
        end_time = start_time + processing_time
        schedule[job_idx][op_idx] = (machine_id, start_time, processing_time)
        machine_ready_time[machine_id] = end_time
        job_ready_time[job_idx] = end_time
        next_op[job_idx] += 1
        remaining -= 1

    return schedule
