def fjsp_solver(n_jobs, n_machines, durations):
    # schedule[job][op] = (machine_id, start_time, processing_time)
    schedule = [[None] * len(durations[j]) for j in range(n_jobs)]
    machine_ready_time = [0] * n_machines
    job_ready_time = [0] * n_jobs
    next_op = [0] * n_jobs
    remaining = sum(len(job) for job in durations)

    # track current total load (sum of processing times) on each machine
    machine_load = [0] * n_machines

    while remaining > 0:
        # pick the ready job with smallest job_ready_time
        ready_jobs = [j for j in range(n_jobs) if next_op[j] < len(durations[j])]
        job_idx = min(ready_jobs, key=lambda j: job_ready_time[j])
        op_idx = next_op[job_idx]

        # choose machine based on earliest completion time + current machine load
        candidates = durations[job_idx][op_idx]
        best_machine = None
        best_proc_time = None
        best_key = None

        for machine_id, proc_time in candidates:
            start = max(machine_ready_time[machine_id], job_ready_time[job_idx])
            completion = start + proc_time
            # key = earliest completion time + current machine load
            key = (completion + machine_load[machine_id], proc_time, machine_id)
            if best_key is None or key < best_key:
                best_key = key
                best_machine = machine_id
                best_proc_time = proc_time

        machine_id, processing_time = best_machine, best_proc_time
        start_time = max(machine_ready_time[machine_id], job_ready_time[job_idx])
        schedule[job_idx][op_idx] = (machine_id, start_time, processing_time)
        end_time = start_time + processing_time

        # update times and load
        machine_ready_time[machine_id] = end_time
        job_ready_time[job_idx] = end_time
        machine_load[machine_id] += processing_time
        next_op[job_idx] += 1
        remaining -= 1

    return schedule
