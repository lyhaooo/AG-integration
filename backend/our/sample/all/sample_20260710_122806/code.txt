def fjsp_solver(n_jobs, n_machines, durations):
    # Fixed-length schedule structure
    schedule = [[None] * len(durations[j]) for j in range(n_jobs)]
    
    # State variables
    machine_ready_time = [0] * n_machines
    machine_load = [0] * n_machines          # current total processing time assigned to each machine
    job_ready_time = [0] * n_jobs
    next_op = [0] * n_jobs
    remaining = sum(len(job) for job in durations)
    
    while remaining > 0:
        # Select the job with the earliest ready time among those with unscheduled operations
        ready_jobs = [j for j in range(n_jobs) if next_op[j] < len(durations[j])]
        job_idx = min(ready_jobs, key=lambda j: job_ready_time[j])
        op_idx = next_op[job_idx]
        candidates = durations[job_idx][op_idx]
        
        # Variant 4: minimize (earliest completion time + current machine load)
        # Key: (max(machine_ready_time[m], job_ready_time[job_idx]) + processing_time + machine_load[m],
        #        processing_time, m)
        chosen_pair = min(
            candidates,
            key=lambda pair: (
                max(machine_ready_time[pair[0]], job_ready_time[job_idx]) + pair[1] + machine_load[pair[0]],
                pair[1],
                pair[0]
            )
        )
        machine_id, processing_time = chosen_pair
        
        # Compute start time and schedule
        start_time = max(machine_ready_time[machine_id], job_ready_time[job_idx])
        schedule[job_idx][op_idx] = (machine_id, start_time, processing_time)
        end_time = start_time + processing_time
        
        # Update state
        machine_ready_time[machine_id] = end_time
        machine_load[machine_id] += processing_time
        job_ready_time[job_idx] = end_time
        next_op[job_idx] += 1
        remaining -= 1
    
    return schedule
