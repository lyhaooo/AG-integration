def fjsp_solver(n_jobs, n_machines, durations):
    # Precompute minimum processing time for each operation
    min_time = []
    total_remaining_work = [0] * n_jobs
    for j in range(n_jobs):
        job_ops = durations[j]
        mins = []
        for candidates in job_ops:
            mt = min(p for _, p in candidates)
            mins.append(mt)
        min_time.append(mins)
        total_remaining_work[j] = sum(mins)

    # Initialize schedule structure
    schedule = [[None] * len(durations[j]) for j in range(n_jobs)]
    machine_ready = [0] * n_machines
    job_ready = [0] * n_jobs
    next_op = [0] * n_jobs
    total_ops = sum(len(job) for job in durations)
    remaining = total_ops

    while remaining > 0:
        # Collect ready jobs (those with unscheduled operations)
        ready = [j for j in range(n_jobs) if next_op[j] < len(durations[j])]

        # Select job by heuristic: fewer eligible machines first, then larger remaining work
        def job_key(j):
            op = next_op[j]
            # number of eligible machines for current operation
            num_eligible = len(durations[j][op])
            # remaining work (sum of min times)
            rem_work = total_remaining_work[j]
            # primary key: num_eligible (ascending), secondary: -rem_work (descending)
            return (num_eligible, -rem_work, j)

        job_idx = min(ready, key=job_key)
        op_idx = next_op[job_idx]
        candidates = durations[job_idx][op_idx]

        # Choose machine and start time: minimize earliest completion time
        machine_id, processing_time = min(
            candidates,
            key=lambda pair: (
                max(machine_ready[pair[0]], job_ready[job_idx]) + pair[1],
                pair[1],
                pair[0],
            ),
        )

        start_time = max(machine_ready[machine_id], job_ready[job_idx])
        schedule[job_idx][op_idx] = (machine_id, start_time, processing_time)
        end_time = start_time + processing_time

        # Update state
        machine_ready[machine_id] = end_time
        job_ready[job_idx] = end_time
        next_op[job_idx] += 1
        total_remaining_work[job_idx] -= min_time[job_idx][op_idx]
        remaining -= 1

    return schedule
