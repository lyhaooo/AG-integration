def fjsp_solver(n_jobs, n_machines, durations):
    schedule = [[None] * len(durations[j]) for j in range(n_jobs)]
    machine_ready_time = [0] * n_machines
    job_ready_time = [0] * n_jobs
    next_op = [0] * n_jobs
    remaining = sum(len(job) for job in durations)

    while remaining > 0:
        ready_jobs = [j for j in range(n_jobs) if next_op[j] < len(durations[j])]
        job_idx = min(ready_jobs, key=lambda j: job_ready_time[j])
        op_idx = next_op[job_idx]
        candidates = durations[job_idx][op_idx]

        # Add one-step lookahead to consider the next operation's completion
        next_op_candidates = durations[job_idx][op_idx + 1] if op_idx + 1 < len(durations[job_idx]) else []
        best_next_completion = float('inf')
        
        if next_op_candidates:
            for next_machine_id, next_processing_time in next_op_candidates:
                next_completion_time = max(machine_ready_time[next_machine_id], job_ready_time[job_idx]) + next_processing_time
                best_next_completion = min(best_next_completion, next_completion_time)

        machine_id, processing_time = min(
            candidates,
            key=lambda pair: (max(machine_ready_time[pair[0]], job_ready_time[job_idx]) + pair[1], 
                              best_next_completion, 
                              pair[1], 
                              pair[0]),
        )

        start_time = max(machine_ready_time[machine_id], job_ready_time[job_idx])
        schedule[job_idx][op_idx] = (machine_id, start_time, processing_time)
        end_time = start_time + processing_time
        machine_ready_time[machine_id] = end_time
        job_ready_time[job_idx] = end_time
        next_op[job_idx] += 1
        remaining -= 1

    return schedule
