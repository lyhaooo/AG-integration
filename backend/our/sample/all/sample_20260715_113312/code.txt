def fjsp_solver(n_jobs, n_machines, durations):
    # ---------- helper data ----------
    n_ops = [len(job) for job in durations]
    total_ops = sum(n_ops)
    # Precompute remaining minimum work for each job from each operation
    remaining_min_work = [[0] * (n_ops[j] + 1) for j in range(n_jobs)]
    for j in range(n_jobs):
        for op in range(n_ops[j] - 1, -1, -1):
            best = min(durations[j][op], key=lambda x: x[1])[1]
            remaining_min_work[j][op] = best + remaining_min_work[j][op + 1]

    ALPHA = 0.85
    # mutated machine scoring weights – increase penalty for machine load to reduce bottleneck congestion
    W_LOAD = 0.7      # increased from 0.4
    W_REM = 0.1       # decreased from 0.3
    W_PROC = 0.2      # unchanged

    # ---------- job priority functions (identical to elite parent) ----------
    def base_job_priority(job_idx, mrt, jrt, nxt):
        op_idx = nxt[job_idx]
        candidates = durations[job_idx][op_idx]
        min_start = min(
            max(mrt[m], jrt[job_idx])
            for m, _ in candidates
        )
        return min_start

    def elite_job_priority(job_idx, mrt, jrt, nxt):
        op_idx = nxt[job_idx]
        candidates = durations[job_idx][op_idx]
        min_start = min(
            max(mrt[m], jrt[job_idx])
            for m, _ in candidates
        )
        optimistic_completion = min_start + remaining_min_work[job_idx][op_idx]
        avg_machine_load = sum(mrt[m] for m, _ in candidates) / len(candidates) if candidates else 0
        return min_start - 0.3 * remaining_min_work[job_idx][op_idx] + 0.2 * avg_machine_load + 0.25 * optimistic_completion

    def job_priority(job_idx, mrt, jrt, nxt):
        hybrid = ALPHA * elite_job_priority(job_idx, mrt, jrt, nxt) + (1 - ALPHA) * base_job_priority(job_idx, mrt, jrt, nxt)
        return (hybrid, job_idx)

    # --- mutated machine priority with adjusted weights and tie-breaker ---
    def new_machine_priority(job_idx, machine_id, processing_time, mrt, jrt, nxt):
        op_idx = nxt[job_idx]
        start_time = max(mrt[machine_id], jrt[job_idx])
        completion = start_time + processing_time
        machine_load = mrt[machine_id] + processing_time
        job_rem = remaining_min_work[job_idx][op_idx]
        # increased weight on machine load to discourage congested machines
        score = completion + W_LOAD * machine_load + W_REM * job_rem - W_PROC * processing_time
        return score

    def base_machine_priority(job_idx, machine_id, processing_time, mrt, jrt, nxt):
        start_time = max(mrt[machine_id], jrt[job_idx])
        return start_time + processing_time

    def machine_priority(job_idx, machine_id, processing_time, mrt, jrt, nxt):
        hybrid = ALPHA * new_machine_priority(job_idx, machine_id, processing_time, mrt, jrt, nxt) + \
                 (1 - ALPHA) * base_machine_priority(job_idx, machine_id, processing_time, mrt, jrt, nxt)
        # tie-breaker: prefer lower machine load, then shorter processing time, then machine id
        return (hybrid, mrt[machine_id] + processing_time, processing_time, machine_id)

    # ---------- constructive scheduler ----------
    def build_schedule(forced_machine=None):
        # reinitialize state
        sched = [[None] * n_ops[j] for j in range(n_jobs)]
        mrt = [0] * n_machines
        jrt = [0] * n_jobs
        nxt = [0] * n_jobs
        rem = total_ops

        # local wrappers that use the current state
        def local_job_priority(job_idx):
            return job_priority(job_idx, mrt, jrt, nxt)

        def local_machine_priority(job_idx, machine_id, processing_time):
            return machine_priority(job_idx, machine_id, processing_time, mrt, jrt, nxt)

        while rem > 0:
            ready_jobs = [j for j in range(n_jobs) if nxt[j] < n_ops[j]]
            # choose job
            job_idx = min(ready_jobs, key=local_job_priority)
            op_idx = nxt[job_idx]
            candidates = durations[job_idx][op_idx]

            # check forced machine (if any)
            if forced_machine is not None and (job_idx, op_idx) in forced_machine:
                forced_m_id = forced_machine[(job_idx, op_idx)]
                selected = None
                for m_id, p_time in candidates:
                    if m_id == forced_m_id:
                        selected = (m_id, p_time)
                        break
                if selected is None:
                    return None  # infeasible
                machine_id, processing_time = selected
            else:
                # normal selection using mutated machine priority
                machine_id, processing_time = min(candidates, key=lambda pair: local_machine_priority(job_idx, pair[0], pair[1]))

            start_time = max(mrt[machine_id], jrt[job_idx])
            sched[job_idx][op_idx] = (machine_id, start_time, processing_time)
            end_time = start_time + processing_time
            mrt[machine_id] = end_time
            jrt[job_idx] = end_time
            nxt[job_idx] += 1
            rem -= 1

        return sched

    # run the builder with no forced machines
    schedule = build_schedule()
    return schedule
