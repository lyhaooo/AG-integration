import random

def fjsp_solver(n_jobs, n_machines, durations):
    # Precompute remaining minimum processing time for each job (lookahead)
    remaining_min_work = []
    for j in range(n_jobs):
        job_ops = durations[j]
        rem = [0] * (len(job_ops) + 1)
        for i in range(len(job_ops) - 1, -1, -1):
            best = min(t for _, t in job_ops[i])
            rem[i] = rem[i + 1] + best
        remaining_min_work.append(rem)

    # Schedule state
    schedule = [[None] * len(durations[j]) for j in range(n_jobs)]
    machine_ready_time = [0] * n_machines
    job_ready_time = [0] * n_jobs
    next_op = [0] * n_jobs
    remaining = sum(len(job) for job in durations)

    # Random seed for reproducibility
    rng = random.Random(42)

    while remaining > 0:
        # Ready jobs: those with unfinished operations
        ready_jobs = [j for j in range(n_jobs) if next_op[j] < len(durations[j])]

        # Job priority: earliest possible completion + lookahead
        def job_priority(j):
            op = next_op[j]
            best_cur = min(t for _, t in durations[j][op])
            return (
                job_ready_time[j] + 0.4 * best_cur - 0.2 * remaining_min_work[j][op],
                j
            )

        job_idx = min(ready_jobs, key=job_priority)
        op_idx = next_op[job_idx]
        candidates = durations[job_idx][op_idx]

        # Machine selection with deterministic score
        # Score = completion + 0.2 * lookahead + 0.1 * (current_best - proc_time)
        def machine_score(pair):
            m, t = pair
            start = max(machine_ready_time[m], job_ready_time[job_idx])
            comp = start + t
            # lookahead to next operation (if any)
            lookahead = 0
            if op_idx + 1 < len(durations[job_idx]):
                next_candidates = durations[job_idx][op_idx + 1]
                lookahead = min(
                    max(machine_ready_time[nm], comp) + nt
                    for nm, nt in next_candidates
                )
            cur_best = min(pt for _, pt in candidates)
            return comp + 0.2 * lookahead + 0.1 * (cur_best - t), t, m

        # Sort candidates by score (lowest is best)
        scored = sorted(candidates, key=machine_score)

        # Deterministic best
        best_pair = scored[0]

        # With small probability (≤5%), randomly pick among top 2 (if available)
        if len(scored) >= 2 and rng.random() < 0.05:
            # pick second best (or random among top 2)
            best_pair = scored[1] if rng.random() < 0.5 else scored[0]

        machine_id, processing_time = best_pair
        start_time = max(machine_ready_time[machine_id], job_ready_time[job_idx])
        schedule[job_idx][op_idx] = (machine_id, start_time, processing_time)
        end_time = start_time + processing_time
        machine_ready_time[machine_id] = end_time
        job_ready_time[job_idx] = end_time
        next_op[job_idx] += 1
        remaining -= 1

    return schedule
