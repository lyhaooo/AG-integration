def fjsp_solver(n_jobs, n_machines, durations):
    # Precompute static machine demand: sum of processing times for each machine across all operations
    machine_total_demand = [0] * n_machines
    for job in durations:
        for op in job:
            for m, p in op:
                machine_total_demand[m] += p

    # Number of operations per job
    ops_per_job = [len(job) for job in durations]
    schedule = [[None] * ops_per_job[j] for j in range(n_jobs)]

    # State arrays
    machine_ready_time = [0] * n_machines
    machine_current_load = [0] * n_machines  # total processing time already scheduled on each machine
    job_ready_time = [0] * n_jobs
    next_op = [0] * n_jobs
    remaining = sum(ops_per_job)

    while remaining > 0:
        # Ready jobs: those with remaining operations
        ready_jobs = [j for j in range(n_jobs) if next_op[j] < ops_per_job[j]]

        # Compute bottleneck score for each ready job's next operation
        # Score = sum of machine_total_demand over all candidate machines of that operation
        scores = []
        for j in ready_jobs:
            op_idx = next_op[j]
            candidates = durations[j][op_idx]
            score = sum(machine_total_demand[m] for m, _ in candidates)
            scores.append(score)

        # Select job with highest bottleneck score (tie: earliest job_ready_time)
        # Combine scores and job_ready_time for stable selection
        min_ready = min(job_ready_time[j] for j in ready_jobs)
        # Use tuple: (-score, job_ready_time[j]) so max score, then min ready time
        best_idx = max(range(len(ready_jobs)),
                       key=lambda i: (-scores[i], job_ready_time[ready_jobs[i]]))
        job_idx = ready_jobs[best_idx]

        op_idx = next_op[job_idx]
        candidates = durations[job_idx][op_idx]

        # Choose machine for this operation using bottleneck-aware rule:
        # primary: completion time (earliest finish), secondary: current machine load (lower is better)
        def machine_key(machine_pair):
            m, p = machine_pair
            start = max(machine_ready_time[m], job_ready_time[job_idx])
            completion = start + p
            return (completion, machine_current_load[m])

        machine_id, processing_time = min(candidates, key=machine_key)

        # Schedule the operation
        start_time = max(machine_ready_time[machine_id], job_ready_time[job_idx])
        schedule[job_idx][op_idx] = (machine_id, start_time, processing_time)
        end_time = start_time + processing_time

        # Update state
        machine_ready_time[machine_id] = end_time
        machine_current_load[machine_id] += processing_time
        job_ready_time[job_idx] = end_time
        next_op[job_idx] += 1
        remaining -= 1

    return schedule
