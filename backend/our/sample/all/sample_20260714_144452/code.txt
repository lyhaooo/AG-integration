def fjsp_solver(n_jobs, n_machines, durations):
    # Initialize schedule as list of lists of None
    schedule = [[None] * len(durations[j]) for j in range(n_jobs)]
    # Track machine availability (next free time) and cumulative load
    machine_ready_time = [0] * n_machines
    machine_load = [0] * n_machines
    # Track job availability (next operation start time)
    job_ready_time = [0] * n_jobs
    # Next operation index for each job
    next_op = [0] * n_jobs
    # Total remaining operations to schedule
    remaining = sum(len(job) for job in durations)

    while remaining > 0:
        # Select job with earliest ready time (tie-break by index for determinism)
        ready_jobs = [j for j in range(n_jobs) if next_op[j] < len(durations[j])]
        job_idx = min(ready_jobs, key=lambda j: (job_ready_time[j], j))
        op_idx = next_op[job_idx]

        # Evaluate each candidate (machine, processing_time) for this operation
        candidates = durations[job_idx][op_idx]

        def candidate_key(pair):
            machine_id, processing_time = pair
            # Earliest completion time for this candidate
            start = max(machine_ready_time[machine_id], job_ready_time[job_idx])
            completion_time = start + processing_time
            # Current load of the machine (sum of assigned durations so far)
            current_load = machine_load[machine_id]
            # Metric: completion_time + current_load (variant 1)
            metric = completion_time + current_load
            # Tie-breaking: shorter processing_time, then lower machine_id
            return (metric, processing_time, machine_id)

        machine_id, processing_time = min(candidates, key=candidate_key)

        # Compute start time and schedule the operation
        start_time = max(machine_ready_time[machine_id], job_ready_time[job_idx])
        schedule[job_idx][op_idx] = (machine_id, start_time, processing_time)

        # Update times and loads
        end_time = start_time + processing_time
        machine_ready_time[machine_id] = end_time
        machine_load[machine_id] += processing_time
        job_ready_time[job_idx] = end_time

        # Move to next operation
        next_op[job_idx] += 1
        remaining -= 1

    return schedule
