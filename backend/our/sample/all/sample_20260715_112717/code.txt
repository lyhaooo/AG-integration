def fjsp_solver(n_jobs, n_machines, durations):
    # Precompute minimum processing time for each operation (for remaining work)
    min_times = []
    for job_ops in durations:
        job_min = [min(candidates, key=lambda x: x[1])[1] for candidates in job_ops]
        min_times.append(job_min)
    
    # Precompute remaining work from each operation index onward
    remaining_work = []
    for job_min in min_times:
        n_ops = len(job_min)
        cum = [0] * (n_ops + 1)
        for i in range(n_ops - 1, -1, -1):
            cum[i] = cum[i + 1] + job_min[i]
        remaining_work.append(cum)  # cum[op] = total remaining based on min times

    schedule = [[None] * len(durations[j]) for j in range(n_jobs)]
    machine_ready_time = [0] * n_machines
    job_ready_time = [0] * n_jobs
    next_op = [0] * n_jobs
    remaining = sum(len(job) for job in durations)

    while remaining > 0:
        # Select job using variant 3: fewer eligible machines, then larger remaining work
        ready_jobs = [j for j in range(n_jobs) if next_op[j] < len(durations[j])]
        job_idx = min(
            ready_jobs,
            key=lambda j: (
                len(durations[j][next_op[j]]),          # number of eligible machines
                -remaining_work[j][next_op[j]]          # larger remaining work first
            )
        )

        op_idx = next_op[job_idx]
        candidates = durations[job_idx][op_idx]

        # Choose machine: earliest completion time, then shortest processing time, then machine id
        machine_id, processing_time = min(
            candidates,
            key=lambda pair: (
                max(machine_ready_time[pair[0]], job_ready_time[job_idx]) + pair[1],
                pair[1],
                pair[0]
            )
        )

        start_time = max(machine_ready_time[machine_id], job_ready_time[job_idx])
        schedule[job_idx][op_idx] = (machine_id, start_time, processing_time)
        end_time = start_time + processing_time
        machine_ready_time[machine_id] = end_time
        job_ready_time[job_idx] = end_time
        next_op[job_idx] += 1
        remaining -= 1

    return schedule
