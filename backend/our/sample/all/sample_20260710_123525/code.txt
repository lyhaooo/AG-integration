def fjsp_solver(n_jobs, n_machines, durations):
    """
    Flexible Job Shop Scheduling Problem solver.
    Objective: minimize makespan.
    Selection variant: penalize machines that are bottlenecks for future operations.
    """
    # Compute total operations per job
    ops_per_job = [len(durations[j]) for j in range(n_jobs)]
    total_ops = sum(ops_per_job)

    # Initialize schedule as list of lists of None
    schedule = [[None] * ops_per_job[j] for j in range(n_jobs)]

    # Ready times
    job_ready_time = [0] * n_jobs
    machine_ready_time = [0] * n_machines
    next_op = [0] * n_jobs

    remaining = total_ops
    # Penalty factor for bottleneck load
    BOTTLENECK_PENALTY = 0.01

    while remaining > 0:
        # Compute future load for each machine:
        # total processing time of unscheduled operations that can use each machine
        future_load = [0] * n_machines
        for j in range(n_jobs):
            if next_op[j] < ops_per_job[j]:
                # All remaining operations of job j
                for op_idx in range(next_op[j], ops_per_job[j]):
                    for m, p in durations[j][op_idx]:
                        future_load[m] += p

        # Determine ready jobs (jobs with unscheduled operations)
        ready_jobs = [j for j in range(n_jobs) if next_op[j] < ops_per_job[j]]

        # Choose job with earliest ready time (tie-breaking by job index for determinism)
        job_idx = min(ready_jobs, key=lambda j: (job_ready_time[j], j))
        op_idx = next_op[job_idx]
        candidates = durations[job_idx][op_idx]

        # Select machine: minimize (start_time + proc_time + penalty * future_load[machine])
        # with secondary keys processing_time and machine_id.
        best = min(
            candidates,
            key=lambda pair: (
                max(machine_ready_time[pair[0]], job_ready_time[job_idx])
                + pair[1]
                + BOTTLENECK_PENALTY * future_load[pair[0]],
                pair[1],
                pair[0]
            )
        )
        machine_id, processing_time = best

        # Compute start time and schedule the operation
        start_time = max(machine_ready_time[machine_id], job_ready_time[job_idx])
        schedule[job_idx][op_idx] = (machine_id, start_time, processing_time)
        end_time = start_time + processing_time

        # Update ready times
        machine_ready_time[machine_id] = end_time
        job_ready_time[job_idx] = end_time

        next_op[job_idx] += 1
        remaining -= 1

    return schedule
