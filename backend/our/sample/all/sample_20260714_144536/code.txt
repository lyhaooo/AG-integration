def fjsp_solver(n_jobs, n_machines, durations):
    """
    Flexible Job Shop Scheduling Solver.
    Variant 4: machine selection minimizes earliest completion time plus current machine load.
    """
    # Initialize schedule as 2D list of None
    schedule = [[None] * len(durations[j]) for j in range(n_jobs)]

    # Track ready times
    machine_ready = [0] * n_machines
    job_ready = [0] * n_jobs

    # Track next operation index for each job
    next_op = [0] * n_jobs

    # Track cumulative load on each machine (total processing time assigned)
    machine_load = [0] * n_machines

    # Total number of operations to schedule
    remaining = sum(len(job_ops) for job_ops in durations)

    while remaining > 0:
        # Jobs that still have unscheduled operations
        ready_jobs = [j for j in range(n_jobs) if next_op[j] < len(durations[j])]

        # Select the job with the earliest ready time (can be changed, but focus is on machine selection)
        job_idx = min(ready_jobs, key=lambda j: job_ready[j])
        op_idx = next_op[job_idx]

        # Candidates for this operation: list of (machine, processing_time)
        candidates = durations[job_idx][op_idx]

        # Machine selection: minimize (earliest completion time + machine_load)
        # Also use processing time and machine index as tiebreakers
        machine_id, processing_time = min(
            candidates,
            key=lambda pair: (
                max(machine_ready[pair[0]], job_ready[job_idx]) + pair[1] + machine_load[pair[0]],
                pair[1],
                pair[0]
            )
        )

        # Compute start time
        start = max(machine_ready[machine_id], job_ready[job_idx])
        end = start + processing_time

        # Store the decision
        schedule[job_idx][op_idx] = (machine_id, start, processing_time)

        # Update ready times
        machine_ready[machine_id] = end
        job_ready[job_idx] = end

        # Update machine load
        machine_load[machine_id] += processing_time

        # Move to next operation of this job
        next_op[job_idx] += 1
        remaining -= 1

    return schedule
