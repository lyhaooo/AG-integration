def fjsp_solver(n_jobs, n_machines, durations):
    # ---------- precompute remaining min work ----------
    n_ops = [len(job) for job in durations]
    total_ops = sum(n_ops)
    remaining_min_work = [[0] * (n_ops[j] + 1) for j in range(n_jobs)]
    for j in range(n_jobs):
        for op in range(n_ops[j] - 1, -1, -1):
            best = min(durations[j][op], key=lambda x: x[1])[1]
            remaining_min_work[j][op] = best + remaining_min_work[j][op + 1]

    ALPHA = 0.85  # hybridization coefficient

    # ---------- state initialization ----------
    schedule = [[None] * len(durations[j]) for j in range(n_jobs)]
    machine_ready_time = [0] * n_machines
    machine_total_load = [0] * n_machines   # NEW: total processing time assigned to each machine
    job_ready_time = [0] * n_jobs
    next_op = [0] * n_jobs
    remaining = total_ops

    # ---------- priority functions (capture current state) ----------
    def base_job_priority(job_idx):
        op_idx = next_op[job_idx]
        candidates = durations[job_idx][op_idx]
        min_start = min(max(machine_ready_time[m], job_ready_time[job_idx]) for m, _ in candidates)
        return min_start

    def elite_job_priority(job_idx):
        op_idx = next_op[job_idx]
        candidates = durations[job_idx][op_idx]
        min_start = min(max(machine_ready_time[m], job_ready_time[job_idx]) for m, _ in candidates)
        optimistic_completion = min_start + remaining_min_work[job_idx][op_idx]
        avg_machine_load = sum(machine_ready_time[m] for m, _ in candidates) / len(candidates) if candidates else 0
        return min_start - 0.3 * remaining_min_work[job_idx][op_idx] + 0.2 * avg_machine_load + 0.25 * optimistic_completion

    def job_priority(job_idx):
        hybrid = ALPHA * elite_job_priority(job_idx) + (1 - ALPHA) * base_job_priority(job_idx)
        return (hybrid, job_idx)

    # NEW machine scoring: stronger weight on machine congestion (total load)
    def new_machine_priority(job_idx, machine_id, processing_time):
        op_idx = next_op[job_idx]
        start_time = max(machine_ready_time[machine_id], job_ready_time[job_idx])
        completion = start_time + processing_time
        # Congestion measure: total work assigned to this machine
        congestion = machine_total_load[machine_id] + processing_time
        job_rem = remaining_min_work[job_idx][op_idx]
        # Score: penalize congestion heavily, reward jobs with high remaining work
        return completion + 0.6 * congestion + 0.3 * job_rem - 0.2 * processing_time

    def base_machine_priority(job_idx, machine_id, processing_time):
        start_time = max(machine_ready_time[machine_id], job_ready_time[job_idx])
        return start_time + processing_time

    def machine_priority(job_idx, machine_id, processing_time):
        hybrid = ALPHA * new_machine_priority(job_idx, machine_id, processing_time) + \
                 (1 - ALPHA) * base_machine_priority(job_idx, machine_id, processing_time)
        # Tie-breaker: prefer machine with lower total load, then lower ready time, then lower processing time
        return (hybrid, machine_total_load[machine_id], machine_ready_time[machine_id], processing_time, machine_id)

    # ---------- main scheduling loop ----------
    while remaining > 0:
        ready_jobs = [j for j in range(n_jobs) if next_op[j] < n_ops[j]]
        # select job by hybrid priority
        job_idx = min(ready_jobs, key=job_priority)
        op_idx = next_op[job_idx]
        candidates = durations[job_idx][op_idx]

        # select machine by hybrid priority
        machine_id, processing_time = min(candidates, key=lambda pair: machine_priority(job_idx, pair[0], pair[1]))

        start_time = max(machine_ready_time[machine_id], job_ready_time[job_idx])
        end_time = start_time + processing_time

        # record assignment
        schedule[job_idx][op_idx] = (machine_id, start_time, processing_time)

        # update state
        machine_ready_time[machine_id] = end_time
        machine_total_load[machine_id] += processing_time  # NEW: track total load
        job_ready_time[job_idx] = end_time
        next_op[job_idx] += 1
        remaining -= 1

    return schedule
