def fjsp_solver(n_jobs, n_machines, durations):
    # Precompute remaining minimum processing time for each job operation
    n_ops = [len(job) for job in durations]
    remaining_min_work = [[0] * (n_ops[j] + 1) for j in range(n_jobs)]
    for j in range(n_jobs):
        for op in range(n_ops[j] - 1, -1, -1):
            best = min(durations[j][op], key=lambda x: x[1])[1]
            remaining_min_work[j][op] = best + remaining_min_work[j][op + 1]

    ALPHA = 0.85  # keep same hybrid blend

    # --- Job priority (mutated weights to target worst gaps) ---
    def job_priority(job_idx, mrt, jrt, nxt):
        op_idx = nxt[job_idx]
        candidates = durations[job_idx][op_idx]
        min_start = min(max(mrt[m], jrt[job_idx]) for m, _ in candidates)
        # Optimistic completion = min_start + remaining work of job from this op
        opt_completion = min_start + remaining_min_work[job_idx][op_idx]
        avg_machine_load = sum(mrt[m] for m, _ in candidates) / len(candidates) if candidates else 0
        # Mutated weights: increase remaining work impact, reduce machine load impact
        el = min_start - 0.5 * remaining_min_work[job_idx][op_idx] + 0.0 * avg_machine_load + 0.3 * opt_completion
        base = min_start
        return ALPHA * el + (1 - ALPHA) * base

    # --- Machine priority (stable, only minor change) ---
    def machine_priority(job_idx, machine_id, processing_time, mrt, jrt, nxt):
        op_idx = nxt[job_idx]
        start_time = max(mrt[machine_id], jrt[job_idx])
        completion = start_time + processing_time
        machine_load = mrt[machine_id] + processing_time
        job_rem = remaining_min_work[job_idx][op_idx]
        new = completion + 0.4 * machine_load + 0.3 * job_rem - 0.2 * processing_time
        base = start_time + processing_time
        return ALPHA * new + (1 - ALPHA) * base

    # --- Constructive scheduler ---
    schedule = [[None] * n_ops[j] for j in range(n_jobs)]
    machine_ready_time = [0] * n_machines
    job_ready_time = [0] * n_jobs
    next_op = [0] * n_jobs
    remaining = sum(n_ops)

    while remaining > 0:
        # Select job with smallest priority value
        ready_jobs = [j for j in range(n_jobs) if next_op[j] < n_ops[j]]
        job_idx = min(ready_jobs, key=lambda j: job_priority(j, machine_ready_time, job_ready_time, next_op))

        op_idx = next_op[job_idx]
        candidates = durations[job_idx][op_idx]

        # Select best machine for this job
        best_machine, best_time = min(
            candidates,
            key=lambda pair: machine_priority(job_idx, pair[0], pair[1],
                                               machine_ready_time, job_ready_time, next_op)
        )

        start_time = max(machine_ready_time[best_machine], job_ready_time[job_idx])
        schedule[job_idx][op_idx] = (best_machine, start_time, best_time)
        end_time = start_time + best_time

        machine_ready_time[best_machine] = end_time
        job_ready_time[job_idx] = end_time
        next_op[job_idx] += 1
        remaining -= 1

    return schedule
