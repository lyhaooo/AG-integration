def fjsp_solver(n_jobs, n_machines, durations):
    # Initialize schedule
    schedule = [[None] * len(durations[j]) for j in range(n_jobs)]
    machine_ready_time = [0] * n_machines
    job_ready_time = [0] * n_jobs
    next_op = [0] * n_jobs
    total_ops = sum(len(job) for job in durations)
    
    # Precompute machine remaining load: total processing time of all
    # unscheduled operations that could be processed on each machine.
    machine_remaining_load = [0.0] * n_machines
    for job in durations:
        for op in job:
            for m, p in op:
                machine_remaining_load[m] += p
    
    remaining = total_ops
    while remaining > 0:
        # Determine ready jobs and their next operation's flexibility (number of candidate machines)
        ready_jobs = []
        flexibilities = []
        for j in range(n_jobs):
            if next_op[j] < len(durations[j]):
                ready_jobs.append(j)
                flexibilities.append(len(durations[j][next_op[j]]))
        
        # Select job with smallest flexibility (most constrained)
        # Break ties using earliest job_ready_time
        min_flex = min(flexibilities)
        candidates_jobs = [j for j, f in zip(ready_jobs, flexibilities) if f == min_flex]
        job_idx = min(candidates_jobs, key=lambda j: job_ready_time[j])
        
        op_idx = next_op[job_idx]
        candidates = durations[job_idx][op_idx]
        
        # Current machine load factor (normalized)
        max_load = max(machine_remaining_load) if max(machine_remaining_load) > 0 else 1.0
        
        best_score = float('inf')
        best_machine = None
        best_start = None
        best_pt = None
        
        for machine_id, processing_time in candidates:
            start_time = max(machine_ready_time[machine_id], job_ready_time[job_idx])
            finish_time = start_time + processing_time
            # Bottleneck penalty: add a fraction of processing_time weighted by relative machine load
            load_factor = machine_remaining_load[machine_id] / max_load
            score = finish_time + 0.2 * processing_time * load_factor
            if score < best_score:
                best_score = score
                best_machine = machine_id
                best_start = start_time
                best_pt = processing_time
        
        # Schedule the operation
        schedule[job_idx][op_idx] = (best_machine, best_start, best_pt)
        end_time = best_start + best_pt
        machine_ready_time[best_machine] = end_time
        job_ready_time[job_idx] = end_time
        next_op[job_idx] += 1
        remaining -= 1
        
        # Update machine remaining load: remove this operation from all machines that could have processed it
        for m, p in candidates:
            machine_remaining_load[m] -= p
    
    return schedule
