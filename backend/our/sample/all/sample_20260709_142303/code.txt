def fjsp_solver(n_jobs, n_machines, durations):
    # Initialize scheduling structures
    schedule = [[None] * len(durations[j]) for j in range(n_jobs)]
    machine_ready = [0] * n_machines
    job_ready = [0] * n_jobs
    next_op = [0] * n_jobs
    remaining = sum(len(job) for job in durations)

    # Determine a scaling factor for bottleneck penalty
    max_dur = 0
    for job in durations:
        for op in job:
            for _, dur in op:
                if dur > max_dur:
                    max_dur = dur
    alpha = max_dur * 0.5   # penalty weight

    while remaining > 0:
        # ----- Compute bottleneck counts (future operations per machine) -----
        future_cnt = [0] * n_machines
        for j in range(n_jobs):
            if next_op[j] < len(durations[j]):
                for op_idx in range(next_op[j], len(durations[j])):
                    for m, _ in durations[j][op_idx]:
                        future_cnt[m] += 1

        # ----- Select next job (earliest ready time) -----
        ready_jobs = [j for j in range(n_jobs) if next_op[j] < len(durations[j])]
        job_idx = min(ready_jobs, key=lambda j: job_ready[j])
        op_idx = next_op[job_idx]
        candidates = durations[job_idx][op_idx]

        # ----- Choose machine with bottleneck penalty -----
        machine_id, processing_time = min(
            candidates,
            key=lambda pair: (
                max(machine_ready[pair[0]], job_ready[job_idx])
                + pair[1]
                + alpha * future_cnt[pair[0]],   # penalise bottleneck machines
                pair[1],
                pair[0]
            )
        )

        # ----- Schedule the operation -----
        start = max(machine_ready[machine_id], job_ready[job_idx])
        schedule[job_idx][op_idx] = (machine_id, start, processing_time)
        end = start + processing_time
        machine_ready[machine_id] = end
        job_ready[job_idx] = end
        next_op[job_idx] += 1
        remaining -= 1

    return schedule
