def fjsp_solver(n_jobs, n_machines, durations):
    # Precompute minimum processing time for each operation
    min_pt = []
    for job in durations:
        job_min = [min(candidates, key=lambda p: p[1])[1] for candidates in job]
        min_pt.append(job_min)

    # Initialize data structures
    schedule = [[None] * len(durations[j]) for j in range(n_jobs)]
    machine_ready_time = [0] * n_machines
    job_ready_time = [0] * n_jobs
    next_op = [0] * n_jobs
    remaining = sum(len(job) for job in durations)

    while remaining > 0:
        # Identify ready jobs (jobs with unscheduled operations)
        ready_jobs = [j for j in range(n_jobs) if next_op[j] < len(durations[j])]

        # Compute remaining total minimum processing time for each ready job
        # and select the one with the largest sum.
        # Tie-break: job with earlier ready time (smaller job_ready_time) gets priority.
        def job_key(j):
            # Sum of min processing times from current operation to end
            rem_sum = sum(min_pt[j][op] for op in range(next_op[j], len(durations[j])))
            # Primary: larger rem_sum; secondary: smaller job_ready_time
            return (-rem_sum, job_ready_time[j])

        job_idx = min(ready_jobs, key=job_key)  # min with negative yields larger sum first
        op_idx = next_op[job_idx]

        # Choose machine for this operation (same heuristic as skeleton to keep focus on job selection)
        candidates = durations[job_idx][op_idx]
        machine_id, processing_time = min(
            candidates,
            key=lambda pair: (
                max(machine_ready_time[pair[0]], job_ready_time[job_idx]) + pair[1],
                pair[1],
                pair[0]
            )
        )

        # Schedule operation
        start_time = max(machine_ready_time[machine_id], job_ready_time[job_idx])
        schedule[job_idx][op_idx] = (machine_id, start_time, processing_time)
        end_time = start_time + processing_time

        # Update resource states
        machine_ready_time[machine_id] = end_time
        job_ready_time[job_idx] = end_time
        next_op[job_idx] += 1
        remaining -= 1

    return schedule
