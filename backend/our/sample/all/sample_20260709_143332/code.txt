def fjsp_solver(n_jobs, n_machines, durations):
    schedule = [[None] * len(durations[j]) for j in range(n_jobs)]
    machine_ready_time = [0] * n_machines
    job_ready_time = [0] * n_jobs
    next_op = [0] * n_jobs

    # Precompute remaining min work per job per operation
    remaining_min_work = [[0] * (len(durations[j]) + 1) for j in range(n_jobs)]
    for job_idx in range(n_jobs):
        for op_idx in range(len(durations[job_idx]) - 1, -1, -1):
            best_time = min(p for _, p in durations[job_idx][op_idx])
            remaining_min_work[job_idx][op_idx] = remaining_min_work[job_idx][op_idx + 1] + best_time

    # ---------- Elite priority functions (original) ----------
    def elite_job_priority(job_idx):
        op_idx = next_op[job_idx]
        candidates = durations[job_idx][op_idx]
        min_start = min(
            max(machine_ready_time[m], job_ready_time[job_idx])
            for m, _ in candidates
        )
        avg_machine_load = sum(machine_ready_time[m] for m, _ in candidates) / max(len(candidates), 1)
        remaining = remaining_min_work[job_idx][op_idx]
        return min_start - 0.2 * remaining + 0.2 * avg_machine_load

    def elite_machine_priority(job_idx, machine_id, processing_time):
        op_idx = next_op[job_idx]
        start_time = max(machine_ready_time[machine_id], job_ready_time[job_idx])
        completion_time = start_time + processing_time
        lookahead_completion = 0
        if op_idx + 1 < len(durations[job_idx]):
            lookahead_completion = min(
                max(machine_ready_time[next_machine], completion_time) + next_time
                for next_machine, next_time in durations[job_idx][op_idx + 1]
            )
        avg_ready = sum(machine_ready_time) / max(n_machines, 1)
        load_bias = 0.7 * (machine_ready_time[machine_id] - avg_ready)
        min_p = min(p for _, p in durations[job_idx][op_idx])
        return (
            completion_time
            + 0.3 * lookahead_completion
            + 0.2 * (min_p - processing_time)
            + load_bias
        )

    # ---------- Historical lesson priority functions (modified coefficients) ----------
    def historical_job_priority(job_idx):
        op_idx = next_op[job_idx]
        candidates = durations[job_idx][op_idx]
        min_start = min(
            max(machine_ready_time[m], job_ready_time[job_idx])
            for m, _ in candidates
        )
        avg_machine_load = sum(machine_ready_time[m] for m, _ in candidates) / max(len(candidates), 1)
        remaining = remaining_min_work[job_idx][op_idx]
        # Slightly different weighting: give more importance to remaining work
        return min_start - 0.3 * remaining + 0.1 * avg_machine_load

    def historical_machine_priority(job_idx, machine_id, processing_time):
        op_idx = next_op[job_idx]
        start_time = max(machine_ready_time[machine_id], job_ready_time[job_idx])
        completion_time = start_time + processing_time
        lookahead_completion = 0
        if op_idx + 1 < len(durations[job_idx]):
            lookahead_completion = min(
                max(machine_ready_time[next_machine], completion_time) + next_time
                for next_machine, next_time in durations[job_idx][op_idx + 1]
            )
        avg_ready = sum(machine_ready_time) / max(n_machines, 1)
        # Historical lesson uses a stronger load bias to avoid congested machines
        load_bias = 1.0 * (machine_ready_time[machine_id] - avg_ready)
        min_p = min(p for _, p in durations[job_idx][op_idx])
        return (
            completion_time
            + 0.25 * lookahead_completion
            + 0.15 * (min_p - processing_time)
            + load_bias
        )

    # ---------- Hybrid priority combining elite and historical ----------
    def job_priority(job_idx):
        e = elite_job_priority(job_idx)
        h = historical_job_priority(job_idx)
        return (0.5 * e + 0.5 * h, job_idx)

    def machine_priority(job_idx, machine_id, processing_time):
        e = elite_machine_priority(job_idx, machine_id, processing_time)
        h = historical_machine_priority(job_idx, machine_id, processing_time)
        # Tie-break by processing_time and machine_id for stability
        return (0.5 * e + 0.5 * h, processing_time, machine_id)

    # ---------- Main dispatch loop ----------
    remaining = sum(len(job) for job in durations)
    while remaining > 0:
        ready_jobs = [j for j in range(n_jobs) if next_op[j] < len(durations[j])]
        job_idx = min(ready_jobs, key=job_priority)
        op_idx = next_op[job_idx]
        candidates = durations[job_idx][op_idx]
        machine_id, processing_time = min(
            candidates,
            key=lambda pair: machine_priority(job_idx, pair[0], pair[1]),
        )
        start_time = max(machine_ready_time[machine_id], job_ready_time[job_idx])
        end_time = start_time + processing_time
        schedule[job_idx][op_idx] = (machine_id, start_time, processing_time)
        machine_ready_time[machine_id] = end_time
        job_ready_time[job_idx] = end_time
        next_op[job_idx] += 1
        remaining -= 1

    return schedule
