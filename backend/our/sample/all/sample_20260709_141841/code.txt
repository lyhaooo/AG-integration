def fjsp_solver(n_jobs, n_machines, durations):
    # Precompute static machine load: total processing time of all operations that can use the machine
    machine_load = [0] * n_machines
    for job in durations:
        for op_candidates in job:
            for m, t in op_candidates:
                machine_load[m] += t
    # Normalize load to avoid huge penalty values (divide by total op count if needed)
    total_ops = sum(len(job) for job in durations)
    avg_load = sum(machine_load) / n_machines if n_machines else 1
    load_weight = 0.01  # small weight, relative to processing times

    schedule = [[None] * len(durations[j]) for j in range(n_jobs)]
    machine_ready_time = [0] * n_machines
    job_ready_time = [0] * n_jobs
    next_op = [0] * n_jobs
    remaining = sum(len(job) for job in durations)

    while remaining > 0:
        # Identify ready jobs
        ready_jobs = [j for j in range(n_jobs) if next_op[j] < len(durations[j])]
        # Select job: prioritize operation with fewest machine alternatives (constrained operation)
        # Tie-break by earliest job ready time
        def job_key(j):
            op_idx = next_op[j]
            n_alternatives = len(durations[j][op_idx])
            return (n_alternatives, job_ready_time[j])
        job_idx = min(ready_jobs, key=job_key)

        op_idx = next_op[job_idx]
        candidates = durations[job_idx][op_idx]
        job_ready = job_ready_time[job_idx]

        # Choose machine: minimize completion time with penalty for high static load
        def machine_key(pair):
            m, t = pair
            start = max(machine_ready_time[m], job_ready)
            # penalty proportional to machine load (static)
            penalty = load_weight * machine_load[m] / avg_load if avg_load > 0 else 0
            return (start + t + penalty, t, m)
        machine_id, processing_time = min(candidates, key=machine_key)

        start_time = max(machine_ready_time[machine_id], job_ready)
        schedule[job_idx][op_idx] = (machine_id, start_time, processing_time)

        end_time = start_time + processing_time
        machine_ready_time[machine_id] = end_time
        job_ready_time[job_idx] = end_time
        next_op[job_idx] += 1
        remaining -= 1

    return schedule
