def fjsp_solver(n_jobs, n_machines, durations):
    # Precompute total demand for each machine across all operations
    machine_demand = [0] * n_machines
    for job in durations:
        for op in job:
            for m, t in op:
                machine_demand[m] += t

    # Initialize scheduling structures
    schedule = [[None] * len(durations[j]) for j in range(n_jobs)]
    machine_ready_time = [0] * n_machines
    job_ready_time = [0] * n_jobs
    next_op = [0] * n_jobs
    remaining = sum(len(job) for job in durations)

    while remaining > 0:
        # Get all jobs that still have unscheduled operations
        ready_jobs = [j for j in range(n_jobs) if next_op[j] < len(durations[j])]

        # Pick the job whose next operation has the fewest candidate machines
        # (most constrained operation), breaking ties by earliest job ready time
        job_idx = min(
            ready_jobs,
            key=lambda j: (len(durations[j][next_op[j]]), job_ready_time[j])
        )

        op_idx = next_op[job_idx]
        candidates = durations[job_idx][op_idx]

        # Choose (machine, processing_time) that minimizes:
        #  1) estimated completion time = max(machine_ready, job_ready) + processing_time
        #  2) processing_time itself
        #  3) global machine demand (lower is better)
        machine_id, processing_time = min(
            candidates,
            key=lambda pair: (
                max(machine_ready_time[pair[0]], job_ready_time[job_idx]) + pair[1],
                pair[1],
                machine_demand[pair[0]]
            )
        )

        start_time = max(machine_ready_time[machine_id], job_ready_time[job_idx])
        end_time = start_time + processing_time

        # Record scheduling decision
        schedule[job_idx][op_idx] = (machine_id, start_time, processing_time)

        # Advance timers
        machine_ready_time[machine_id] = end_time
        job_ready_time[job_idx] = end_time
        next_op[job_idx] += 1
        remaining -= 1

    return schedule
